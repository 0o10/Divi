18 uint32_t ParseChecksum(uint256 nChecksum, CoinDenomination denomination)
19 {
20  //shift to the beginning bit of this denomination and trim any remaining bits by returning 32 bits only
21  int pos = distance(zerocoinDenomList.begin(), find(zerocoinDenomList.begin(), zerocoinDenomList.end(), denomination));
22  nChecksum = nChecksum >> (32*((zerocoinDenomList.size() - 1) - pos));
23  return nChecksum.Get32();
24 }

119 bool EraseCheckpoints(int nStartHeight, int nEndHeight)
120 {
121  if (chainActive.Height() < nStartHeight)
122  return false;
123 
124  nEndHeight = min(chainActive.Height(), nEndHeight);
125 
126  CBlockIndex* pindex = chainActive[nStartHeight];
127  uint256 nCheckpointPrev = pindex->pprev->nAccumulatorCheckpoint;
128 
129  //Keep a list of checkpoints from the previous block so that we don't delete them
130  list<uint32_t> listCheckpointsPrev;
131  for (auto denom : zerocoinDenomList)
132  listCheckpointsPrev.emplace_back(ParseChecksum(nCheckpointPrev, denom));
133 
134  while (true) {
135  uint256 nCheckpointDelete = pindex->nAccumulatorCheckpoint;
136 
137  for (auto denom : zerocoinDenomList) {
138  uint32_t nChecksumDelete = ParseChecksum(nCheckpointDelete, denom);
139  if (count(listCheckpointsPrev.begin(), listCheckpointsPrev.end(), nCheckpointDelete))
140  continue;
141  EraseChecksum(nChecksumDelete);
142  }
143  LogPrintf("%s : erasing checksums for block %d\n", __func__, pindex->nHeight);
144 
145  if (pindex->nHeight + 1 <= nEndHeight)
146  pindex = chainActive.Next(pindex);
147  else
148  break;
149  }
150 
151  return true;
152 }

155 bool CalculateAccumulatorCheckpoint(int nHeight, uint256& nCheckpoint)
156 {
157  if (nHeight < Params().Zerocoin_StartHeight()) {
158  nCheckpoint = 0;
159  return true;
160  }
161 
162  //the checkpoint is updated every ten blocks, return current active checkpoint if not update block
163  if (nHeight % 10 != 0) {
164  nCheckpoint = chainActive[nHeight - 1]->nAccumulatorCheckpoint;
165  return true;
166  }
167 
168  //set the accumulators to last checkpoint value
169  AccumulatorMap mapAccumulators;
170  if (!mapAccumulators.Load(chainActive[nHeight - 1]->nAccumulatorCheckpoint)) {
171  if (chainActive[nHeight - 1]->nAccumulatorCheckpoint == 0) {
172  //Before zerocoin is fully activated so set to init state
173  mapAccumulators.Reset();
174  } else {
175  LogPrintf("%s: failed to reset to previous checkpoint\n", __func__);
176  return false;
177  }
178  }
179 
180  //Whether this should filter out invalid/fraudulent outpoints
181  bool fFilterInvalid = nHeight >= Params().Zerocoin_Block_RecalculateAccumulators();
182 
183  //Accumulate all coins over the last ten blocks that havent been accumulated (height - 20 through height - 11)
184  int nTotalMintsFound = 0;
185  CBlockIndex *pindex = chainActive[nHeight - 20];
186 
187  //On a specific block, a recalculation of the accumulators will be forced
188  if (nHeight == Params().Zerocoin_Block_RecalculateAccumulators()) {
189  pindex = chainActive[Params().Zerocoin_Block_LastGoodCheckpoint() - 10];
190  mapAccumulators.Reset();
191  if (!mapAccumulators.Load(chainActive[Params().Zerocoin_Block_LastGoodCheckpoint()]->nAccumulatorCheckpoint)) {
192  LogPrintf("%s: failed to reset to previous checkpoint when recalculating accumulators\n", __func__);
193  return false;
194  }
195  LogPrintf("*** %s recalculating checkpoint\n", __func__);
196 
197  // Erase the checkpoints from the period of time that bad mints were being made
198  if (!EraseCheckpoints(Params().Zerocoin_Block_LastGoodCheckpoint() + 1, nHeight)) {
199  LogPrintf("%s : failed to erase Checkpoints while recalculating checkpoints\n", __func__);
200  return false;
201  }
202  }
203 
204  while (pindex->nHeight < nHeight - 10) {
205  // checking whether we should stop this process due to a shutdown request
206  if (ShutdownRequested()) {
207  return false;
208  }
209 
210  //make sure this block is eligible for accumulation
211  if (pindex->nHeight < Params().Zerocoin_StartHeight()) {
212  pindex = chainActive[pindex->nHeight + 1];
213  continue;
214  }
215 
216  //grab mints from this block
217  CBlock block;
218  if(!ReadBlockFromDisk(block, pindex)) {
219  LogPrint("zero","%s: failed to read block from disk\n", __func__);
220  return false;
221  }
222 
223  std::list<PublicCoin> listPubcoins;
224  if (!BlockToPubcoinList(block, listPubcoins, fFilterInvalid)) {
225  LogPrint("zero","%s: failed to get zerocoin mintlist from block %n\n", __func__, pindex->nHeight);
226  return false;
227  }
228 
229  nTotalMintsFound += listPubcoins.size();
230  LogPrint("zero", "%s found %d mints\n", __func__, listPubcoins.size());
231 
232  //add the pubcoins to accumulator
233  for (const PublicCoin pubcoin : listPubcoins) {
234  if(!mapAccumulators.Accumulate(pubcoin, true)) {
235  LogPrintf("%s: failed to add pubcoin to accumulator at height %n\n", __func__, pindex->nHeight);
236  return false;
237  }
238  }
239  pindex = chainActive.Next(pindex);
240  }
241 
242  // if there were no new mints found, the accumulator checkpoint will be the same as the last checkpoint
243  if (nTotalMintsFound == 0) {
244  nCheckpoint = chainActive[nHeight - 1]->nAccumulatorCheckpoint;
245  }
246  else
247  nCheckpoint = mapAccumulators.GetCheckpoint();
248 
249  // make sure that these values are databased because reorgs may have deleted the checksums from DB
250  DatabaseChecksums(mapAccumulators);
251 
252  LogPrint("zero", "%s checkpoint=%s\n", __func__, nCheckpoint.GetHex());
253  return true;
254 }

261 bool GenerateAccumulatorWitness(const PublicCoin &coin, Accumulator& accumulator, AccumulatorWitness& witness, int nSecurityLevel, int& nMintsAdded, string& strError)
262 {
263  uint256 txid;
264  if (!zerocoinDB->ReadCoinMint(coin.getValue(), txid)) {
265  LogPrint("zero","%s failed to read mint from db\n", __func__);
266  return false;
267  }
268 
269  CTransaction txMinted;
270  uint256 hashBlock;
271  if (!GetTransaction(txid, txMinted, hashBlock)) {
272  LogPrint("zero","%s failed to read tx\n", __func__);
273  return false;
274  }
275 
276  int nHeightMintAdded= mapBlockIndex[hashBlock]->nHeight;
277  uint256 nCheckpointBeforeMint = 0;
278  CBlockIndex* pindex = chainActive[nHeightMintAdded];
279  int nChanges = 0;
280 
281  //find the checksum when this was added to the accumulator officially, which will be two checksum changes later
282  //reminder that checksums are generated when the block height is a multiple of 10
283  while (pindex->nHeight < chainActive.Tip()->nHeight - 1) {
284  if (pindex->nHeight == nHeightMintAdded) {
285  pindex = chainActive[pindex->nHeight + 1];
286  continue;
287  }
288 
289  //check if the next checksum was generated
290  if (pindex->nHeight % 10 == 0) {
291  nChanges++;
292 
293  if (nChanges == 1) {
294  nCheckpointBeforeMint = pindex->nAccumulatorCheckpoint;
295  break;
296  }
297  }
298  pindex = chainActive.Next(pindex);
299  }
300 
301  //the height to start accumulating coins to add to witness
302  int nAccStartHeight = nHeightMintAdded - (nHeightMintAdded % 10);
303 
304  //If the checkpoint is from the recalculated checkpoint period, then adjust it
305  int nHeight_LastGoodCheckpoint = Params().Zerocoin_Block_LastGoodCheckpoint();
306  int nHeight_Recalculate = Params().Zerocoin_Block_RecalculateAccumulators();
307  if (pindex->nHeight < nHeight_Recalculate - 10 && pindex->nHeight > nHeight_LastGoodCheckpoint) {
308  //The checkpoint before the mint will be the last good checkpoint
309  nCheckpointBeforeMint = chainActive[nHeight_LastGoodCheckpoint]->nAccumulatorCheckpoint;
310  nAccStartHeight = nHeight_LastGoodCheckpoint - 10;
311  }
312 
313  //Get the accumulator that is right before the cluster of blocks containing our mint was added to the accumulator
314  CBigNum bnAccValue = 0;
315  if (GetAccumulatorValueFromDB(nCheckpointBeforeMint, coin.getDenomination(), bnAccValue)) {
316  if (bnAccValue > 0) {
317  accumulator.setValue(bnAccValue);
318  witness.resetValue(accumulator, coin);
319  }
320  }
321 
322  //security level: this is an important prevention of tracing the coins via timing. Security level represents how many checkpoints
323  //of accumulated coins are added *beyond* the checkpoint that the mint being spent was added too. If each spend added the exact same
324  //amounts of checkpoints after the mint was accumulated, then you could know the range of blocks that the mint originated from.
325  if (nSecurityLevel < 100) {
326  //add some randomness to the user's selection so that it is not always the same
327  nSecurityLevel += CBigNum::randBignum(10).getint();
328 
329  //security level 100 represents adding all available coins that have been accumulated - user did not select this
330  if (nSecurityLevel >= 100)
331  nSecurityLevel = 99;
332  }
333 
334  //add the pubcoins (zerocoinmints that have been published to the chain) up to the next checksum starting from the block
335  pindex = chainActive[nAccStartHeight];
336  int nChainHeight = chainActive.Height();
337  int nHeightStop = nChainHeight % 10;
338  nHeightStop = nChainHeight - nHeightStop - 20; // at least two checkpoints deep
339  int nCheckpointsAdded = 0;
340  nMintsAdded = 0;
341  while (pindex->nHeight < nHeightStop + 1) {
342  if (pindex->nHeight != nAccStartHeight && pindex->pprev->nAccumulatorCheckpoint != pindex->nAccumulatorCheckpoint)
343  ++nCheckpointsAdded;
344 
345  //if a new checkpoint was generated on this block, and we have added the specified amount of checkpointed accumulators,
346  //then initialize the accumulator at this point and break
347  if (!InvalidCheckpointRange(pindex->nHeight) && (pindex->nHeight >= nHeightStop || (nSecurityLevel != 100 && nCheckpointsAdded >= nSecurityLevel))) {
348  uint32_t nChecksum = ParseChecksum(chainActive[pindex->nHeight + 10]->nAccumulatorCheckpoint, coin.getDenomination());
349  CBigNum bnAccValue = 0;
350  if (!zerocoinDB->ReadAccumulatorValue(nChecksum, bnAccValue)) {
351  LogPrintf("%s : failed to find checksum in database for accumulator\n", __func__);
352  return false;
353  }
354  accumulator.setValue(bnAccValue);
355  break;
356  }
357 
358  // if this block contains mints of the denomination that is being spent, then add them to the witness
359  if (pindex->MintedDenomination(coin.getDenomination())) {
360  //grab mints from this block
361  CBlock block;
362  if(!ReadBlockFromDisk(block, pindex)) {
363  LogPrintf("%s: failed to read block from disk while adding pubcoins to witness\n", __func__);
364  return false;
365  }
366 
367  list<PublicCoin> listPubcoins;
368  if(!BlockToPubcoinList(block, listPubcoins, true)) {
369  LogPrintf("%s: failed to get zerocoin mintlist from block %n\n", __func__, pindex->nHeight);
370  return false;
371  }
372 
373  //add the mints to the witness
374  for (const PublicCoin pubcoin : listPubcoins) {
375  if (pubcoin.getDenomination() != coin.getDenomination())
376  continue;
377 
378  if (pindex->nHeight == nHeightMintAdded && pubcoin.getValue() == coin.getValue())
379  continue;
380 
381  witness.addRawValue(pubcoin.getValue());
382  ++nMintsAdded;
383  }
384  }
385 
386  pindex = chainActive[pindex->nHeight + 1];
387  }
388 
389  if (nMintsAdded < Params().Zerocoin_RequiredAccumulation()) {
390  strError = _(strprintf("Less than %d mints added, unable to create spend", Params().Zerocoin_RequiredAccumulation()).c_str());
391  LogPrintf("%s : %s\n", __func__, strError);
392  return false;
393  }
394 
395  // calculate how many mints of this denomination existed in the accumulator we initialized
396  int nZerocoinStartHeight = GetZerocoinStartHeight();
397  pindex = chainActive[nZerocoinStartHeight];
398  while (pindex->nHeight < nAccStartHeight) {
399  nMintsAdded += count(pindex->vMintDenominationsInBlock.begin(), pindex->vMintDenominationsInBlock.end(), coin.getDenomination());
400  pindex = chainActive[pindex->nHeight + 1];
401  }
402 
403  LogPrint("zero","%s : %d mints added to witness\n", __func__, nMintsAdded);
404  return true;
405 }
