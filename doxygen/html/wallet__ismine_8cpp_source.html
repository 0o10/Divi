38 isminetype IsMine(const CKeyStore& keystore, const CScript& scriptPubKey)
39 {
40  if(keystore.HaveWatchOnly(scriptPubKey))
41  return ISMINE_WATCH_ONLY;
42  if(keystore.HaveMultiSig(scriptPubKey))
43  return ISMINE_MULTISIG;
44 
45  vector<valtype> vSolutions;
46  txnouttype whichType;
47  if(!Solver(scriptPubKey, whichType, vSolutions)) {
48  if(keystore.HaveWatchOnly(scriptPubKey))
49  return ISMINE_WATCH_ONLY;
50  if(keystore.HaveMultiSig(scriptPubKey))
51  return ISMINE_MULTISIG;
52 
53  return ISMINE_NO;
54  }
55 
56  CKeyID keyID;
57  switch (whichType) {
58  case TX_NONSTANDARD:
59  case TX_NULL_DATA:
60  break;
61  case TX_ZEROCOINMINT:
62  case TX_PUBKEY:
63  keyID = CPubKey(vSolutions[0]).GetID();
64  if(keystore.HaveKey(keyID))
65  return ISMINE_SPENDABLE;
66  break;
67  case TX_PUBKEYHASH:
68  keyID = CKeyID(uint160(vSolutions[0]));
69  if(keystore.HaveKey(keyID))
70  return ISMINE_SPENDABLE;
71  break;
72  case TX_SCRIPTHASH: {
73  CScriptID scriptID = CScriptID(uint160(vSolutions[0]));
74  CScript subscript;
75  if(keystore.GetCScript(scriptID, subscript)) {
76  isminetype ret = IsMine(keystore, subscript);
77  if(ret != ISMINE_NO)
78  return ret;
79  }
80  break;
81  }
82  case TX_MULTISIG: {
83  // Only consider transactions "mine" if we own ALL the
84  // keys involved. multi-signature transactions that are
85  // partially owned (somebody else has a key that can spend
86  // them) enable spend-out-from-under-you attacks, especially
87  // in shared-wallet situations.
88  vector<valtype> keys(vSolutions.begin() + 1, vSolutions.begin() + vSolutions.size() - 1);
89  if(HaveKeys(keys, keystore) == keys.size())
90  return ISMINE_SPENDABLE;
91  break;
92  }
93  }
94 
95  if(keystore.HaveWatchOnly(scriptPubKey))
96  return ISMINE_WATCH_ONLY;
97  if(keystore.HaveMultiSig(scriptPubKey))
98  return ISMINE_MULTISIG;
99 
100  return ISMINE_NO;
101 }
