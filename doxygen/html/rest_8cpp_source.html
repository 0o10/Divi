92 static bool rest_block(AcceptedConnection* conn,
93  string& strReq,
94  map<string, string>& mapHeaders,
95  bool fRun,
96  bool showTxDetails)
97 {
98  vector<string> params;
99  enum RetFormat rf = ParseDataFormat(params, strReq);
100 
101  string hashStr = params[0];
102  uint256 hash;
103  if (!ParseHashStr(hashStr, hash))
104  throw RESTERR(HTTP_BAD_REQUEST, "Invalid hash: " + hashStr);
105 
106  CBlock block;
107  CBlockIndex* pblockindex = NULL;
108  {
109  LOCK(cs_main);
110  if (mapBlockIndex.count(hash) == 0)
111  throw RESTERR(HTTP_NOT_FOUND, hashStr + " not found");
112 
113  pblockindex = mapBlockIndex[hash];
114  if (!ReadBlockFromDisk(block, pblockindex))
115  throw RESTERR(HTTP_NOT_FOUND, hashStr + " not found");
116  }
117 
118  CDataStream ssBlock(SER_NETWORK, PROTOCOL_VERSION);
119  ssBlock << block;
120 
121  switch (rf) {
122  case RF_BINARY: {
123  string binaryBlock = ssBlock.str();
124  conn->stream() << HTTPReplyHeader(HTTP_OK, fRun, binaryBlock.size(), "application/octet-stream") << binaryBlock << std::flush;
125  return true;
126  }
127 
128  case RF_HEX: {
129  string strHex = HexStr(ssBlock.begin(), ssBlock.end()) + "\n";
130  conn->stream() << HTTPReply(HTTP_OK, strHex, fRun, false, "text/plain") << std::flush;
131  return true;
132  }
133 
134  case RF_JSON: {
135  Object objBlock = blockToJSON(block, pblockindex, showTxDetails);
136  string strJSON = write_string(Value(objBlock), false) + "\n";
137  conn->stream() << HTTPReply(HTTP_OK, strJSON, fRun) << std::flush;
138  return true;
139  }
140 
141  default: {
142  throw RESTERR(HTTP_NOT_FOUND, "output format not found (available: " + AvailableDataFormatsString() + ")");
143  }
144  }
145 
146  // not reached
147  return true; // continue to process further HTTP reqs on this cxn
148 }

241  if (strURI.substr(0, plen) == uri_prefixes[i].prefix) {
242  string strReq = strURI.substr(plen);

244  }
245  }

249  }
