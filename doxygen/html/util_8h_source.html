215 template <typename Callable>
216 void LoopForever(const char* name, Callable func, int64_t msecs)
217 {
218  std::string s = strprintf("divi-%s", name);
219  RenameThread(s.c_str());
220  LogPrintf("%s thread start\n", name);
221  try {
222  while (1) {
223  MilliSleep(msecs);
224  func();
225  }
226  } catch (boost::thread_interrupted) {
227  LogPrintf("%s thread stop\n", name);
228  throw;
229  } catch (std::exception& e) {
230  PrintExceptionContinue(&e, name);
231  throw;
232  } catch (...) {
233  PrintExceptionContinue(NULL, name);
234  throw;
235  }
236 }
