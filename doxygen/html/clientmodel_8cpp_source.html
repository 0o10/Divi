117 void ClientModel::updateTimer()
118 {
119  // Get required lock upfront. This avoids the GUI from getting stuck on
120  // periodical polls if the core is holding the locks for a longer time -
121  // for example, during a wallet rescan.
122  TRY_LOCK(cs_main, lockMain);
123  if (!lockMain)
124  return;
125  // Some quantities (such as number of blocks) change so fast that we don't want to be notified for each change.
126  // Periodically check and update with a timer.
127  int newNumBlocks = getNumBlocks();
128 
129  static int prevAttempt = -1;
130  static int prevAssets = -1;
131 
132  // check for changed number of blocks we have, number of blocks peers claim to have, reindexing state and importing state
133  if (cachedNumBlocks != newNumBlocks ||
134  cachedReindexing != fReindex || cachedImporting != fImporting ||
135  masternodeSync.RequestedMasternodeAttempt != prevAttempt || masternodeSync.RequestedMasternodeAssets != prevAssets) {
136  cachedNumBlocks = newNumBlocks;
137  cachedReindexing = fReindex;
138  cachedImporting = fImporting;
139  prevAttempt = masternodeSync.RequestedMasternodeAttempt;
140  prevAssets = masternodeSync.RequestedMasternodeAssets;
141 
142  emit numBlocksChanged(newNumBlocks);
143  }
144 
145  emit bytesChanged(getTotalBytesRecv(), getTotalBytesSent());
146 }

170 void ClientModel::updateAlert(const QString& hash, int status)
171 {
172  // Show error message notification for new alert
173  if (status == CT_NEW) {
174  uint256 hash_256;
175  hash_256.SetHex(hash.toStdString());

177  if (!alert.IsNull()) {
178  emit message(tr("Network Alert"), QString::fromStdString(alert.strStatusBar), CClientUIInterface::ICON_ERROR);
