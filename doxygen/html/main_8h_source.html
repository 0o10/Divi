562  {
563  chRejectCode = chRejectCodeIn;
564  strRejectReason = strRejectReasonIn;
565  corruptionPossible = corruptionIn;

567  return ret;

570  return ret;

573  unsigned char _chRejectCode = 0,

576  return DoS(0, ret, _chRejectCode, _strRejectReason);

580  if (mode == MODE_VALID)
581  strRejectReason = strRejectReasonIn;
582  mode = MODE_ERROR;
583  return false;
584  }

587  AbortNode(msg);
588  return Error(msg);
589  }

593  }
594  bool IsInvalid() const
595  {
596  return mode == MODE_INVALID;
597  }

600  return mode == MODE_ERROR;
601  }

603  {
604  if (IsInvalid()) {

613  }
