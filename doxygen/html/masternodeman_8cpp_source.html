88 CMasternodeDB::ReadResult CMasternodeDB::Read(CMasternodeMan& mnodemanToLoad, bool fDryRun)
89 {
90  int64_t nStart = GetTimeMillis();
91  // open input file, and associate with CAutoFile
92  FILE* file = fopen(pathMN.string().c_str(), "rb");
93  CAutoFile filein(file, SER_DISK, CLIENT_VERSION);
94  if (filein.IsNull()) {
95  error("%s : Failed to open file %s", __func__, pathMN.string());
96  return FileError;
97  }
98 
99  // use file size to size memory buffer
100  int fileSize = boost::filesystem::file_size(pathMN);
101  int dataSize = fileSize - sizeof(uint256);
102  // Don't try to resize to a negative number if file is small
103  if (dataSize < 0)
104  dataSize = 0;
105  vector<unsigned char> vchData;
106  vchData.resize(dataSize);
107  uint256 hashIn;
108 
109  // read data and checksum from file
110  try {
111  filein.read((char*)&vchData[0], dataSize);
112  filein >> hashIn;
113  } catch (std::exception& e) {
114  error("%s : Deserialize or I/O error - %s", __func__, e.what());
115  return HashReadError;
116  }
117  filein.fclose();
118 
119  CDataStream ssMasternodes(vchData, SER_DISK, CLIENT_VERSION);
120 
121  // verify stored checksum matches input data
122  uint256 hashTmp = Hash(ssMasternodes.begin(), ssMasternodes.end());
123  if (hashIn != hashTmp) {
124  error("%s : Checksum mismatch, data corrupted", __func__);
125  return IncorrectHash;
126  }
127 
128  unsigned char pchMsgTmp[4];
129  std::string strMagicMessageTmp;
130  try {
131  // de-serialize file header (masternode cache file specific magic message) and ..
132 
133  ssMasternodes >> strMagicMessageTmp;
134 
135  // ... verify the message matches predefined one
136  if (strMagicMessage != strMagicMessageTmp) {
137  error("%s : Invalid masternode cache magic message", __func__);
138  return IncorrectMagicMessage;
139  }
140 
141  // de-serialize file header (network specific magic number) and ..
142  ssMasternodes >> FLATDATA(pchMsgTmp);
143 
144  // ... verify the network matches ours
145  if (memcmp(pchMsgTmp, Params().MessageStart(), sizeof(pchMsgTmp))) {
146  error("%s : Invalid network magic number", __func__);
147  return IncorrectMagicNumber;
148  }
149  // de-serialize data into CMasternodeMan object
150  ssMasternodes >> mnodemanToLoad;
151  } catch (std::exception& e) {
152  mnodemanToLoad.Clear();
153  error("%s : Deserialize or I/O error - %s", __func__, e.what());
154  return IncorrectFormat;
155  }
156 
157  LogPrint("masternode","Loaded info from mncache.dat %dms\n", GetTimeMillis() - nStart);
158  LogPrint("masternode"," %s\n", mnodemanToLoad.ToString());
159  if (!fDryRun) {
160  LogPrint("masternode","Masternode manager - cleaning....\n");
161  mnodemanToLoad.CheckAndRemove(true);
162  LogPrint("masternode","Masternode manager - result:\n");
163  LogPrint("masternode"," %s\n", mnodemanToLoad.ToString());
164  }
165 
166  return Ok;
167 }

243 void CMasternodeMan::CheckAndRemove(bool forceExpiredRemoval)
244 {
245  Check();
246 
247  LOCK(cs);
248 
249  //remove inactive and outdated
250  vector<CMasternode>::iterator it = vMasternodes.begin();
251  while (it != vMasternodes.end()) {
252  if ((*it).activeState == CMasternode::MASTERNODE_REMOVE ||
253  (*it).activeState == CMasternode::MASTERNODE_VIN_SPENT ||
254  (forceExpiredRemoval && (*it).activeState == CMasternode::MASTERNODE_EXPIRED) ||
255  (*it).protocolVersion < masternodePayments.GetMinMasternodePaymentsProto()) {
256  LogPrint("masternode", "CMasternodeMan: Removing inactive Masternode %s - %i now\n", (*it).vin.prevout.hash.ToString(), size() - 1);
257 
258  //erase all of the broadcasts we've seen from this vin
259  // -- if we missed a few pings and the node was removed, this will allow is to get it back without them
260  // sending a brand new mnb
261  map<uint256, CMasternodeBroadcast>::iterator it3 = mapSeenMasternodeBroadcast.begin();
262  while (it3 != mapSeenMasternodeBroadcast.end()) {
263  if ((*it3).second.vin == (*it).vin) {
264  masternodeSync.mapSeenSyncMNB.erase((*it3).first);
265  mapSeenMasternodeBroadcast.erase(it3++);
266  } else {
267  ++it3;
268  }
269  }
270 
271  // allow us to ask for this masternode again if we see another ping
272  map<COutPoint, int64_t>::iterator it2 = mWeAskedForMasternodeListEntry.begin();
273  while (it2 != mWeAskedForMasternodeListEntry.end()) {
274  if ((*it2).first == (*it).vin.prevout) {
275  mWeAskedForMasternodeListEntry.erase(it2++);
276  } else {
277  ++it2;
278  }
279  }
280 
281  it = vMasternodes.erase(it);
282  } else {
283  ++it;
284  }
285  }
286 
287  // check who's asked for the Masternode list
288  map<CNetAddr, int64_t>::iterator it1 = mAskedUsForMasternodeList.begin();
289  while (it1 != mAskedUsForMasternodeList.end()) {
290  if ((*it1).second < GetTime()) {
291  mAskedUsForMasternodeList.erase(it1++);
292  } else {
293  ++it1;
294  }
295  }
296 
297  // check who we asked for the Masternode list
298  it1 = mWeAskedForMasternodeList.begin();
299  while (it1 != mWeAskedForMasternodeList.end()) {
300  if ((*it1).second < GetTime()) {
301  mWeAskedForMasternodeList.erase(it1++);
302  } else {
303  ++it1;
304  }
305  }
306 
307  // check which Masternodes we've asked for
308  map<COutPoint, int64_t>::iterator it2 = mWeAskedForMasternodeListEntry.begin();
309  while (it2 != mWeAskedForMasternodeListEntry.end()) {
310  if ((*it2).second < GetTime()) {
311  mWeAskedForMasternodeListEntry.erase(it2++);
312  } else {
313  ++it2;
314  }
315  }
316 
317  // remove expired mapSeenMasternodeBroadcast
318  map<uint256, CMasternodeBroadcast>::iterator it3 = mapSeenMasternodeBroadcast.begin();

477 CMasternode* CMasternodeMan::GetNextMasternodeInQueueForPayment(int nBlockHeight, bool fFilterSigTime, int& nCount)
478 {
479  LOCK(cs);
480 
481  CMasternode* pBestMasternode = NULL;
482  std::vector<pair<int64_t, CTxIn> > vecMasternodeLastPaid;
483 
484  /*
485  Make a vector with all of the last paid times
486  */
487 
488  int nMnCount = CountEnabled();
489  BOOST_FOREACH (CMasternode& mn, vMasternodes) {
490  mn.Check();
491  if (!mn.IsEnabled()) continue;
492 
493  // //check protocol version
494  if (mn.protocolVersion < masternodePayments.GetMinMasternodePaymentsProto()) continue;
495 
496  //it's in the list (up to 8 entries ahead of current block to allow propagation) -- so let's skip it
497  if (masternodePayments.IsScheduled(mn, nBlockHeight)) continue;
498 
499  //it's too new, wait for a cycle
500  if (fFilterSigTime && mn.sigTime + (nMnCount * 2.6 * 60) > GetAdjustedTime()) continue;
501 
502  //make sure it has as many confirmations as there are masternodes
503  if (mn.GetMasternodeInputAge() < nMnCount) continue;
504 
505  vecMasternodeLastPaid.push_back(make_pair(mn.SecondsSincePayment(), mn.vin));
506  }
507 
508  nCount = (int)vecMasternodeLastPaid.size();
509 
510  //when the network is in the process of upgrading, don't penalize nodes that recently restarted
511  if (fFilterSigTime && nCount < nMnCount / 3) return GetNextMasternodeInQueueForPayment(nBlockHeight, false, nCount);
512 
513  // Sort them high to low
514  sort(vecMasternodeLastPaid.rbegin(), vecMasternodeLastPaid.rend(), CompareLastPaid());
515 
516  // Look at 1/10 of the oldest nodes (by last payment), calculate their scores and pay the best one
517  // -- This doesn't look at who is being paid in the +8-10 blocks, allowing for double payments very rarely
518  // -- 1/100 payments should be a double payment on mainnet - (1/(3000/10))*2
519  // -- (chance per block * chances before IsScheduled will fire)
520  int nTenthNetwork = CountEnabled() / 10;
521  int nCountTenth = 0;
522  uint256 nHigh = 0;
523  BOOST_FOREACH (PAIRTYPE(int64_t, CTxIn) & s, vecMasternodeLastPaid) {
524  CMasternode* pmn = Find(s.second);
525  if (!pmn) break;
526 
527  uint256 n = pmn->CalculateScore(1, nBlockHeight - 100);
528  if (n > nHigh) {
529  nHigh = n;
530  pBestMasternode = pmn;
531  }
532  nCountTenth++;
533  if (nCountTenth >= nTenthNetwork) break;
534  }
535  return pBestMasternode;
536 }

594 int CMasternodeMan::GetMasternodeRank(const CTxIn& vin, int64_t nBlockHeight, int minProtocol, bool fOnlyActive)
595 {
596  std::vector<pair<int64_t, CTxIn> > vecMasternodeScores;
597  int64_t nMasternode_Min_Age = MN_WINNER_MINIMUM_AGE;
598  int64_t nMasternode_Age = 0;
599 
600  //make sure we know about this block
601  uint256 hash = 0;
602  if (!GetBlockHash(hash, nBlockHeight)) return -1;
603 
604  // scan for winner
605  BOOST_FOREACH (CMasternode& mn, vMasternodes) {
606  if (mn.protocolVersion < minProtocol) {
607  LogPrint("masternode","Skipping Masternode with obsolete version %d\n", mn.protocolVersion);
608  continue; // Skip obsolete versions
609  }
610 
611  if (IsSporkActive(SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT)) {
612  nMasternode_Age = GetAdjustedTime() - mn.sigTime;
613  if ((nMasternode_Age) < nMasternode_Min_Age) {
614  if (fDebug) LogPrint("masternode","Skipping just activated Masternode. Age: %ld\n", nMasternode_Age);
615  continue; // Skip masternodes younger than (default) 1 hour
616  }
617  }
618  if (fOnlyActive) {
619  mn.Check();
620  if (!mn.IsEnabled()) continue;
621  }
622  uint256 n = mn.CalculateScore(1, nBlockHeight);
623  int64_t n2 = n.GetCompact(false);
624 
625  vecMasternodeScores.push_back(make_pair(n2, mn.vin));
626  }
627 
628  sort(vecMasternodeScores.rbegin(), vecMasternodeScores.rend(), CompareScoreTxIn());
629
