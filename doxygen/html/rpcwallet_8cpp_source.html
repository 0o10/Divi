48 void WalletTxToJSON(const CWalletTx& wtx, Object& entry)
49 {
50  int confirms = wtx.GetDepthInMainChain(false);
51  int confirmsTotal = GetIXConfirmations(wtx.GetHash()) + confirms;
52  entry.push_back(Pair("confirmations", confirmsTotal));
53  entry.push_back(Pair("bcconfirmations", confirms));
54  if (wtx.IsCoinBase() || wtx.IsCoinStake())
55  entry.push_back(Pair("generated", true));
56  if (confirms > 0) {
57  entry.push_back(Pair("blockhash", wtx.hashBlock.GetHex()));
58  entry.push_back(Pair("blockindex", wtx.nIndex));
59  entry.push_back(Pair("blocktime", mapBlockIndex[wtx.hashBlock]->GetBlockTime()));
60  }
61  uint256 hash = wtx.GetHash();
62  entry.push_back(Pair("txid", hash.GetHex()));
63  Array conflicts;
64  BOOST_FOREACH (const uint256& conflict, wtx.GetConflicts())
65  conflicts.push_back(conflict.GetHex());
66  entry.push_back(Pair("walletconflicts", conflicts));
67  entry.push_back(Pair("time", wtx.GetTxTime()));
68  entry.push_back(Pair("timereceived", (int64_t)wtx.nTimeReceived));
69  BOOST_FOREACH (const PAIRTYPE(string, string) & item, wtx.mapValue)
70  entry.push_back(Pair(item.first, item.second));
71 }

116 CBitcoinAddress GetAccountAddress(string strAccount, bool bForceNew = false)
117 {
118  CWalletDB walletdb(pwalletMain->strWalletFile);
119 
120  CAccount account;
121  walletdb.ReadAccount(strAccount, account);
122 
123  bool bKeyUsed = false;
124 
125  // Check if the current key has been used
126  if (account.vchPubKey.IsValid()) {
127  CScript scriptPubKey = GetScriptForDestination(account.vchPubKey.GetID());
128  for (map<uint256, CWalletTx>::iterator it = pwalletMain->mapWallet.begin();
129  it != pwalletMain->mapWallet.end() && account.vchPubKey.IsValid();
130  ++it) {
131  const CWalletTx& wtx = (*it).second;
132  BOOST_FOREACH (const CTxOut& txout, wtx.vout)
133  if (txout.scriptPubKey == scriptPubKey)
134  bKeyUsed = true;
135  }
136  }
137 
138  // Generate a new key
139  if (!account.vchPubKey.IsValid() || bForceNew || bKeyUsed) {
140  if (!pwalletMain->GetKeyFromPool(account.vchPubKey))
141  throw JSONRPCError(RPC_WALLET_KEYPOOL_RAN_OUT, "Error: Keypool ran out, please call keypoolrefill first");
142 
143  pwalletMain->SetAddressBook(account.vchPubKey.GetID(), strAccount, "receive");
144  walletdb.WriteAccount(strAccount, account);
145  }
146 
147  return CBitcoinAddress(account.vchPubKey.GetID());

293 void SendMoney(const CTxDestination& address, CAmount nValue, CWalletTx& wtxNew, bool fUseIX = false)
294 {
295  // Check amount
296  if (nValue <= 0)
297  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid amount");
298 
299  if (nValue > pwalletMain->GetBalance())
300  throw JSONRPCError(RPC_WALLET_INSUFFICIENT_FUNDS, "Insufficient funds");
301 
302  string strError;
303  if (pwalletMain->IsLocked()) {
304  strError = "Error: Wallet locked, unable to create transaction!";
305  LogPrintf("SendMoney() : %s", strError);
306  throw JSONRPCError(RPC_WALLET_ERROR, strError);
307  }
308 
309  // Parse DIVX address
310  CScript scriptPubKey = GetScriptForDestination(address);
311 
312  // Create and send the transaction
313  CReserveKey reservekey(pwalletMain);
314  CAmount nFeeRequired;
315  if (!pwalletMain->CreateTransaction(scriptPubKey, nValue, wtxNew, reservekey, nFeeRequired, strError, NULL, ALL_COINS, fUseIX, (CAmount)0)) {
316  if (nValue + nFeeRequired > pwalletMain->GetBalance())
317  strError = strprintf("Error: This transaction requires a transaction fee of at least %s because of its amount, complexity, or use of recently received funds!", FormatMoney(nFeeRequired));
318  LogPrintf("SendMoney() : %s\n", strError);
319  throw JSONRPCError(RPC_WALLET_ERROR, strError);
320  }
321  if (!pwalletMain->CommitTransaction(wtxNew, reservekey, (!fUseIX ? "tx" : "ix")))
322  throw JSONRPCError(RPC_WALLET_ERROR, "Error: The transaction was rejected! This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
323 }
