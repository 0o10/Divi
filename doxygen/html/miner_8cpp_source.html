38 //
39 // Unconfirmed transactions in the memory pool often depend on other
40 // transactions in the memory pool. When we select transactions from the
41 // pool, we select by highest priority or fee rate, so we might consider
42 // transactions that depend on transactions that aren't yet in the block.
43 // The COrphan class keeps track of these 'temporary orphans' while
44 // CreateBlock is figuring out which transactions to include.
45 //

50  set<uint256> setDependsOn;
51  CFeeRate feeRate;
52  double dPriority;

54  COrphan(const CTransaction* ptxIn) : ptx(ptxIn), feeRate(0), dPriority(0)
55  {
56  }

63 // We want to sort transactions by priority and fee rate, so:
64 typedef boost::tuple<double, CFeeRate, const CTransaction*> TxPriority;
65 class TxPriorityCompare
66 {
67  bool byFee;
68 
69 public:
70  TxPriorityCompare(bool _byFee) : byFee(_byFee) {}
71 
72  bool operator()(const TxPriority& a, const TxPriority& b)
73  {
74  if (byFee) {
75  if (a.get<1>() == b.get<1>())
76  return a.get<0>() < b.get<0>();
77  return a.get<1>() < b.get<1>();
78  } else {
79  if (a.get<0>() == b.get<0>())
80  return a.get<1>() < b.get<1>();
81  return a.get<0>() < b.get<0>();
82  }
83  }
84 };

95 CBlockTemplate* CreateNewBlock(const CScript& scriptPubKeyIn, CWallet* pwallet, bool fProofOfStake)
96 {
97  CReserveKey reservekey(pwallet);
98 
99  // Create new block
100  unique_ptr<CBlockTemplate> pblocktemplate(new CBlockTemplate());
101  if (!pblocktemplate.get())
102  return NULL;
103  CBlock* pblock = &pblocktemplate->block; // pointer for convenience
104 
105  // -regtest only: allow overriding block.nVersion with
106  // -blockversion=N to test forking scenarios
107  if (Params().MineBlocksOnDemand())
108  pblock->nVersion = GetArg("-blockversion", pblock->nVersion);
109 
110  // Make sure to create the correct block version after zerocoin is enabled
111  bool fZerocoinActive = GetAdjustedTime() >= Params().Zerocoin_StartTime();
112  if (fZerocoinActive)
113  pblock->nVersion = 4;
114  else
115  pblock->nVersion = 3;
116 
117  // Create coinbase tx
118  CMutableTransaction txNew;
119  txNew.vin.resize(1);
120  txNew.vin[0].prevout.SetNull();
121  txNew.vout.resize(1);
122  txNew.vout[0].scriptPubKey = scriptPubKeyIn;
123  pblock->vtx.push_back(txNew);
124  pblocktemplate->vTxFees.push_back(-1); // updated at end
125  pblocktemplate->vTxSigOps.push_back(-1); // updated at end
126 
127  // ppcoin: if coinstake available add coinstake tx
128  static int64_t nLastCoinStakeSearchTime = GetAdjustedTime(); // only initialized at startup
129 
130  if (fProofOfStake) {
131  boost::this_thread::interruption_point();
132  pblock->nTime = GetAdjustedTime();
133  CBlockIndex* pindexPrev = chainActive.Tip();
134  pblock->nBits = GetNextWorkRequired(pindexPrev, pblock);
135  CMutableTransaction txCoinStake;
136  int64_t nSearchTime = pblock->nTime; // search to current time
137  bool fStakeFound = false;
138  if (nSearchTime >= nLastCoinStakeSearchTime) {
139  unsigned int nTxNewTime = 0;
140  if (pwallet->CreateCoinStake(*pwallet, pblock->nBits, nSearchTime - nLastCoinStakeSearchTime, txCoinStake, nTxNewTime)) {
141  pblock->nTime = nTxNewTime;
142  pblock->vtx[0].vout[0].SetEmpty();
143  pblock->vtx.push_back(CTransaction(txCoinStake));
144  fStakeFound = true;
145  }
146  nLastCoinStakeSearchInterval = nSearchTime - nLastCoinStakeSearchTime;
147  nLastCoinStakeSearchTime = nSearchTime;
148  }
149 
150  if (!fStakeFound)
151  return NULL;
152  }
153 
154  // Largest block you're willing to create:
155  unsigned int nBlockMaxSize = GetArg("-blockmaxsize", DEFAULT_BLOCK_MAX_SIZE);
156  // Limit to betweeen 1K and MAX_BLOCK_SIZE-1K for sanity:
157  unsigned int nBlockMaxSizeNetwork = MAX_BLOCK_SIZE_CURRENT;
158  nBlockMaxSize = std::max((unsigned int)1000, std::min((nBlockMaxSizeNetwork - 1000), nBlockMaxSize));
159 
160  // How much of the block should be dedicated to high-priority transactions,
161  // included regardless of the fees they pay
162  unsigned int nBlockPrioritySize = GetArg("-blockprioritysize", DEFAULT_BLOCK_PRIORITY_SIZE);
163  nBlockPrioritySize = std::min(nBlockMaxSize, nBlockPrioritySize);
164 
165  // Minimum block size you want to create; block will be filled with free transactions
166  // until there are no more or the block reaches this size:
167  unsigned int nBlockMinSize = GetArg("-blockminsize", DEFAULT_BLOCK_MIN_SIZE);
168  nBlockMinSize = std::min(nBlockMaxSize, nBlockMinSize);
169 
170  // Collect memory pool transactions into the block
171  CAmount nFees = 0;
172 
173  {
174  LOCK2(cs_main, mempool.cs);
175 
176  CBlockIndex* pindexPrev = chainActive.Tip();
177  const int nHeight = pindexPrev->nHeight + 1;
178  CCoinsViewCache view(pcoinsTip);
179 
180  // Priority order to process transactions
181  list<COrphan> vOrphan; // list memory doesn't move
182  map<uint256, vector<COrphan*> > mapDependers;
183  bool fPrintPriority = GetBoolArg("-printpriority", false);
184 
185  // This vector will be sorted into a priority queue:
186  vector<TxPriority> vecPriority;
187  vecPriority.reserve(mempool.mapTx.size());
188  for (map<uint256, CTxMemPoolEntry>::iterator mi = mempool.mapTx.begin();
189  mi != mempool.mapTx.end(); ++mi) {
190  const CTransaction& tx = mi->second.GetTx();
191  if (tx.IsCoinBase() || tx.IsCoinStake() || !IsFinalTx(tx, nHeight)){
192  continue;
193  }
194 
195  if (GetAdjustedTime() > GetSporkValue(SPORK_16_ZEROCOIN_MAINTENANCE_MODE) && tx.ContainsZerocoins())
196  continue;
197 
198  COrphan* porphan = NULL;
199  double dPriority = 0;
200  CAmount nTotalIn = 0;
201  bool fMissingInputs = false;
202  for (const CTxIn& txin : tx.vin) {
203  //zerocoinspend has special vin
204  if (tx.IsZerocoinSpend()) {
205  nTotalIn = tx.GetZerocoinSpent();
206  break;
207  }
208 
209  // Read prev transaction
210  if (!view.HaveCoins(txin.prevout.hash)) {
211  // This should never happen; all transactions in the memory
212  // pool should connect to either transactions in the chain
213  // or other transactions in the memory pool.
214  if (!mempool.mapTx.count(txin.prevout.hash)) {
215  LogPrintf("ERROR: mempool transaction missing input\n");
216  if (fDebug) assert("mempool transaction missing input" == 0);
217  fMissingInputs = true;
218  if (porphan)
219  vOrphan.pop_back();
220  break;
221  }
222 
223  // Has to wait for dependencies
224  if (!porphan) {
225  // Use list for automatic deletion
226  vOrphan.push_back(COrphan(&tx));
227  porphan = &vOrphan.back();
228  }
229  mapDependers[txin.prevout.hash].push_back(porphan);
230  porphan->setDependsOn.insert(txin.prevout.hash);
231  nTotalIn += mempool.mapTx[txin.prevout.hash].GetTx().vout[txin.prevout.n].nValue;
232  continue;
233  }
234 
235  //Check for invalid/fraudulent inputs. They shouldn't make it through mempool, but check anyways.
236  if (mapInvalidOutPoints.count(txin.prevout)) {
237  LogPrintf("%s : found invalid input %s in tx %s", __func__, txin.prevout.ToString(), tx.GetHash().ToString());
238  fMissingInputs = true;
239  break;
240  }
241 
242  const CCoins* coins = view.AccessCoins(txin.prevout.hash);
243  assert(coins);
244 
245  CAmount nValueIn = coins->vout[txin.prevout.n].nValue;
246  nTotalIn += nValueIn;
247 
248  int nConf = nHeight - coins->nHeight;
249 
250  dPriority += (double)nValueIn * nConf;
251  }
252  if (fMissingInputs) continue;
253 
254  // Priority is sum(valuein * age) / modified_txsize
255  unsigned int nTxSize = ::GetSerializeSize(tx, SER_NETWORK, PROTOCOL_VERSION);
256  dPriority = tx.ComputePriority(dPriority, nTxSize);
257 
258  uint256 hash = tx.GetHash();
259  mempool.ApplyDeltas(hash, dPriority, nTotalIn);
260 
261  CFeeRate feeRate(nTotalIn - tx.GetValueOut(), nTxSize);
262 
263  if (porphan) {
264  porphan->dPriority = dPriority;
265  porphan->feeRate = feeRate;
266  } else
267  vecPriority.push_back(TxPriority(dPriority, feeRate, &mi->second.GetTx()));
268  }
269 
270  // Collect transactions into block
271  uint64_t nBlockSize = 1000;
272  uint64_t nBlockTx = 0;
273  int nBlockSigOps = 100;
274  bool fSortedByFee = (nBlockPrioritySize <= 0);
275 
276  TxPriorityCompare comparer(fSortedByFee);
277  std::make_heap(vecPriority.begin(), vecPriority.end(), comparer);
278 
279  vector<CBigNum> vBlockSerials;
280  vector<CBigNum> vTxSerials;
281  while (!vecPriority.empty()) {
282  // Take highest priority transaction off the priority queue:
283  double dPriority = vecPriority.front().get<0>();
284  CFeeRate feeRate = vecPriority.front().get<1>();
285  const CTransaction& tx = *(vecPriority.front().get<2>());
286 
287  std::pop_heap(vecPriority.begin(), vecPriority.end(), comparer);
288  vecPriority.pop_back();
289 
290  // Size limits
291  unsigned int nTxSize = ::GetSerializeSize(tx, SER_NETWORK, PROTOCOL_VERSION);
292  if (nBlockSize + nTxSize >= nBlockMaxSize)
293  continue;
294 
295  // Legacy limits on sigOps:
296  unsigned int nMaxBlockSigOps = MAX_BLOCK_SIGOPS_CURRENT;
297  unsigned int nTxSigOps = GetLegacySigOpCount(tx);
298  if (nBlockSigOps + nTxSigOps >= nMaxBlockSigOps)
299  continue;
300 
301  // Skip free transactions if we're past the minimum block size:
302  const uint256& hash = tx.GetHash();
303  double dPriorityDelta = 0;
304  CAmount nFeeDelta = 0;
305  mempool.ApplyDeltas(hash, dPriorityDelta, nFeeDelta);
306  if (!tx.IsZerocoinSpend() && fSortedByFee && (dPriorityDelta <= 0) && (nFeeDelta <= 0) && (feeRate < ::minRelayTxFee) && (nBlockSize + nTxSize >= nBlockMinSize))
307  continue;
308 
309  // Prioritise by fee once past the priority size or we run out of high-priority
310  // transactions:
311  if (!fSortedByFee &&
312  ((nBlockSize + nTxSize >= nBlockPrioritySize) || !AllowFree(dPriority))) {
313  fSortedByFee = true;
314  comparer = TxPriorityCompare(fSortedByFee);
315  std::make_heap(vecPriority.begin(), vecPriority.end(), comparer);
316  }
317 
318  if (!view.HaveInputs(tx))
319  continue;
320 
321  // double check that there are no double spent zDiv spends in this block or tx
322  if (tx.IsZerocoinSpend()) {
323  int nHeightTx = 0;
324  if (IsTransactionInChain(tx.GetHash(), nHeightTx))
325  continue;
326 
327  bool fDoubleSerial = false;
328  for (const CTxIn txIn : tx.vin) {
329  if (txIn.scriptSig.IsZerocoinSpend()) {
330  libzerocoin::CoinSpend spend = TxInToZerocoinSpend(txIn);
331  if (!spend.HasValidSerial(Params().Zerocoin_Params()))
332  fDoubleSerial = true;
333  if (count(vBlockSerials.begin(), vBlockSerials.end(), spend.getCoinSerialNumber()))
334  fDoubleSerial = true;
335  if (count(vTxSerials.begin(), vTxSerials.end(), spend.getCoinSerialNumber()))
336  fDoubleSerial = true;
337  if (fDoubleSerial)
338  break;
339  vTxSerials.emplace_back(spend.getCoinSerialNumber());
340  }
341  }
342  //This zDiv serial has already been included in the block, do not add this tx.
343  if (fDoubleSerial)
344  continue;
345  }
346 
347  CAmount nTxFees = view.GetValueIn(tx) - tx.GetValueOut();
348 
349  nTxSigOps += GetP2SHSigOpCount(tx, view);
350  if (nBlockSigOps + nTxSigOps >= nMaxBlockSigOps)
351  continue;
352 
353  // Note that flags: we don't want to set mempool/IsStandard()
354  // policy here, but we still have to ensure that the block we
355  // create only contains transactions that are valid in new blocks.
356  CValidationState state;
357  if (!CheckInputs(tx, state, view, true, MANDATORY_SCRIPT_VERIFY_FLAGS, true))
358  continue;
359 
360  CTxUndo txundo;
361  UpdateCoins(tx, state, view, txundo, nHeight);
362 
363  // Added
364  pblock->vtx.push_back(tx);
365  pblocktemplate->vTxFees.push_back(nTxFees);
366  pblocktemplate->vTxSigOps.push_back(nTxSigOps);
367  nBlockSize += nTxSize;
368  ++nBlockTx;
369  nBlockSigOps += nTxSigOps;
370  nFees += nTxFees;
371 
372  for (const CBigNum bnSerial : vTxSerials)
373  vBlockSerials.emplace_back(bnSerial);
374 
375  if (fPrintPriority) {
376  LogPrintf("priority %.1f fee %s txid %s\n",
377  dPriority, feeRate.ToString(), tx.GetHash().ToString());
378  }
379 
380  // Add transactions that depend on this one to the priority queue
381  if (mapDependers.count(hash)) {
382  BOOST_FOREACH (COrphan* porphan, mapDependers[hash]) {
383  if (!porphan->setDependsOn.empty()) {
384  porphan->setDependsOn.erase(hash);
385  if (porphan->setDependsOn.empty()) {
386  vecPriority.push_back(TxPriority(porphan->dPriority, porphan->feeRate, porphan->ptx));
387  std::push_heap(vecPriority.begin(), vecPriority.end(), comparer);
388  }
389  }
390  }
391  }
392  }
393 
394  if (!fProofOfStake) {
395  //Masternode and general budget payments
396  FillBlockPayee(txNew, nFees, fProofOfStake);
397 
398  //Make payee
399  if (txNew.vout.size() > 1) {
400  pblock->payee = txNew.vout[1].scriptPubKey;
401  }
402  }
403 
404  nLastBlockTx = nBlockTx;
405  nLastBlockSize = nBlockSize;
406  LogPrintf("CreateNewBlock(): total size %u\n", nBlockSize);
407 
408  // Compute final coinbase transaction.
409  pblock->vtx[0].vin[0].scriptSig = CScript() << nHeight << OP_0;
410  if (!fProofOfStake) {
411  pblock->vtx[0] = txNew;
412  pblocktemplate->vTxFees[0] = -nFees;
413  }
414 
415  // Fill in header
416  pblock->hashPrevBlock = pindexPrev->GetBlockHash();
417  if (!fProofOfStake)
418  UpdateTime(pblock, pindexPrev);
419  pblock->nBits = GetNextWorkRequired(pindexPrev, pblock);
420  pblock->nNonce = 0;
421  uint256 nCheckpoint = 0;
422  if(fZerocoinActive && !CalculateAccumulatorCheckpoint(nHeight, nCheckpoint)){
423  LogPrintf("%s: failed to get accumulator checkpoint\n", __func__);
424  }
425  pblock->nAccumulatorCheckpoint = nCheckpoint;
426  pblocktemplate->vTxSigOps[0] = GetLegacySigOpCount(pblock->vtx[0]);
427 
428  CValidationState state;
429  if (!TestBlockValidity(state, *pblock, pindexPrev, false, false)) {
430  LogPrintf("CreateNewBlock() : TestBlockValidity failed\n");
431  mempool.clear();
432  return NULL;
433  }
434  }
435 
436  return pblocktemplate.release();
437 }

512 void BitcoinMiner(CWallet* pwallet, bool fProofOfStake)
513 {
514  LogPrintf("DIVXMiner started\n");
515  SetThreadPriority(THREAD_PRIORITY_LOWEST);
516  RenameThread("divx-miner");
517 
518  // Each thread has its own key and counter
519  CReserveKey reservekey(pwallet);
520  unsigned int nExtraNonce = 0;
521 
522  //control the amount of times the client will check for mintable coins
523  static bool fMintableCoins = false;
524  static int nMintableLastCheck = 0;
525 
526  if (fProofOfStake && (GetTime() - nMintableLastCheck > 5 * 60)) // 5 minute check time
527  {
528  nMintableLastCheck = GetTime();
529  fMintableCoins = pwallet->MintableCoins();
530  }
531 
532  while (fGenerateBitcoins || fProofOfStake) {
533  if (fProofOfStake) {
534  if (chainActive.Tip()->nHeight < Params().LAST_POW_BLOCK()) {
535  MilliSleep(5000);
536  continue;
537  }
538 
539  while (chainActive.Tip()->nTime < 1471482000 || vNodes.empty() || pwallet->IsLocked() || !fMintableCoins || nReserveBalance >= pwallet->GetBalance() || !masternodeSync.IsSynced()) {
540  nLastCoinStakeSearchInterval = 0;
541  MilliSleep(5000);
542  if (!fGenerateBitcoins && !fProofOfStake)
543  continue;
544  }
545 
546  if (mapHashedBlocks.count(chainActive.Tip()->nHeight)) //search our map of hashed blocks, see if bestblock has been hashed yet
547  {
548  if (GetTime() - mapHashedBlocks[chainActive.Tip()->nHeight] < max(pwallet->nHashInterval, (unsigned int)1)) // wait half of the nHashDrift with max wait of 3 minutes
549  {
550  MilliSleep(5000);
551  continue;
552  }
553  }
554  }
555 
556  //
557  // Create new block
558  //
559  unsigned int nTransactionsUpdatedLast = mempool.GetTransactionsUpdated();
560  CBlockIndex* pindexPrev = chainActive.Tip();
561  if (!pindexPrev)
562  continue;
563 
564  unique_ptr<CBlockTemplate> pblocktemplate(CreateNewBlockWithKey(reservekey, pwallet, fProofOfStake));
565  if (!pblocktemplate.get())
566  continue;
567 
568  CBlock* pblock = &pblocktemplate->block;
569  IncrementExtraNonce(pblock, pindexPrev, nExtraNonce);
570 
571  //Stake miner main
572  if (fProofOfStake) {
573  LogPrintf("CPUMiner : proof-of-stake block found %s \n", pblock->GetHash().ToString().c_str());
574 
575  if (!pblock->SignBlock(*pwallet)) {
576  LogPrintf("BitcoinMiner(): Signing new block failed \n");
577  continue;
578  }
579 
580  LogPrintf("CPUMiner : proof-of-stake block was signed %s \n", pblock->GetHash().ToString().c_str());
581  SetThreadPriority(THREAD_PRIORITY_NORMAL);
582  ProcessBlockFound(pblock, *pwallet, reservekey);
583  SetThreadPriority(THREAD_PRIORITY_LOWEST);
584 
585  continue;
586  }
587 
588  LogPrintf("Running DIVXMiner with %u transactions in block (%u bytes)\n", pblock->vtx.size(),
589  ::GetSerializeSize(*pblock, SER_NETWORK, PROTOCOL_VERSION));
590 
591  //
592  // Search
593  //
594  int64_t nStart = GetTime();
595  uint256 hashTarget = uint256().SetCompact(pblock->nBits);
596  while (true) {
597  unsigned int nHashesDone = 0;
598 
599  uint256 hash;
600  while (true) {
601  hash = pblock->GetHash();
602  if (hash <= hashTarget) {
603  // Found a solution
604  SetThreadPriority(THREAD_PRIORITY_NORMAL);
605  LogPrintf("BitcoinMiner:\n");
606  LogPrintf("proof-of-work found \n hash: %s \ntarget: %s\n", hash.GetHex(), hashTarget.GetHex());
607  ProcessBlockFound(pblock, *pwallet, reservekey);
608  SetThreadPriority(THREAD_PRIORITY_LOWEST);
609 
610  // In regression test mode, stop mining after a block is found. This
611  // allows developers to controllably generate a block on demand.
612  if (Params().MineBlocksOnDemand())
613  throw boost::thread_interrupted();
614 
615  break;
616  }
617  pblock->nNonce += 1;
618  nHashesDone += 1;
619  if ((pblock->nNonce & 0xFF) == 0)
620  break;
621  }
622 
623  // Meter hashes/sec
624  static int64_t nHashCounter;
625  if (nHPSTimerStart == 0) {
626  nHPSTimerStart = GetTimeMillis();
627  nHashCounter = 0;
628  } else
629  nHashCounter += nHashesDone;
630  if (GetTimeMillis() - nHPSTimerStart > 4000) {
631  static CCriticalSection cs;
632  {
633  LOCK(cs);
634  if (GetTimeMillis() - nHPSTimerStart > 4000) {
635  dHashesPerSec = 1000.0 * nHashCounter / (GetTimeMillis() - nHPSTimerStart);
636  nHPSTimerStart = GetTimeMillis();
637  nHashCounter = 0;
638  static int64_t nLogTime;
639  if (GetTime() - nLogTime > 30 * 60) {
640  nLogTime = GetTime();
641  LogPrintf("hashmeter %6.0f khash/s\n", dHashesPerSec / 1000.0);
642  }
643  }
644  }
645  }
646 
647  // Check for stop or if block needs to be rebuilt
648  boost::this_thread::interruption_point();
649  // Regtest mode doesn't require peers
650  if (vNodes.empty() && Params().MiningRequiresPeers())
651  break;
652  if (pblock->nNonce >= 0xffff0000)
653  break;
654  if (mempool.GetTransactionsUpdated() != nTransactionsUpdatedLast && GetTime() - nStart > 60)
655  break;
656  if (pindexPrev != chainActive.Tip())
657  break;
658 
659  // Update nTime every few seconds
660  UpdateTime(pblock, pindexPrev);
661  if (Params().AllowMinDifficultyBlocks()) {
662  // Changing pblock->nTime can change work required on testnet:
663  hashTarget.SetCompact(pblock->nBits);
664  }
665  }
666  }
667 }
