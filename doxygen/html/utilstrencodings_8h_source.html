70 std::string HexStr(const T itbegin, const T itend, bool fSpaces = false)
71 {
72  std::string rv;
73  static const char hexmap[16] = {'0', '1', '2', '3', '4', '5', '6', '7',
74  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'};
75  rv.reserve((itend - itbegin) * 3);
76  for (T it = itbegin; it < itend; ++it) {
77  unsigned char val = (unsigned char)(*it);
78  if (fSpaces && it != itbegin)
79  rv.push_back(' ');
80  rv.push_back(hexmap[val >> 4]);
81  rv.push_back(hexmap[val & 15]);
82  }
83 
84  return rv;
85 }

117 bool TimingResistantEqual(const T& a, const T& b)
118 {
119  if (b.size() == 0) return a.size() == 0;
120  size_t accumulator = a.size() ^ b.size();
121  for (size_t i = 0; i < a.size(); i++)
122  accumulator |= a[i] ^ b[i % b.size()];
123  return accumulator == 0;
