95 class CObfuScationEntry
96 {
97 public:
98  bool isSet;
99  std::vector<CTxDSIn> sev;
100  std::vector<CTxDSOut> vout;
101  CAmount amount;
102  CTransaction collateral;
103  CTransaction txSupporting;
104  int64_t addedTime; // time in UTC milliseconds
105 
106  CObfuScationEntry()
107  {
108  isSet = false;
109  collateral = CTransaction();
110  amount = 0;
111  }
112 
114  bool Add(const std::vector<CTxIn> vinIn, int64_t amountIn, const CTransaction collateralIn, const std::vector<CTxOut> voutIn)
115  {
116  if (isSet) {
117  return false;
118  }
119 
120  BOOST_FOREACH (const CTxIn& in, vinIn)
121  sev.push_back(in);
122 
123  BOOST_FOREACH (const CTxOut& out, voutIn)
124  vout.push_back(out);
125 
126  amount = amountIn;
127  collateral = collateralIn;
128  isSet = true;
129  addedTime = GetTime();
130 
131  return true;
132  }
133 
134  bool AddSig(const CTxIn& vin)
135  {
136  BOOST_FOREACH (CTxDSIn& s, sev) {
137  if (s.prevout == vin.prevout && s.nSequence == vin.nSequence) {
138  if (s.fHasSig) {
139  return false;
140  }
141  s.scriptSig = vin.scriptSig;
142  s.prevPubKey = vin.prevPubKey;
143  s.fHasSig = true;
144 
145  return true;
146  }
147  }
148 
149  return false;
150  }
151 
152  bool IsExpired()
153  {
154  return (GetTime() - addedTime) > OBFUSCATION_QUEUE_TIMEOUT; // 120 seconds
155  }
156 };
