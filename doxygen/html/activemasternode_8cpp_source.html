17 void CActiveMasternode::ManageStatus()
18 {
19  std::string errorMessage;
20 
21  if (!fMasterNode) return;
22 
23  if (fDebug) LogPrintf("CActiveMasternode::ManageStatus() - Begin\n");
24 
25  //need correct blocks to send ping
26  if (Params().NetworkID() != CBaseChainParams::REGTEST && !masternodeSync.IsBlockchainSynced()) {
27  status = ACTIVE_MASTERNODE_SYNC_IN_PROCESS;
28  LogPrintf("CActiveMasternode::ManageStatus() - %s\n", GetStatus());
29  return;
30  }
31 
32  if (status == ACTIVE_MASTERNODE_SYNC_IN_PROCESS) status = ACTIVE_MASTERNODE_INITIAL;
33 
34  if (status == ACTIVE_MASTERNODE_INITIAL) {
35  CMasternode* pmn;
36  pmn = mnodeman.Find(pubKeyMasternode);
37  if (pmn != NULL) {
38  pmn->Check();
39  if (pmn->IsEnabled() && pmn->protocolVersion == PROTOCOL_VERSION) EnableHotColdMasterNode(pmn->vin, pmn->addr);
40  }
41  }
42 
43  if (status != ACTIVE_MASTERNODE_STARTED) {
44  // Set defaults
45  status = ACTIVE_MASTERNODE_NOT_CAPABLE;
46  notCapableReason = "";
47 
48  if (pwalletMain->IsLocked()) {
49  notCapableReason = "Wallet is locked.";
50  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
51  return;
52  }
53 
54  if (pwalletMain->GetBalance() == 0) {
55  notCapableReason = "Hot node, waiting for remote activation.";
56  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
57  return;
58  }
59 
60  if (strMasterNodeAddr.empty()) {
61  if (!GetLocal(service)) {
62  notCapableReason = "Can't detect external address. Please use the masternodeaddr configuration option.";
63  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
64  return;
65  }
66  } else {
67  service = CService(strMasterNodeAddr);
68  }
69 
70  if (Params().NetworkID() == CBaseChainParams::MAIN) {
71  if (service.GetPort() != 51472) {
72  notCapableReason = strprintf("Invalid port: %u - only 51472 is supported on mainnet.", service.GetPort());
73  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
74  return;
75  }
76  } else if (service.GetPort() == 51472) {
77  notCapableReason = strprintf("Invalid port: %u - 51472 is only supported on mainnet.", service.GetPort());
78  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
79  return;
80  }
81 
82  LogPrintf("CActiveMasternode::ManageStatus() - Checking inbound connection to '%s'\n", service.ToString());
83 
84  CNode* pnode = ConnectNode((CAddress)service, NULL, false);
85  if (!pnode) {
86  notCapableReason = "Could not connect to " + service.ToString();
87  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason);
88  return;
89  }
90  pnode->Release();
91 
92  // Choose coins to use
93  CPubKey pubKeyCollateralAddress;
94  CKey keyCollateralAddress;
95 
96  if (GetMasterNodeVin(vin, pubKeyCollateralAddress, keyCollateralAddress)) {
97  if (GetInputAge(vin) < MASTERNODE_MIN_CONFIRMATIONS) {
98  status = ACTIVE_MASTERNODE_INPUT_TOO_NEW;
99  notCapableReason = strprintf("%s - %d confirmations", GetStatus(), GetInputAge(vin));
100  LogPrintf("CActiveMasternode::ManageStatus() - %s\n", notCapableReason);
101  return;
102  }
103 
104  LOCK(pwalletMain->cs_wallet);
105  pwalletMain->LockCoin(vin.prevout);
106 
107  // send to all nodes
108  CPubKey pubKeyMasternode;
109  CKey keyMasternode;
110 
111  if (!obfuScationSigner.SetKey(strMasterNodePrivKey, errorMessage, keyMasternode, pubKeyMasternode)) {
112  notCapableReason = "Error upon calling SetKey: " + errorMessage;
113  LogPrintf("Register::ManageStatus() - %s\n", notCapableReason);
114  return;
115  }
116 
117  if (!Register(vin, service, keyCollateralAddress, pubKeyCollateralAddress, keyMasternode, pubKeyMasternode, errorMessage)) {
118  notCapableReason = "Error on Register: " + errorMessage;
119  LogPrintf("Register::ManageStatus() - %s\n", notCapableReason);
120  return;
121  }
122 
123  LogPrintf("CActiveMasternode::ManageStatus() - Is capable master node!\n");
124  status = ACTIVE_MASTERNODE_STARTED;
125 
126  return;
127  } else {
128  notCapableReason = "Could not find suitable coins!";
129  LogPrintf("CActiveMasternode::ManageStatus() - %s\n", notCapableReason);
130  return;
131  }
132  }
133 
134  //send to all peers
135  if (!SendMasternodePing(errorMessage)) {
136  LogPrintf("CActiveMasternode::ManageStatus() - Error on Ping: %s\n", errorMessage);
137  }

140 std::string CActiveMasternode::GetStatus()
141 {
142  switch (status) {
143  case ACTIVE_MASTERNODE_INITIAL:
144  return "Node just started, not yet activated";
145  case ACTIVE_MASTERNODE_SYNC_IN_PROCESS:

147  case ACTIVE_MASTERNODE_INPUT_TOO_NEW:
148  return strprintf("Masternode input must have at least %d confirmations", MASTERNODE_MIN_CONFIRMATIONS);

158 bool CActiveMasternode::SendMasternodePing(std::string& errorMessage)
159 {
160  if (status != ACTIVE_MASTERNODE_STARTED) {
161  errorMessage = "Masternode is not in a running status";
162  return false;
163  }
164 
165  CPubKey pubKeyMasternode;
166  CKey keyMasternode;
167 
168  if (!obfuScationSigner.SetKey(strMasterNodePrivKey, errorMessage, keyMasternode, pubKeyMasternode)) {
169  errorMessage = strprintf("Error upon calling SetKey: %s\n", errorMessage);
170  return false;
171  }
172 
173  LogPrintf("CActiveMasternode::SendMasternodePing() - Relay Masternode Ping vin = %s\n", vin.ToString());
174 
175  CMasternodePing mnp(vin);
176  if (!mnp.Sign(keyMasternode, pubKeyMasternode)) {
177  errorMessage = "Couldn't sign Masternode Ping";
178  return false;
179  }
180 
181  // Update lastPing for our masternode in Masternode list
182  CMasternode* pmn = mnodeman.Find(vin);
183  if (pmn != NULL) {
184  if (pmn->IsPingedWithin(MASTERNODE_PING_SECONDS, mnp.sigTime)) {
185  errorMessage = "Too early to send Masternode Ping";
186  return false;
187  }
188 
189  pmn->lastPing = mnp;
190  mnodeman.mapSeenMasternodePing.insert(make_pair(mnp.GetHash(), mnp));
191 
192  //mnodeman.mapSeenMasternodeBroadcast.lastPing is probably outdated, so we'll update it
193  CMasternodeBroadcast mnb(*pmn);
194  uint256 hash = mnb.GetHash();
195  if (mnodeman.mapSeenMasternodeBroadcast.count(hash)) mnodeman.mapSeenMasternodeBroadcast[hash].lastPing = mnp;
196 
197  mnp.Relay();
198 
199  /*
200  * IT'S SAFE TO REMOVE THIS IN FURTHER VERSIONS
201  * AFTER MIGRATION TO V12 IS DONE
202  */
203 
204  if (IsSporkActive(SPORK_10_MASTERNODE_PAY_UPDATED_NODES)) return true;
205  // for migration purposes ping our node on old masternodes network too
206  std::string retErrorMessage;
207  std::vector<unsigned char> vchMasterNodeSignature;
208  int64_t masterNodeSignatureTime = GetAdjustedTime();
209 
210  std::string strMessage = service.ToString() + boost::lexical_cast<std::string>(masterNodeSignatureTime) + boost::lexical_cast<std::string>(false);
211 
212  if (!obfuScationSigner.SignMessage(strMessage, retErrorMessage, vchMasterNodeSignature, keyMasternode)) {
213  errorMessage = "dseep sign message failed: " + retErrorMessage;
214  return false;
215  }
216 
217  if (!obfuScationSigner.VerifyMessage(pubKeyMasternode, vchMasterNodeSignature, strMessage, retErrorMessage)) {
218  errorMessage = "dseep verify message failed: " + retErrorMessage;
219  return false;
220  }
221 
222  LogPrint("masternode", "dseep - relaying from active mn, %s \n", vin.ToString().c_str());
223  LOCK(cs_vNodes);
224  BOOST_FOREACH (CNode* pnode, vNodes)
225  pnode->PushMessage("dseep", vin, vchMasterNodeSignature, masterNodeSignatureTime, false);
226 
227  /*
228  * END OF "REMOVE"
229  */
230 
231  return true;
232  } else {
233  // Seems like we are trying to send a ping while the Masternode is not registered in the network
234  errorMessage = "Obfuscation Masternode List doesn't include our Masternode, shutting down Masternode pinging service! " + vin.ToString();
235  status = ACTIVE_MASTERNODE_NOT_CAPABLE;
236  notCapableReason = errorMessage;
237  return false;
238  }

286 bool CActiveMasternode::Register(CTxIn vin, CService service, CKey keyCollateralAddress, CPubKey pubKeyCollateralAddress, CKey keyMasternode, CPubKey pubKeyMasternode, std::string& errorMessage)
287 {
288  CMasternodeBroadcast mnb;
289  CMasternodePing mnp(vin);
290  if (!mnp.Sign(keyMasternode, pubKeyMasternode)) {
291  errorMessage = strprintf("Failed to sign ping, vin: %s", vin.ToString());
292  LogPrintf("CActiveMasternode::Register() - %s\n", errorMessage);
293  return false;
294  }
295  mnodeman.mapSeenMasternodePing.insert(make_pair(mnp.GetHash(), mnp));
296 
297  LogPrintf("CActiveMasternode::Register() - Adding to Masternode list\n service: %s\n vin: %s\n", service.ToString(), vin.ToString());
298  mnb = CMasternodeBroadcast(service, vin, pubKeyCollateralAddress, pubKeyMasternode, PROTOCOL_VERSION);
299  mnb.lastPing = mnp;
300  if (!mnb.Sign(keyCollateralAddress)) {
301  errorMessage = strprintf("Failed to sign broadcast, vin: %s", vin.ToString());
302  LogPrintf("CActiveMasternode::Register() - %s\n", errorMessage);
303  return false;
304  }
305  mnodeman.mapSeenMasternodeBroadcast.insert(make_pair(mnb.GetHash(), mnb));
306  masternodeSync.AddedMasternodeList(mnb.GetHash());
307 
308  CMasternode* pmn = mnodeman.Find(vin);
309  if (pmn == NULL) {
310  CMasternode mn(mnb);
311  mnodeman.Add(mn);
312  } else {
313  pmn->UpdateFromNewBroadcast(mnb);
314  }
315 
316  //send to all peers
317  LogPrintf("CActiveMasternode::Register() - RelayElectionEntry vin = %s\n", vin.ToString());
318  mnb.Relay();
319 
320  /*
321  * IT'S SAFE TO REMOVE THIS IN FURTHER VERSIONS
322  * AFTER MIGRATION TO V12 IS DONE
323  */
324 
325  if (IsSporkActive(SPORK_10_MASTERNODE_PAY_UPDATED_NODES)) return true;
326  // for migration purposes inject our node in old masternodes' list too
327  std::string retErrorMessage;
328  std::vector<unsigned char> vchMasterNodeSignature;
329  int64_t masterNodeSignatureTime = GetAdjustedTime();
330  std::string donationAddress = "";
331  int donationPercantage = 0;
332 
333  std::string vchPubKey(pubKeyCollateralAddress.begin(), pubKeyCollateralAddress.end());
334  std::string vchPubKey2(pubKeyMasternode.begin(), pubKeyMasternode.end());
335 
336  std::string strMessage = service.ToString() + boost::lexical_cast<std::string>(masterNodeSignatureTime) + vchPubKey + vchPubKey2 + boost::lexical_cast<std::string>(PROTOCOL_VERSION) + donationAddress + boost::lexical_cast<std::string>(donationPercantage);
337 
338  if (!obfuScationSigner.SignMessage(strMessage, retErrorMessage, vchMasterNodeSignature, keyCollateralAddress)) {
339  errorMessage = "dsee sign message failed: " + retErrorMessage;
340  LogPrintf("CActiveMasternode::Register() - Error: %s\n", errorMessage.c_str());
341  return false;
342  }
343 
344  if (!obfuScationSigner.VerifyMessage(pubKeyCollateralAddress, vchMasterNodeSignature, strMessage, retErrorMessage)) {
345  errorMessage = "dsee verify message failed: " + retErrorMessage;
346  LogPrintf("CActiveMasternode::Register() - Error: %s\n", errorMessage.c_str());
347  return false;
348  }
349 
350  LOCK(cs_vNodes);
351  BOOST_FOREACH (CNode* pnode, vNodes)
352  pnode->PushMessage("dsee", vin, service, vchMasterNodeSignature, masterNodeSignatureTime, pubKeyCollateralAddress, pubKeyMasternode, -1, -1, masterNodeSignatureTime, PROTOCOL_VERSION, donationAddress, donationPercantage);
353 
354  /*
355  * END OF "REMOVE"
356  */
357 
358  return true;

366 bool CActiveMasternode::GetMasterNodeVin(CTxIn& vin, CPubKey& pubkey, CKey& secretKey, std::string strTxHash, std::string strOutputIndex)
367 {
368  // Find possible candidates
369  TRY_LOCK(pwalletMain->cs_wallet, fWallet);
370  if (!fWallet) return false;
371 
372  vector<COutput> possibleCoins = SelectCoinsMasternode();
373  COutput* selectedOutput;
374 
375  // Find the vin
376  if (!strTxHash.empty()) {
377  // Let's find it
378  uint256 txHash(strTxHash);
379  int outputIndex;
380  try {
381  outputIndex = std::stoi(strOutputIndex.c_str());
382  } catch (const std::exception& e) {
383  LogPrintf("%s: %s on strOutputIndex\n", __func__, e.what());
384  return false;
385  }
386 
387  bool found = false;
388  BOOST_FOREACH (COutput& out, possibleCoins) {
389  if (out.tx->GetHash() == txHash && out.i == outputIndex) {
390  selectedOutput = &out;
391  found = true;
392  break;
393  }
394  }
395  if (!found) {
396  LogPrintf("CActiveMasternode::GetMasterNodeVin - Could not locate valid vin\n");
397  return false;
398  }
399  } else {
400  // No output specified, Select the first one
401  if (possibleCoins.size() > 0) {
402  selectedOutput = &possibleCoins[0];
403  } else {
404  LogPrintf("CActiveMasternode::GetMasterNodeVin - Could not locate specified vin from possible list\n");
405  return false;
406  }
407  }
408 
409  // At this point we have a selected output, retrieve the associated info
410  return GetVinFromOutput(*selectedOutput, vin, pubkey, secretKey);
411 }

441 // get all possible outputs for running Masternode
442 vector<COutput> CActiveMasternode::SelectCoinsMasternode()
443 {
444  vector<COutput> vCoins;
445  vector<COutput> filteredCoins;
446  vector<COutPoint> confLockedCoins;
447 
448  // Temporary unlock MN coins from masternode.conf
449  if (GetBoolArg("-mnconflock", true)) {
450  uint256 mnTxHash;
451  BOOST_FOREACH (CMasternodeConfig::CMasternodeEntry mne, masternodeConfig.getEntries()) {
452  mnTxHash.SetHex(mne.getTxHash());
453 
454  int nIndex;
455  if(!mne.castOutputIndex(nIndex))
456  continue;
457 
458  COutPoint outpoint = COutPoint(mnTxHash, nIndex);
459  confLockedCoins.push_back(outpoint);
460  pwalletMain->UnlockCoin(outpoint);
461  }
462  }
463 
464  // Retrieve all possible outputs
465  pwalletMain->AvailableCoins(vCoins);
466 
467  // Lock MN coins from masternode.conf back if they where temporary unlocked
468  if (!confLockedCoins.empty()) {
469  BOOST_FOREACH (COutPoint outpoint, confLockedCoins)
470  pwalletMain->LockCoin(outpoint);
471  }
472 
473  // Filter
474  BOOST_FOREACH (const COutput& out, vCoins) {
475  if (out.tx->vout[out.i].nValue == 10000 * COIN) { //exactly
476  filteredCoins.push_back(out);
477  }
478  }
479  return filteredCoins;
