34 static SendCoinsRecipient handleRequest(PaymentServer* server, std::vector<unsigned char>& data)
35 {
36  RecipientCatcher sigCatcher;
37  QObject::connect(server, SIGNAL(receivedPaymentRequest(SendCoinsRecipient)),
38  &sigCatcher, SLOT(getRecipient(SendCoinsRecipient)));
39 
40  // Write data to a temp file:
41  QTemporaryFile f;
42  f.open();
43  f.write((const char*)&data[0], data.size());
44  f.close();
45 
46  // Create a QObject, install event filter from PaymentServer
47  // and send a file open event to the object
48  QObject object;
49  object.installEventFilter(server);
50  QFileOpenEvent event(f.fileName());
51  // If sending the event fails, this will cause sigCatcher to be empty,
52  // which will lead to a test failure anyway.
53  QCoreApplication::sendEvent(&object, &event);
54 
55  QObject::disconnect(server, SIGNAL(receivedPaymentRequest(SendCoinsRecipient)),
56  &sigCatcher, SLOT(getRecipient(SendCoinsRecipient)));
57 
58  // Return results from sigCatcher
59  return sigCatcher.recipient;
60 }
