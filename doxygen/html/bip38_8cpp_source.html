80 std::string BIP38_Encrypt(std::string strAddress, std::string strPassphrase, uint256 privKey, bool fCompressed)
81 {
82  string strAddressHash = AddressToBip38Hash(strAddress);
83 
84  uint512 hashed;
85  uint64_t salt = uint256(ReverseEndianString(strAddressHash)).Get64();
86  scrypt_hash(strPassphrase.c_str(), strPassphrase.size(), BEGIN(salt), strAddressHash.size() / 2, BEGIN(hashed), 16384, 8, 8, 64);
87 
88  uint256 derivedHalf1(hashed.ToString().substr(64, 64));
89  uint256 derivedHalf2(hashed.ToString().substr(0, 64));
90 
91  //block1 = (pointb[1...16] xor derivedhalf1[0...15])
92  uint256 block1 = uint256((privKey << 128) ^ (derivedHalf1 << 128)) >> 128;
93 
94  //encrypt part 1
95  uint512 encrypted1;
96  AES_KEY key;
97  AES_set_encrypt_key(derivedHalf2.begin(), 256, &key);
98  AES_encrypt(block1.begin(), encrypted1.begin(), &key);
99 
100  //block2 = (pointb[17...32] xor derivedhalf1[16...31]
101  uint256 p2 = privKey >> 128;
102  uint256 dh12 = derivedHalf1 >> 128;
103  uint256 block2 = uint256(p2 ^ dh12);
104 
105  //encrypt part 2
106  uint512 encrypted2;
107  AES_encrypt(block2.begin(), encrypted2.begin(), &key);
108 
109  string strPrefix = "0142";
110  strPrefix += (fCompressed ? "E0" : "C0");
111 
112  uint512 encryptedKey(ReverseEndianString(strPrefix + strAddressHash));
113 
114  //add encrypted1 to the end of encryptedKey
115  encryptedKey = encryptedKey | (encrypted1 << 56);
116 
117  //add encrypted2 to the end of encryptedKey
118  encryptedKey = encryptedKey | (encrypted2 << (56 + 128));
119 
120  //Base58 checksum is the 4 bytes of dSHA256 hash of the encrypted key
121  uint256 hashChecksum = Hash(encryptedKey.begin(), encryptedKey.begin() + 39);
122  uint512 b58Checksum(hashChecksum.ToString().substr(64 - 8, 8));
123 
124  // append the encrypted key with checksum (currently occupies 312 bits)
125  encryptedKey = encryptedKey | (b58Checksum << 312);
126 
127  //43 bytes is the total size that we are encoding
128  return EncodeBase58(encryptedKey.begin(), encryptedKey.begin() + 43);
129 }

131 bool BIP38_Decrypt(std::string strPassphrase, std::string strEncryptedKey, uint256& privKey, bool& fCompressed)
132 {
133  std::string strKey = DecodeBase58(strEncryptedKey.c_str());
134 
135  //incorrect encoding of key, it must be 39 bytes - and another 4 bytes for base58 checksum
136  if (strKey.size() != (78 + 8))
137  return false;
138 
139  //invalid prefix
140  if (uint256(ReverseEndianString(strKey.substr(0, 2))) != uint256(0x01))
141  return false;
142 
143  uint256 type(ReverseEndianString(strKey.substr(2, 2)));
144  uint256 flag(ReverseEndianString(strKey.substr(4, 2)));
145  std::string strAddressHash = strKey.substr(6, 8);
146  std::string ownersalt = strKey.substr(14, 16);
147  uint256 encryptedPart1(ReverseEndianString(strKey.substr(30, 16)));
148  uint256 encryptedPart2(ReverseEndianString(strKey.substr(46, 32)));
149 
150  fCompressed = (flag & uint256(0x20)) != 0;
151 
152  //not ec multiplied
153  if (type == uint256(0x42)) {
154  uint512 hashed;
155  encryptedPart1 = uint256(ReverseEndianString(strKey.substr(14, 32)));
156  uint64_t salt = uint256(ReverseEndianString(strAddressHash)).Get64();
157  scrypt_hash(strPassphrase.c_str(), strPassphrase.size(), BEGIN(salt), strAddressHash.size() / 2, BEGIN(hashed), 16384, 8, 8, 64);
158 
159  uint256 derivedHalf1(hashed.ToString().substr(64, 64));
160  uint256 derivedHalf2(hashed.ToString().substr(0, 64));
161 
162  uint256 decryptedPart1;
163  DecryptAES(encryptedPart1, derivedHalf2, decryptedPart1);
164 
165  uint256 decryptedPart2;
166  DecryptAES(encryptedPart2, derivedHalf2, decryptedPart2);
167 
168  //combine decrypted parts into 64 bytes
169  uint256 temp1 = decryptedPart2 << 128;
170  temp1 = temp1 | decryptedPart1;
171 
172  //xor the decryption with the derived half 1 for the final key
173  privKey = temp1 ^ derivedHalf1;
174 
175  return true;
176  } else if (type != uint256(0x43)) //invalid type
177  return false;
178 
179  bool fLotSequence = (flag & 0x04) != 0;
180 
181  std::string prefactorSalt = ownersalt;
182  if (fLotSequence)
183  prefactorSalt = ownersalt.substr(0, 8);
184 
185  uint256 prefactor;
186  ComputePreFactor(strPassphrase, prefactorSalt, prefactor);
187 
188  uint256 passfactor;
189  if (fLotSequence)
190  ComputePassfactor(ownersalt, prefactor, passfactor);
191  else
192  passfactor = prefactor;
193 
194  CPubKey passpoint;
195  if (!ComputePasspoint(passfactor, passpoint))
196  return false;
197 
198  uint512 seedBPass;
199  ComputeSeedBPass(passpoint, strAddressHash, ownersalt, seedBPass);
200 
201  //get derived halfs, being mindful for endian switch
202  uint256 derivedHalf1(seedBPass.ToString().substr(64, 64));
203  uint256 derivedHalf2(seedBPass.ToString().substr(0, 64));
204 
206  uint256 decryptedPart2;
207  DecryptAES(encryptedPart2, derivedHalf2, decryptedPart2);
208 
209  //xor decryptedPart2 and 2nd half of derived half 1
210  uint256 x0 = derivedHalf1 >> 128; //drop off the first half (note: endian)
211  uint256 x1 = decryptedPart2 ^ x0;
212  uint256 seedbPart2 = x1 >> 64;
213 
215  uint256 decryptedPart1;
216  uint256 x2 = x1 & uint256("0xffffffffffffffff"); // set x2 to seedbPart1 (still encrypted)
217  x2 = x2 << 64; //make room to add encryptedPart1 to the front
218  x2 = encryptedPart1 | x2; //combine with encryptedPart1
219  DecryptAES(x2, derivedHalf2, decryptedPart1);
220 
221  //decrypted part 1: seedb[0..15] xor derivedhalf1[0..15]
222  uint256 x3 = derivedHalf1 & uint256("0xffffffffffffffffffffffffffffffff");
223  uint256 seedbPart1 = decryptedPart1 ^ x3;
224  uint256 seedB = seedbPart1 | (seedbPart2 << 128);
225 
226  uint256 factorB;
227  ComputeFactorB(seedB, factorB);
228 
229  //multiply passfactor by factorb mod N to yield the priv key
230  privKey = factorB;
231  if (!secp256k1_ec_privkey_tweak_mul(privKey.begin(), passfactor.begin()))
232  return false;
233 
234  //double check that the address hash matches our final privkey
235  CKey k;
236  k.Set(privKey.begin(), privKey.end(), fCompressed);
237  CPubKey pubkey = k.GetPubKey();
238  string address = CBitcoinAddress(pubkey.GetID()).ToString();
239 
240  return strAddressHash == AddressToBip38Hash(address);
241 }
