44 CommitmentProofOfKnowledge::CommitmentProofOfKnowledge(const IntegerGroupParams* aParams,
45  const IntegerGroupParams* bParams, const Commitment& a, const Commitment& b):
46  ap(aParams),bp(bParams)
47 {
48  CBigNum r1, r2, r3;
49 
50  // First: make sure that the two commitments have the
51  // same contents.
52  if (a.getContents() != b.getContents()) {
53  throw std::runtime_error("Both commitments must contain the same value");
54  }
55 
56  // Select three random values "r1, r2, r3" in the range 0 to (2^l)-1 where l is:
57  // length of challenge value + max(modulus 1, modulus 2, order 1, order 2) + margin.
58  // We set "margin" to be a relatively generous security parameter.
59  //
60  // We choose these large values to ensure statistical zero knowledge.
61  uint32_t randomSize = COMMITMENT_EQUALITY_CHALLENGE_SIZE + COMMITMENT_EQUALITY_SECMARGIN +
62  std::max(std::max(this->ap->modulus.bitSize(), this->bp->modulus.bitSize()),
63  std::max(this->ap->groupOrder.bitSize(), this->bp->groupOrder.bitSize()));
64  CBigNum maxRange = (CBigNum(2).pow(randomSize) - CBigNum(1));
65 
66  r1 = CBigNum::randBignum(maxRange);
67  r2 = CBigNum::randBignum(maxRange);
68  r3 = CBigNum::randBignum(maxRange);
69 
70  // Generate two random, ephemeral commitments "T1, T2"
71  // of the form:
72  // T1 = g1^r1 * h1^r2 mod p1
73  // T2 = g2^r1 * h2^r3 mod p2
74  //
75  // Where (g1, h1, p1) are from "aParams" and (g2, h2, p2) are from "bParams".
76  CBigNum T1 = this->ap->g.pow_mod(r1, this->ap->modulus).mul_mod((this->ap->h.pow_mod(r2, this->ap->modulus)), this->ap->modulus);
77  CBigNum T2 = this->bp->g.pow_mod(r1, this->bp->modulus).mul_mod((this->bp->h.pow_mod(r3, this->bp->modulus)), this->bp->modulus);
78 
79  // Now hash commitment "A" with commitment "B" as well as the
80  // parameters and the two ephemeral commitments "T1, T2" we just generated
81  this->challenge = calculateChallenge(a.getCommitmentValue(), b.getCommitmentValue(), T1, T2);
82 
83  // Let "m" be the contents of the commitments "A, B". We have:
84  // A = g1^m * h1^x mod p1
85  // B = g2^m * h2^y mod p2
86  // T1 = g1^r1 * h1^r2 mod p1
87  // T2 = g2^r1 * h2^r3 mod p2
88  //
89  // Now compute:
90  // S1 = r1 + (m * challenge) -- note, not modular arithmetic
91  // S2 = r2 + (x * challenge) -- note, not modular arithmetic
92  // S3 = r3 + (y * challenge) -- note, not modular arithmetic
93  this->S1 = r1 + (a.getContents() * this->challenge);
94  this->S2 = r2 + (a.getRandomness() * this->challenge);
95  this->S3 = r3 + (b.getRandomness() * this->challenge);
96 
97  // We're done. The proof is S1, S2, S3 and "challenge", all of which
98  // are stored in member variables.
99 }

101 bool CommitmentProofOfKnowledge::Verify(const CBigNum& A, const CBigNum& B) const
102 {
103  // Compute the maximum range of S1, S2, S3 and verify that the given values are
104  // in a correct range. This might be an unnecessary check.
105  uint32_t maxSize = 64 * (COMMITMENT_EQUALITY_CHALLENGE_SIZE + COMMITMENT_EQUALITY_SECMARGIN +
106  std::max(std::max(this->ap->modulus.bitSize(), this->bp->modulus.bitSize()),
107  std::max(this->ap->groupOrder.bitSize(), this->bp->groupOrder.bitSize())));
108 
109  if ((uint32_t)this->S1.bitSize() > maxSize ||
110  (uint32_t)this->S2.bitSize() > maxSize ||
111  (uint32_t)this->S3.bitSize() > maxSize ||
112  this->S1 < CBigNum(0) ||
113  this->S2 < CBigNum(0) ||
114  this->S3 < CBigNum(0) ||
115  this->challenge < CBigNum(0) ||
116  this->challenge > (CBigNum(2).pow(COMMITMENT_EQUALITY_CHALLENGE_SIZE) - CBigNum(1))) {
117  // Invalid inputs. Reject.
118  return false;
119  }
120 
121  // Compute T1 = g1^S1 * h1^S2 * inverse(A^{challenge}) mod p1
122  CBigNum T1 = A.pow_mod(this->challenge, ap->modulus).inverse(ap->modulus).mul_mod(
123  (ap->g.pow_mod(S1, ap->modulus).mul_mod(ap->h.pow_mod(S2, ap->modulus), ap->modulus)),
124  ap->modulus);
125 
126  // Compute T2 = g2^S1 * h2^S3 * inverse(B^{challenge}) mod p2
127  CBigNum T2 = B.pow_mod(this->challenge, bp->modulus).inverse(bp->modulus).mul_mod(
128  (bp->g.pow_mod(S1, bp->modulus).mul_mod(bp->h.pow_mod(S3, bp->modulus), bp->modulus)),
129  bp->modulus);
130 
131  // Hash T1 and T2 along with all of the public parameters
132  CBigNum computedChallenge = calculateChallenge(A, B, T1, T2);
133 
134  // Return success if the computed challenge matches the incoming challenge
135  if(computedChallenge == this->challenge) {
136  return true;
137  }
138 
139  // Otherwise return failure
140  return false;
141 }

143 const CBigNum CommitmentProofOfKnowledge::calculateChallenge(const CBigNum& a, const CBigNum& b, const CBigNum &commitOne, const CBigNum &commitTwo) const {
144  CHashWriter hasher(0,0);
145 
146  // Hash together the following elements:
147  // * A string identifying the proof
148  // * Commitment A
149  // * Commitment B
150  // * Ephemeral commitment T1
151  // * Ephemeral commitment T2
152  // * A serialized instance of the commitment A parameters
153  // * A serialized instance of the commitment B parameters
154 
155  hasher << std::string(ZEROCOIN_COMMITMENT_EQUALITY_PROOF);
156  hasher << commitOne;
157  hasher << std::string("||");
158  hasher << commitTwo;
159  hasher << std::string("||");
160  hasher << a;
161  hasher << std::string("||");
162  hasher << b;
163  hasher << std::string("||");
164  hasher << *(this->ap);
165  hasher << std::string("||");
166  hasher << *(this->bp);
167 
168  // Convert the SHA256 result into a Bignum
169  // Note that if we ever change the size of the hash function we will have
170  // to update COMMITMENT_EQUALITY_CHALLENGE_SIZE appropriately!
171  return CBigNum(hasher.GetHash());
172 }
