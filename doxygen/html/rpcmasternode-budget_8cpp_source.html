53 {
54  string strCommand;

60  throw runtime_error(

62  "\nVote or show current budgets\n"
63  "This command is depreciated, please see individual command documentation for future reference\n\n"

155 
156  if (pwalletMain->IsLocked())
157  throw JSONRPCError(RPC_WALLET_UNLOCK_NEEDED, "Error: Please enter the wallet passphrase with walletpassphrase first.");
158 
159  std::string strProposalName = SanitizeString(params[0].get_str());
160  if (strProposalName.size() > 20)
161  throw runtime_error("Invalid proposal name, limit of 20 characters.");
162 
163  std::string strURL = SanitizeString(params[1].get_str());
164  if (strURL.size() > 64)
165  throw runtime_error("Invalid url, limit of 64 characters.");
166 
167  int nPaymentCount = params[2].get_int();
168  if (nPaymentCount < 1)
169  throw runtime_error("Invalid payment count, must be more than zero.");
170 
171  // Start must be in the next budget cycle
172  if (pindexPrev != NULL) nBlockMin = pindexPrev->nHeight - pindexPrev->nHeight % GetBudgetPaymentCycleBlocks() + GetBudgetPaymentCycleBlocks();
173 
174  int nBlockStart = params[3].get_int();
175  if (nBlockStart % GetBudgetPaymentCycleBlocks() != 0) {
176  int nNext = pindexPrev->nHeight - pindexPrev->nHeight % GetBudgetPaymentCycleBlocks() + GetBudgetPaymentCycleBlocks();
177  throw runtime_error(strprintf("Invalid block start - must be a budget cycle block. Next valid block: %d", nNext));
178  }
179 
180  int nBlockEnd = nBlockStart + GetBudgetPaymentCycleBlocks() * nPaymentCount; // End must be AFTER current cycle
181 
182  if (nBlockStart < nBlockMin)
183  throw runtime_error("Invalid block start, must be more than current height.");
184 
185  if (nBlockEnd < pindexPrev->nHeight)
186  throw runtime_error("Invalid ending block, starting block + (payment_cycle*payments) must be more than current height.");
187 
188  CBitcoinAddress address(params[4].get_str());
189  if (!address.IsValid())
190  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid Divi address");
191 
192  // Parse Divi address
193  CScript scriptPubKey = GetScriptForDestination(address.Get());
194  CAmount nAmount = AmountFromValue(params[5]);
195 
196  //*************************************************************************
197 
198  // create transaction 15 minutes into the future, to allow for confirmation time
199  CBudgetProposalBroadcast budgetProposalBroadcast(strProposalName, strURL, nPaymentCount, scriptPubKey, nAmount, nBlockStart, 0);
200 
201  std::string strError = "";
202  if (!budgetProposalBroadcast.IsValid(strError, false))
203  throw runtime_error("Proposal is not valid - " + budgetProposalBroadcast.GetHash().ToString() + " - " + strError);
204 
205  bool useIX = false; //true;
206  // if (params.size() > 7) {
207  // if(params[7].get_str() != "false" && params[7].get_str() != "true")
208  // return "Invalid use_ix, must be true or false";
209  // useIX = params[7].get_str() == "true" ? true : false;
210  // }
211 
212  CWalletTx wtx;
213  if (!pwalletMain->GetBudgetSystemCollateralTX(wtx, budgetProposalBroadcast.GetHash(), useIX)) {
214  throw runtime_error("Error making collateral transaction for proposal. Please check your wallet balance.");
215  }
216 
217  // make our change address
218  CReserveKey reservekey(pwalletMain);
219  //send the tx to the network
220  pwalletMain->CommitTransaction(wtx, reservekey, useIX ? "ix" : "tx");
221 
222  return wtx.GetHash().ToString();
223 }
