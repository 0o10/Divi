34  }

37  uint256 bnTargetLimit = (~uint256(0) >> 24);

43  nActualSpacing = pindexLast->GetBlockTime() - pindexLast->pprev->GetBlockTime();
44

49  // ppcoin: retarget with exponential moving toward target spacing

51  bnNew.SetCompact(pindexLast->nBits);
52

54  bnNew *= ((nInterval - 1) * nTargetSpacing + nActualSpacing + nActualSpacing);
55  bnNew /= ((nInterval + 1) * nTargetSpacing);

133 uint256 GetBlockProof(const CBlockIndex& block)
134 {
135  uint256 bnTarget;
136  bool fNegative;
137  bool fOverflow;
138  bnTarget.SetCompact(block.nBits, &fNegative, &fOverflow);
139  if (fNegative || fOverflow || bnTarget == 0)
140  return 0;
141  // We need to compute 2**256 / (bnTarget+1), but we can't represent 2**256
142  // as it's too large for a uint256. However, as 2**256 is at least as large
143  // as bnTarget+1, it is equal to ((2**256 - bnTarget - 1) / (bnTarget+1)) + 1,
144  // or ~bnTarget / (nTarget+1) + 1.
145  return (~bnTarget / (bnTarget + 1)) + 1;
146 }
