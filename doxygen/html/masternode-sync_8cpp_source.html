32 bool CMasternodeSync::IsBlockchainSynced()
33 {
34  static bool fBlockchainSynced = false;
35  static int64_t lastProcess = GetTime();
36 
37  // if the last call to this function was more than 60 minutes ago (client was in sleep mode) reset the sync process
38  if (GetTime() - lastProcess > 60 * 60) {
39  Reset();
40  fBlockchainSynced = false;
41  }
42  lastProcess = GetTime();
43 
44  if (fBlockchainSynced) return true;
45 
46  if (fImporting || fReindex) return false;
47 
48  TRY_LOCK(cs_main, lockMain);
49  if (!lockMain) return false;
50 
51  CBlockIndex* pindex = chainActive.Tip();
52  if (pindex == NULL) return false;
53 
54 
55  if (pindex->nTime + 60 * 60 < GetTime())
56  return false;
57 
58  fBlockchainSynced = true;
59 
60  return true;
61 }

136 void CMasternodeSync::GetNextAsset()
137 {
138  switch (RequestedMasternodeAssets) {
139  case (MASTERNODE_SYNC_INITIAL):
140  case (MASTERNODE_SYNC_FAILED): // should never be used here actually, use Reset() instead
141  ClearFulfilledRequest();
142  RequestedMasternodeAssets = MASTERNODE_SYNC_SPORKS;
143  break;
144  case (MASTERNODE_SYNC_SPORKS):
145  RequestedMasternodeAssets = MASTERNODE_SYNC_LIST;
146  break;
147  case (MASTERNODE_SYNC_LIST):
148  RequestedMasternodeAssets = MASTERNODE_SYNC_MNW;
149  break;
150  case (MASTERNODE_SYNC_MNW):
151  RequestedMasternodeAssets = MASTERNODE_SYNC_BUDGET;
152  break;
153  case (MASTERNODE_SYNC_BUDGET):
154  LogPrintf("CMasternodeSync::GetNextAsset - Sync has finished\n");
155  RequestedMasternodeAssets = MASTERNODE_SYNC_FINISHED;
156  break;
157  }
158  RequestedMasternodeAttempt = 0;
159  nAssetSyncStarted = GetTime();
160 }

183 void CMasternodeSync::ProcessMessage(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
184 {
185  if (strCommand == "ssc") { //Sync status count
186  int nItemID;
187  int nCount;
188  vRecv >> nItemID >> nCount;
189 
190  if (RequestedMasternodeAssets >= MASTERNODE_SYNC_FINISHED) return;
191 
192  //this means we will receive no further communication
193  switch (nItemID) {
194  case (MASTERNODE_SYNC_LIST):
195  if (nItemID != RequestedMasternodeAssets) return;
196  sumMasternodeList += nCount;
197  countMasternodeList++;
198  break;
199  case (MASTERNODE_SYNC_MNW):
200  if (nItemID != RequestedMasternodeAssets) return;
201  sumMasternodeWinner += nCount;
202  countMasternodeWinner++;
203  break;
204  case (MASTERNODE_SYNC_BUDGET_PROP):
205  if (RequestedMasternodeAssets != MASTERNODE_SYNC_BUDGET) return;
206  sumBudgetItemProp += nCount;
207  countBudgetItemProp++;
208  break;
209  case (MASTERNODE_SYNC_BUDGET_FIN):
210  if (RequestedMasternodeAssets != MASTERNODE_SYNC_BUDGET) return;
211  sumBudgetItemFin += nCount;
212  countBudgetItemFin++;
213  break;
214  }
215 
216  LogPrint("masternode", "CMasternodeSync:ProcessMessage - ssc - got inventory count %d %d\n", nItemID, nCount);
217  }
218 }

233 void CMasternodeSync::Process()
234 {
235  static int tick = 0;
236 
237  if (tick++ % MASTERNODE_SYNC_TIMEOUT != 0) return;
238 
239  if (IsSynced()) {
240  /*
241  Resync if we lose all masternodes from sleep/wake or failure to sync originally
242  */
243  if (mnodeman.CountEnabled() == 0) {
244  Reset();
245  } else
246  return;
247  }
248 
249  //try syncing again
250  if (RequestedMasternodeAssets == MASTERNODE_SYNC_FAILED && lastFailure + (1 * 60) < GetTime()) {
251  Reset();
252  } else if (RequestedMasternodeAssets == MASTERNODE_SYNC_FAILED) {
253  return;
254  }
255 
256  LogPrint("masternode", "CMasternodeSync::Process() - tick %d RequestedMasternodeAssets %d\n", tick, RequestedMasternodeAssets);
257 
258  if (RequestedMasternodeAssets == MASTERNODE_SYNC_INITIAL) GetNextAsset();
259 
260  // sporks synced but blockchain is not, wait until we're almost at a recent block to continue
261  if (Params().NetworkID() != CBaseChainParams::REGTEST &&
262  !IsBlockchainSynced() && RequestedMasternodeAssets > MASTERNODE_SYNC_SPORKS) return;
263 
264  TRY_LOCK(cs_vNodes, lockRecv);
265  if (!lockRecv) return;
266 
267  BOOST_FOREACH (CNode* pnode, vNodes) {
268  if (Params().NetworkID() == CBaseChainParams::REGTEST) {
269  if (RequestedMasternodeAttempt <= 2) {
270  pnode->PushMessage("getsporks"); //get current network sporks
271  } else if (RequestedMasternodeAttempt < 4) {
272  mnodeman.DsegUpdate(pnode);
273  } else if (RequestedMasternodeAttempt < 6) {
274  int nMnCount = mnodeman.CountEnabled();
275  pnode->PushMessage("mnget", nMnCount); //sync payees
276  uint256 n = 0;
277  pnode->PushMessage("mnvs", n); //sync masternode votes
278  } else {
279  RequestedMasternodeAssets = MASTERNODE_SYNC_FINISHED;
280  }
281  RequestedMasternodeAttempt++;
282  return;
283  }
284 
285  //set to synced
286  if (RequestedMasternodeAssets == MASTERNODE_SYNC_SPORKS) {
287  if (pnode->HasFulfilledRequest("getspork")) continue;
288  pnode->FulfilledRequest("getspork");
289 
290  pnode->PushMessage("getsporks"); //get current network sporks
291  if (RequestedMasternodeAttempt >= 2) GetNextAsset();
292  RequestedMasternodeAttempt++;
293 
294  return;
295  }
296 
297  if (pnode->nVersion >= masternodePayments.GetMinMasternodePaymentsProto()) {
298  if (RequestedMasternodeAssets == MASTERNODE_SYNC_LIST) {
299  LogPrint("masternode", "CMasternodeSync::Process() - lastMasternodeList %lld (GetTime() - MASTERNODE_SYNC_TIMEOUT) %lld\n", lastMasternodeList, GetTime() - MASTERNODE_SYNC_TIMEOUT);
300  if (lastMasternodeList > 0 && lastMasternodeList < GetTime() - MASTERNODE_SYNC_TIMEOUT * 2 && RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD) { //hasn't received a new item in the last five seconds, so we'll move to the
301  GetNextAsset();
302  return;
303  }
304 
305  if (pnode->HasFulfilledRequest("mnsync")) continue;
306  pnode->FulfilledRequest("mnsync");
307 
308  // timeout
309  if (lastMasternodeList == 0 &&
310  (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3 || GetTime() - nAssetSyncStarted > MASTERNODE_SYNC_TIMEOUT * 5)) {
311  if (IsSporkActive(SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT)) {
312  LogPrintf("CMasternodeSync::Process - ERROR - Sync has failed, will retry later\n");
313  RequestedMasternodeAssets = MASTERNODE_SYNC_FAILED;
314  RequestedMasternodeAttempt = 0;
315  lastFailure = GetTime();
316  nCountFailures++;
317  } else {
318  GetNextAsset();
319  }
320  return;
321  }
322 
323  if (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3) return;
324 
325  mnodeman.DsegUpdate(pnode);
326  RequestedMasternodeAttempt++;
327  return;
328  }
329 
330  if (RequestedMasternodeAssets == MASTERNODE_SYNC_MNW) {
331  if (lastMasternodeWinner > 0 && lastMasternodeWinner < GetTime() - MASTERNODE_SYNC_TIMEOUT * 2 && RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD) { //hasn't received a new item in the last five seconds, so we'll move to the
332  GetNextAsset();
333  return;
334  }
335 
336  if (pnode->HasFulfilledRequest("mnwsync")) continue;
337  pnode->FulfilledRequest("mnwsync");
338 
339  // timeout
340  if (lastMasternodeWinner == 0 &&
341  (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3 || GetTime() - nAssetSyncStarted > MASTERNODE_SYNC_TIMEOUT * 5)) {
342  if (IsSporkActive(SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT)) {
343  LogPrintf("CMasternodeSync::Process - ERROR - Sync has failed, will retry later\n");
344  RequestedMasternodeAssets = MASTERNODE_SYNC_FAILED;
345  RequestedMasternodeAttempt = 0;
346  lastFailure = GetTime();
347  nCountFailures++;
348  } else {
349  GetNextAsset();
350  }
351  return;
352  }
353 
354  if (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3) return;
355 
356  CBlockIndex* pindexPrev = chainActive.Tip();
357  if (pindexPrev == NULL) return;
358 
359  int nMnCount = mnodeman.CountEnabled();
360  pnode->PushMessage("mnget", nMnCount); //sync payees
361  RequestedMasternodeAttempt++;
362 
363  return;
364  }
365  }
366 
367  if (pnode->nVersion >= ActiveProtocol()) {
368  if (RequestedMasternodeAssets == MASTERNODE_SYNC_BUDGET) {
369 
370  // We'll start rejecting votes if we accidentally get set as synced too soon
371  if (lastBudgetItem > 0 && lastBudgetItem < GetTime() - MASTERNODE_SYNC_TIMEOUT * 2 && RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD) {
372 
373  // Hasn't received a new item in the last five seconds, so we'll move to the
374  GetNextAsset();
375 
376  // Try to activate our masternode if possible
377  activeMasternode.ManageStatus();
378 
379  return;
380  }
381 
382  // timeout
383  if (lastBudgetItem == 0 &&
384  (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3 || GetTime() - nAssetSyncStarted > MASTERNODE_SYNC_TIMEOUT * 5)) {
385  // maybe there is no budgets at all, so just finish syncing
386  GetNextAsset();
387  activeMasternode.ManageStatus();
388  return;
389  }
390 
391  if (pnode->HasFulfilledRequest("busync")) continue;
392  pnode->FulfilledRequest("busync");
393 
394  if (RequestedMasternodeAttempt >= MASTERNODE_SYNC_THRESHOLD * 3) return;
395 
396  uint256 n = 0;
397  pnode->PushMessage("mnvs", n); //sync masternode votes
398  RequestedMasternodeAttempt++;
399 
400  return;
401  }
402  }
403  }
