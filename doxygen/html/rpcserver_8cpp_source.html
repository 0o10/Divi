161 string CRPCTable::help(string strCommand) const
162 {
163  string strRet;
164  string category;
165  set<rpcfn_type> setDone;
166  vector<pair<string, const CRPCCommand*> > vCommands;
167 
168  for (map<string, const CRPCCommand*>::const_iterator mi = mapCommands.begin(); mi != mapCommands.end(); ++mi)
169  vCommands.push_back(make_pair(mi->second->category + mi->first, mi->second));
170  sort(vCommands.begin(), vCommands.end());
171 
172  BOOST_FOREACH (const PAIRTYPE(string, const CRPCCommand*) & command, vCommands) {
173  const CRPCCommand* pcmd = command.second;
174  string strMethod = pcmd->name;
175  // We already filter duplicates, but these deprecated screw up the sort order
176  if (strMethod.find("label") != string::npos)
177  continue;
178  if ((strCommand != "" || pcmd->category == "hidden") && strMethod != strCommand)
179  continue;
180 #ifdef ENABLE_WALLET
181  if (pcmd->reqWallet && !pwalletMain)
182  continue;
183 #endif
184 
185  try {
186  Array params;
187  rpcfn_type pfn = pcmd->actor;
188  if (setDone.insert(pfn).second)
189  (*pfn)(params, true);
190  } catch (std::exception& e) {
191  // Help text is returned in an exception
192  string strHelp = string(e.what());
193  if (strCommand == "") {
194  if (strHelp.find('\n') != string::npos)
195  strHelp = strHelp.substr(0, strHelp.find('\n'));
196 
197  if (category != pcmd->category) {
198  if (!category.empty())
199  strRet += "\n";
200  category = pcmd->category;
201  string firstLetter = category.substr(0, 1);
202  boost::to_upper(firstLetter);
203  strRet += "== " + firstLetter + category.substr(1) + " ==\n";
204  }
205  }
206  strRet += strHelp + "\n";
207  }
208  }
209  if (strRet == "")
210  strRet = strprintf("help: unknown command: %s\n", strCommand);
211  strRet = strRet.substr(0, strRet.size() - 1);
212  return strRet;
213 }

462 CNetAddr BoostAsioToCNetAddr(boost::asio::ip::address address)
463 {
464  CNetAddr netaddr;
465  // Make sure that IPv4-compatible and IPv4-mapped IPv6 addresses are treated as IPv4 addresses
466  if (address.is_v6() && (address.to_v6().is_v4_compatible() || address.to_v6().is_v4_mapped()))
467  address = address.to_v6().to_v4();
468 
469  if (address.is_v4()) {

471  netaddr.SetRaw(NET_IPV4, &bytes[0]);

473  boost::asio::ip::address_v6::bytes_type bytes = address.to_v6().to_bytes();
474  netaddr.SetRaw(NET_IPV6, &bytes[0]);
475  }
