41 bool Solver(const CScript& scriptPubKey, txnouttype& typeRet, vector<vector<unsigned char> >& vSolutionsRet)
42 {
43  // Templates
44  static multimap<txnouttype, CScript> mTemplates;
45  if (mTemplates.empty())
46  {
47  // Standard tx, sender provides pubkey, receiver adds signature
48  mTemplates.insert(make_pair(TX_PUBKEY, CScript() << OP_PUBKEY << OP_CHECKSIG));
49 
50  // Bitcoin address tx, sender provides hash of pubkey, receiver provides signature and pubkey
51  mTemplates.insert(make_pair(TX_PUBKEYHASH, CScript() << OP_DUP << OP_HASH160 << OP_PUBKEYHASH << OP_EQUALVERIFY << OP_CHECKSIG));
52 
53  // Sender provides N pubkeys, receivers provides M signatures
54  mTemplates.insert(make_pair(TX_MULTISIG, CScript() << OP_SMALLINTEGER << OP_PUBKEYS << OP_SMALLINTEGER << OP_CHECKMULTISIG));
55  }
56 
57  // Shortcut for pay-to-script-hash, which are more constrained than the other types:
58  // it is always OP_HASH160 20 [20 byte hash] OP_EQUAL
59  if (scriptPubKey.IsPayToScriptHash())
60  {
61  typeRet = TX_SCRIPTHASH;
62  vector<unsigned char> hashBytes(scriptPubKey.begin()+2, scriptPubKey.begin()+22);
63  vSolutionsRet.push_back(hashBytes);
64  return true;
65  }
66 
67  // Zerocoin
68  if (scriptPubKey.IsZerocoinMint()){
69  typeRet = TX_ZEROCOINMINT;
70  if(scriptPubKey.size() > 150) return false;
71  vector<unsigned char> hashBytes(scriptPubKey.begin()+2, scriptPubKey.end());
72  vSolutionsRet.push_back(hashBytes);
73  return true;
74  }
75 
76  // Provably prunable, data-carrying output
77  //
78  // So long as script passes the IsUnspendable() test and all but the first
79  // byte passes the IsPushOnly() test we don't care what exactly is in the
80  // script.
81  if (scriptPubKey.size() >= 1 && scriptPubKey[0] == OP_RETURN && scriptPubKey.IsPushOnly(scriptPubKey.begin()+1)) {
82  typeRet = TX_NULL_DATA;
83  return true;
84  }
85 
86  // Scan templates
87  const CScript& script1 = scriptPubKey;
88  BOOST_FOREACH(const PAIRTYPE(txnouttype, CScript)& tplate, mTemplates)
89  {
90  const CScript& script2 = tplate.second;
91  vSolutionsRet.clear();
92 
93  opcodetype opcode1, opcode2;
94  vector<unsigned char> vch1, vch2;
95 
96  // Compare
97  CScript::const_iterator pc1 = script1.begin();
98  CScript::const_iterator pc2 = script2.begin();
99  while (true)
100  {
101  if (pc1 == script1.end() && pc2 == script2.end())
102  {
103  // Found a match
104  typeRet = tplate.first;
105  if (typeRet == TX_MULTISIG)
106  {
107  // Additional checks for TX_MULTISIG:
108  unsigned char m = vSolutionsRet.front()[0];
109  unsigned char n = vSolutionsRet.back()[0];
110  if (m < 1 || n < 1 || m > n || vSolutionsRet.size()-2 != n)
111  return false;
112  }
113  return true;
114  }
115  if (!script1.GetOp(pc1, opcode1, vch1))
116  break;
117  if (!script2.GetOp(pc2, opcode2, vch2))
118  break;
119 
120  // Template matching opcodes:
121  if (opcode2 == OP_PUBKEYS)
122  {
123  while (vch1.size() >= 33 && vch1.size() <= 65)
124  {
125  vSolutionsRet.push_back(vch1);
126  if (!script1.GetOp(pc1, opcode1, vch1))
127  break;
128  }
129  if (!script2.GetOp(pc2, opcode2, vch2))
130  break;
131  // Normal situation is to fall through
132  // to other if/else statements
133  }
134 
135  if (opcode2 == OP_PUBKEY)
136  {
137  if (vch1.size() < 33 || vch1.size() > 65)
138  break;
139  vSolutionsRet.push_back(vch1);
140  }
141  else if (opcode2 == OP_PUBKEYHASH)
142  {
143  if (vch1.size() != sizeof(uint160))
144  break;
145  vSolutionsRet.push_back(vch1);
146  }
147  else if (opcode2 == OP_SMALLINTEGER)
148  { // Single-byte small integer pushed onto vSolutions
149  if (opcode1 == OP_0 ||
150  (opcode1 >= OP_1 && opcode1 <= OP_16))
151  {
152  char n = (char)CScript::DecodeOP_N(opcode1);
153  vSolutionsRet.push_back(valtype(1, n));
154  }
155  else
156  break;
157  }
158  else if (opcode1 != opcode2 || vch1 != vch2)
159  {
160  // Others must match exactly
161  break;
162  }
163  }
164  }
165 
166  vSolutionsRet.clear();
167  typeRet = TX_NONSTANDARD;
168  return false;
169 }

193 bool IsStandard(const CScript& scriptPubKey, txnouttype& whichType)
194 {
195  vector<valtype> vSolutions;
196  if (!Solver(scriptPubKey, whichType, vSolutions))
197  return false;
198 
199  if (whichType == TX_MULTISIG)
200  {
201  unsigned char m = vSolutions.front()[0];
202  unsigned char n = vSolutions.back()[0];
203  // Support up to x-of-3 multisig txns as standard
204  if (n < 1 || n > 3)
205  return false;
206  if (m < 1 || m > n)
207  return false;
208  } else if (whichType == TX_NULL_DATA &&
209  (!GetBoolArg("-datacarrier", true) || scriptPubKey.size() > nMaxDatacarrierBytes))
210  return false;
211 
212  return whichType != TX_NONSTANDARD;
213 }

215 bool ExtractDestination(const CScript& scriptPubKey, CTxDestination& addressRet)
216 {
217  vector<valtype> vSolutions;
218  txnouttype whichType;
219  if (!Solver(scriptPubKey, whichType, vSolutions))
220  return false;
221 
222  if (whichType == TX_PUBKEY)
223  {
224  CPubKey pubKey(vSolutions[0]);
225  if (!pubKey.IsValid())
226  return false;
227 
228  addressRet = pubKey.GetID();
229  return true;
230  }
231  else if (whichType == TX_PUBKEYHASH)
232  {
233  addressRet = CKeyID(uint160(vSolutions[0]));
234  return true;
235  }
236  else if (whichType == TX_SCRIPTHASH)
237  {
238  addressRet = CScriptID(uint160(vSolutions[0]));
239  return true;
240  }
241  // Multisig txns have more than one address...
242  return false;
243 }

245 bool ExtractDestinations(const CScript& scriptPubKey, txnouttype& typeRet, vector<CTxDestination>& addressRet, int& nRequiredRet)
246 {
247  addressRet.clear();
248  typeRet = TX_NONSTANDARD;
249  vector<valtype> vSolutions;
250  if (!Solver(scriptPubKey, typeRet, vSolutions))
251  return false;
252  if (typeRet == TX_NULL_DATA){
253  // This is data, not addresses
254  return false;
255  }
256 
257  if (typeRet == TX_MULTISIG)
258  {
259  nRequiredRet = vSolutions.front()[0];
260  for (unsigned int i = 1; i < vSolutions.size()-1; i++)
261  {
262  CPubKey pubKey(vSolutions[i]);
263  if (!pubKey.IsValid())
264  continue;
265 
266  CTxDestination address = pubKey.GetID();
267  addressRet.push_back(address);
268  }
269 
270  if (addressRet.empty())
271  return false;
272  }
273  else
274  {
275  nRequiredRet = 1;
276  CTxDestination address;
277  if (!ExtractDestination(scriptPubKey, address))
278  return false;
279  addressRet.push_back(address);
280  }
281 
282  return true;
283 }
