60 void TxToJSON(const CTransaction& tx, const uint256 hashBlock, Object& entry)
61 {
62  entry.push_back(Pair("txid", tx.GetHash().GetHex()));
63  entry.push_back(Pair("version", tx.nVersion));
64  entry.push_back(Pair("locktime", (int64_t)tx.nLockTime));
65  Array vin;
66  BOOST_FOREACH (const CTxIn& txin, tx.vin) {
67  Object in;
68  if (tx.IsCoinBase())
69  in.push_back(Pair("coinbase", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
70  else {
71  in.push_back(Pair("txid", txin.prevout.hash.GetHex()));
72  in.push_back(Pair("vout", (int64_t)txin.prevout.n));
73  Object o;
74  o.push_back(Pair("asm", txin.scriptSig.ToString()));
75  o.push_back(Pair("hex", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
76  in.push_back(Pair("scriptSig", o));
77  }
78  in.push_back(Pair("sequence", (int64_t)txin.nSequence));
79  vin.push_back(in);
80  }
81  entry.push_back(Pair("vin", vin));
82  Array vout;
83  for (unsigned int i = 0; i < tx.vout.size(); i++) {
84  const CTxOut& txout = tx.vout[i];
85  Object out;
86  out.push_back(Pair("value", ValueFromAmount(txout.nValue)));
87  out.push_back(Pair("n", (int64_t)i));
88  Object o;
89  ScriptPubKeyToJSON(txout.scriptPubKey, o, true);
90  out.push_back(Pair("scriptPubKey", o));
91  vout.push_back(out);
92  }
93  entry.push_back(Pair("vout", vout));
94 
95  if (hashBlock != 0) {
96  entry.push_back(Pair("blockhash", hashBlock.GetHex()));
97  BlockMap::iterator mi = mapBlockIndex.find(hashBlock);
98  if (mi != mapBlockIndex.end() && (*mi).second) {
99  CBlockIndex* pindex = (*mi).second;
100  if (chainActive.Contains(pindex)) {
101  entry.push_back(Pair("confirmations", 1 + chainActive.Height() - pindex->nHeight));
102  entry.push_back(Pair("time", pindex->GetBlockTime()));
103  entry.push_back(Pair("blocktime", pindex->GetBlockTime()));
104  } else
105  entry.push_back(Pair("confirmations", 0));
106  }
107  }
108 }
