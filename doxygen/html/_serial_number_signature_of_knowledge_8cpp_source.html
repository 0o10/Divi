20 // Use one 256 bit seed and concatenate 4 unique 256 bit hashes to make a 1024 bit hash
21 CBigNum SeedTo1024(uint256 hashSeed) {
22  CHashWriter hasher(0,0);
23  hasher << hashSeed;
24 
25  vector<unsigned char> vResult;
26  for (int i = 0; i < 4; i ++) {
27  vector<unsigned char> vHash = CBigNum(hasher.GetHash()).getvch();
28  vResult.insert(vResult.end(), vHash.begin(), vHash.end());
29  hasher << vResult;
30  }
31 
32  CBigNum bnResult;
33  bnResult.setvch(vResult);
34  return bnResult;
35 }

37 SerialNumberSignatureOfKnowledge::SerialNumberSignatureOfKnowledge(const
38  ZerocoinParams* p, const PrivateCoin& coin, const Commitment& commitmentToCoin,
39  uint256 msghash):params(p),
40  s_notprime(p->zkp_iterations),
41  sprime(p->zkp_iterations) {
42 
43  // Sanity check: verify that the order of the "accumulatedValueCommitmentGroup" is
44  // equal to the modulus of "coinCommitmentGroup". Otherwise we will produce invalid
45  // proofs.
46  if (params->coinCommitmentGroup.modulus != params->serialNumberSoKCommitmentGroup.groupOrder) {
47  throw std::runtime_error("Groups are not structured correctly.");
48  }
49 
50  CBigNum a = params->coinCommitmentGroup.g;
51  CBigNum b = params->coinCommitmentGroup.h;
52  CBigNum g = params->serialNumberSoKCommitmentGroup.g;
53  CBigNum h = params->serialNumberSoKCommitmentGroup.h;
54 
55  CHashWriter hasher(0,0);
56  hasher << *params << commitmentToCoin.getCommitmentValue() << coin.getSerialNumber() << msghash;
57 
58  vector<CBigNum> r(params->zkp_iterations);
59  vector<CBigNum> v_seed(params->zkp_iterations);
60  vector<CBigNum> v_expanded(params->zkp_iterations);
61  vector<CBigNum> c(params->zkp_iterations);
62 
63  for(uint32_t i=0; i < params->zkp_iterations; i++) {
64  r[i] = CBigNum::randBignum(params->coinCommitmentGroup.groupOrder);
65 
66  //use a random 256 bit seed that expands to 1024 bit for v[i]
67  while (true) {
68  uint256 hashRand = CBigNum::randBignum(CBigNum(~uint256(0))).getuint256();
69  CBigNum bnExpanded = SeedTo1024(hashRand);
70 
71  if(bnExpanded > params->serialNumberSoKCommitmentGroup.groupOrder)
72  continue;
73 
74  v_seed[i] = CBigNum(hashRand);
75  v_expanded[i] = bnExpanded;
76  break;
77  }
78  }
79 
80  for(uint32_t i=0; i < params->zkp_iterations; i++) {
81  // compute g^{ {a^x b^r} h^v} mod p2
82  c[i] = challengeCalculation(coin.getSerialNumber(), r[i], v_expanded[i]);
83  }
84 
85  // We can't hash data in parallel either
86  // because OPENMP cannot not guarantee loops
87  // execute in order.
88  for(uint32_t i=0; i < params->zkp_iterations; i++) {
89  hasher << c[i];
90  }
91  this->hash = hasher.GetHash();
92  unsigned char *hashbytes = (unsigned char*) &hash;
93 
94  for(uint32_t i = 0; i < params->zkp_iterations; i++) {
95  int bit = i % 8;
96  int byte = i / 8;
97 
98  bool challenge_bit = ((hashbytes[byte] >> bit) & 0x01);
99  if (challenge_bit) {
100  s_notprime[i] = r[i];
101  sprime[i] = v_seed[i];
102  } else {
103  s_notprime[i] = r[i] - coin.getRandomness();
104  sprime[i] = v_expanded[i] - (commitmentToCoin.getRandomness() *
105  b.pow_mod(r[i] - coin.getRandomness(), params->serialNumberSoKCommitmentGroup.groupOrder));
106  }
107  }

110 inline CBigNum SerialNumberSignatureOfKnowledge::challengeCalculation(const CBigNum& a_exp,const CBigNum& b_exp,
111  const CBigNum& h_exp) const {
112 
113  CBigNum a = params->coinCommitmentGroup.g;
114  CBigNum b = params->coinCommitmentGroup.h;
115  CBigNum g = params->serialNumberSoKCommitmentGroup.g;
116  CBigNum h = params->serialNumberSoKCommitmentGroup.h;
117 
118  CBigNum exponent = (a.pow_mod(a_exp, params->serialNumberSoKCommitmentGroup.groupOrder)
119  * b.pow_mod(b_exp, params->serialNumberSoKCommitmentGroup.groupOrder)) % params->serialNumberSoKCommitmentGroup.groupOrder;
120 
121  return (g.pow_mod(exponent, params->serialNumberSoKCommitmentGroup.modulus) * h.pow_mod(h_exp, params->serialNumberSoKCommitmentGroup.modulus)) % params->serialNumberSoKCommitmentGroup.modulus;
122 }

124 bool SerialNumberSignatureOfKnowledge::Verify(const CBigNum& coinSerialNumber, const CBigNum& valueOfCommitmentToCoin,
125  const uint256 msghash) const {
126  CBigNum a = params->coinCommitmentGroup.g;
127  CBigNum b = params->coinCommitmentGroup.h;
128  CBigNum g = params->serialNumberSoKCommitmentGroup.g;
129  CBigNum h = params->serialNumberSoKCommitmentGroup.h;
130  CHashWriter hasher(0,0);
131  hasher << *params << valueOfCommitmentToCoin << coinSerialNumber << msghash;
132 
133  vector<CBigNum> tprime(params->zkp_iterations);
134  unsigned char *hashbytes = (unsigned char*) &this->hash;
135 
136  for(uint32_t i = 0; i < params->zkp_iterations; i++) {
137  int bit = i % 8;
138  int byte = i / 8;
139  bool challenge_bit = ((hashbytes[byte] >> bit) & 0x01);
140  if(challenge_bit) {
141  tprime[i] = challengeCalculation(coinSerialNumber, s_notprime[i], SeedTo1024(sprime[i].getuint256()));
142  } else {
143  CBigNum exp = b.pow_mod(s_notprime[i], params->serialNumberSoKCommitmentGroup.groupOrder);
144  tprime[i] = ((valueOfCommitmentToCoin.pow_mod(exp, params->serialNumberSoKCommitmentGroup.modulus) % params->serialNumberSoKCommitmentGroup.modulus) *
145  (h.pow_mod(sprime[i], params->serialNumberSoKCommitmentGroup.modulus) % params->serialNumberSoKCommitmentGroup.modulus)) %
146  params->serialNumberSoKCommitmentGroup.modulus;
147  }
148  }
149  for(uint32_t i = 0; i < params->zkp_iterations; i++) {
150  hasher << tprime[i];
151  }
152  return hasher.GetHash() == hash;
153 }
