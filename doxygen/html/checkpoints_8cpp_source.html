44 double GuessVerificationProgress(CBlockIndex* pindex, bool fSigchecks)
45 {
46  if (pindex == NULL)
47  return 0.0;
48 
49  int64_t nNow = time(NULL);
50 
51  double fSigcheckVerificationFactor = fSigchecks ? SIGCHECK_VERIFICATION_FACTOR : 1.0;
52  double fWorkBefore = 0.0; // Amount of work done before pindex
53  double fWorkAfter = 0.0; // Amount of work left after pindex (estimated)
54  // Work is defined as: 1.0 per transaction before the last checkpoint, and
55  // fSigcheckVerificationFactor per transaction after.
56 
57  const CCheckpointData& data = Params().Checkpoints();
58 
59  if (pindex->nChainTx <= data.nTransactionsLastCheckpoint) {
60  double nCheapBefore = pindex->nChainTx;
61  double nCheapAfter = data.nTransactionsLastCheckpoint - pindex->nChainTx;
62  double nExpensiveAfter = (nNow - data.nTimeLastCheckpoint) / 86400.0 * data.fTransactionsPerDay;
63  fWorkBefore = nCheapBefore;
64  fWorkAfter = nCheapAfter + nExpensiveAfter * fSigcheckVerificationFactor;
65  } else {
66  double nCheapBefore = data.nTransactionsLastCheckpoint;
67  double nExpensiveBefore = pindex->nChainTx - data.nTransactionsLastCheckpoint;
68  double nExpensiveAfter = (nNow - pindex->GetBlockTime()) / 86400.0 * data.fTransactionsPerDay;
69  fWorkBefore = nCheapBefore + nExpensiveBefore * fSigcheckVerificationFactor;
70  fWorkAfter = nExpensiveAfter * fSigcheckVerificationFactor;
71  }
72 
73  return fWorkBefore / (fWorkBefore + fWorkAfter);
74 }
