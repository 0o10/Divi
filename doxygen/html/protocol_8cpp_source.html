58 bool CMessageHeader::IsValid() const
59 {
60  // Check start string
61  if (memcmp(pchMessageStart, Params().MessageStart(), MESSAGE_START_SIZE) != 0)
62  return false;
63 
64  // Check the command string for errors
65  for (const char* p1 = pchCommand; p1 < pchCommand + COMMAND_SIZE; p1++) {
66  if (*p1 == 0) {
67  // Must be all zeros after the first zero
68  for (; p1 < pchCommand + COMMAND_SIZE; p1++)
69  if (*p1 != 0)
70  return false;
71  } else if (*p1 < ' ' || *p1 > 0x7E)
72  return false;
73  }
74 
75  // Message size
76  if (nMessageSize > MAX_SIZE) {
77  LogPrintf("CMessageHeader::IsValid() : (%s, %u bytes) nMessageSize > MAX_SIZE\n", GetCommand(), nMessageSize);
78  return false;
79  }
80 
81  return true;
82 }
