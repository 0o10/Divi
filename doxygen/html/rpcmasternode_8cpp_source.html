22 void SendMoney(const CTxDestination& address, CAmount nValue, CWalletTx& wtxNew, AvailableCoinsType coin_type = ALL_COINS)
23 {
24  // Check amount
25  if (nValue <= 0)
26  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid amount");
27 
28  if (nValue > pwalletMain->GetBalance())
29  throw JSONRPCError(RPC_WALLET_INSUFFICIENT_FUNDS, "Insufficient funds");
30 
31  string strError;
32  if (pwalletMain->IsLocked()) {
33  strError = "Error: Wallet locked, unable to create transaction!";
34  LogPrintf("SendMoney() : %s", strError);
35  throw JSONRPCError(RPC_WALLET_ERROR, strError);
36  }
37 
38  // Parse Divx address
39  CScript scriptPubKey = GetScriptForDestination(address);
40 
41  // Create and send the transaction
42  CReserveKey reservekey(pwalletMain);
43  CAmount nFeeRequired;
44  if (!pwalletMain->CreateTransaction(scriptPubKey, nValue, wtxNew, reservekey, nFeeRequired, strError, NULL, coin_type)) {
45  if (nValue + nFeeRequired > pwalletMain->GetBalance())
46  strError = strprintf("Error: This transaction requires a transaction fee of at least %s because of its amount, complexity, or use of recently received funds!", FormatMoney(nFeeRequired));
47  LogPrintf("SendMoney() : %s\n", strError);
48  throw JSONRPCError(RPC_WALLET_ERROR, strError);
49  }
50  if (!pwalletMain->CommitTransaction(wtxNew, reservekey))
51  throw JSONRPCError(RPC_WALLET_ERROR, "Error: The transaction was rejected! This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
52 }

238 Value listmasternodes(const Array& params, bool fHelp)
239 {
240  std::string strFilter = "";
241 
242  if (params.size() == 1) strFilter = params[0].get_str();
243 
244  if (fHelp || (params.size() > 1))
245  throw runtime_error(
246  "listmasternodes ( \"filter\" )\n"
247  "\nGet a ranked list of masternodes\n"
248 
249  "\nArguments:\n"
250  "1. \"filter\" (string, optional) Filter search text. Partial match by txhash, status, or addr.\n"
251 
252  "\nResult:\n"
253  "[\n"
254  " {\n"
255  " \"rank\": n, (numeric) Masternode Rank (or 0 if not enabled)\n"
256  " \"txhash\": \"hash\", (string) Collateral transaction hash\n"
257  " \"outidx\": n, (numeric) Collateral transaction output index\n"
258  " \"status\": s, (string) Status (ENABLED/EXPIRED/REMOVE/etc)\n"
259  " \"addr\": \"addr\", (string) Masternode DIVX address\n"
260  " \"version\": v, (numeric) Masternode protocol version\n"
261  " \"lastseen\": ttt, (numeric) The time in seconds since epoch (Jan 1 1970 GMT) of the last seen\n"
262  " \"activetime\": ttt, (numeric) The time in seconds since epoch (Jan 1 1970 GMT) masternode has been active\n"
263  " \"lastpaid\": ttt, (numeric) The time in seconds since epoch (Jan 1 1970 GMT) masternode was last paid\n"
264  " }\n"
265  " ,...\n"
266  "]\n"
267  "\nExamples:\n" +
268  HelpExampleCli("masternodelist", "") + HelpExampleRpc("masternodelist", ""));
269 
270  Array ret;
271  int nHeight;
272  {
273  LOCK(cs_main);
274  CBlockIndex* pindex = chainActive.Tip();
275  if(!pindex) return 0;
276  nHeight = pindex->nHeight;
277  }
278  std::vector<pair<int, CMasternode> > vMasternodeRanks = mnodeman.GetMasternodeRanks(nHeight);
279  BOOST_FOREACH (PAIRTYPE(int, CMasternode) & s, vMasternodeRanks) {
280  Object obj;
281  std::string strVin = s.second.vin.prevout.ToStringShort();
282  std::string strTxHash = s.second.vin.prevout.hash.ToString();
283  uint32_t oIdx = s.second.vin.prevout.n;
284 
285  CMasternode* mn = mnodeman.Find(s.second.vin);
286 
287  if (mn != NULL) {
288  if (strFilter != "" && strTxHash.find(strFilter) == string::npos &&
289  mn->Status().find(strFilter) == string::npos &&
290  CBitcoinAddress(mn->pubKeyCollateralAddress.GetID()).ToString().find(strFilter) == string::npos) continue;
291 
292  std::string strStatus = mn->Status();
293  std::string strHost;
294  int port;
295  SplitHostPort(mn->addr.ToString(), port, strHost);
296  CNetAddr node = CNetAddr(strHost, false);
297  std::string strNetwork = GetNetworkName(node.GetNetwork());
298 
299  obj.push_back(Pair("rank", (strStatus == "ENABLED" ? s.first : 0)));
300  obj.push_back(Pair("network", strNetwork));
301  obj.push_back(Pair("txhash", strTxHash));
302  obj.push_back(Pair("outidx", (uint64_t)oIdx));
303  obj.push_back(Pair("status", strStatus));
304  obj.push_back(Pair("addr", CBitcoinAddress(mn->pubKeyCollateralAddress.GetID()).ToString()));
305  obj.push_back(Pair("version", mn->protocolVersion));
306  obj.push_back(Pair("lastseen", (int64_t)mn->lastPing.sigTime));
307  obj.push_back(Pair("activetime", (int64_t)(mn->lastPing.sigTime - mn->sigTime)));
308  obj.push_back(Pair("lastpaid", (int64_t)mn->GetLastPaid()));
309 
310  ret.push_back(obj);
311  }
312  }
313 
314  return ret;
315 }

443  strCommand = params[0].get_str();
444 
445  // Backwards compatibility with legacy 'masternode' super-command forwarder

449  if (strCommand == "start-many") strCommand = "many";

451  if (strCommand == "start-disabled") strCommand = "disabled";

455  (params.size() == 2 && (strCommand != "local" && strCommand != "all" && strCommand != "many" && strCommand != "missing" && strCommand != "disabled")) ||

465 
466  "\nResult: (for 'local' set):\n"
467  "\"status\" (string) Masternode status message\n"

469  "\nResult: (for other sets):\n"

472  " \"detail\": [\n"

476  " \"error\": \"xxxx\" (string) Error message, if failed\n"
477  " }\n"
478  " ,...\n"
479  " ]\n"
480  "}\n"
481  "\nExamples:\n" +
482  HelpExampleCli("startmasternode", "\"alias\" \"0\" \"my_mn\"") + HelpExampleRpc("startmasternode", "\"alias\" \"0\" \"my_mn\""));
483 
484  bool fLock = (params[1].get_str() == "true" ? true : false);
485 
486  if (strCommand == "local") {
487  if (!fMasterNode) throw runtime_error("you must set masternode=1 in the configuration\n");
488 
489  if (pwalletMain->IsLocked())
490  throw JSONRPCError(RPC_WALLET_UNLOCK_NEEDED, "Error: Please enter the wallet passphrase with walletpassphrase first.");
491 
492  if (activeMasternode.status != ACTIVE_MASTERNODE_STARTED) {
493  activeMasternode.status = ACTIVE_MASTERNODE_INITIAL; // TODO: consider better way
494  activeMasternode.ManageStatus();
495  if (fLock)
496  pwalletMain->Lock();
497  }
498 
499  return activeMasternode.GetStatus();
500  }
501 
502  if (strCommand == "all" || strCommand == "many" || strCommand == "missing" || strCommand == "disabled") {
503  if (pwalletMain->IsLocked())
504  throw JSONRPCError(RPC_WALLET_UNLOCK_NEEDED, "Error: Please enter the wallet passphrase with walletpassphrase first.");
505 
506  if ((strCommand == "missing" || strCommand == "disabled") &&
507  (masternodeSync.RequestedMasternodeAssets <= MASTERNODE_SYNC_LIST ||
508  masternodeSync.RequestedMasternodeAssets == MASTERNODE_SYNC_FAILED)) {
509  throw runtime_error("You can't use this command until masternode list is synced\n");
510  }
511 
512  std::vector<CMasternodeConfig::CMasternodeEntry> mnEntries;
513  mnEntries = masternodeConfig.getEntries();
514 
515  int successful = 0;
516  int failed = 0;
517 
518  Array resultsObj;
519 
520  BOOST_FOREACH (CMasternodeConfig::CMasternodeEntry mne, masternodeConfig.getEntries()) {
521  std::string errorMessage;
522  int nIndex;
523  if(!mne.castOutputIndex(nIndex))
524  continue;
525  CTxIn vin = CTxIn(uint256(mne.getTxHash()), uint32_t(nIndex));
526  CMasternode* pmn = mnodeman.Find(vin);
527 
528  if (pmn != NULL) {
529  if (strCommand == "missing") continue;
530  if (strCommand == "disabled" && pmn->IsEnabled()) continue;
531  }
532 
533  bool result = activeMasternode.Register(mne.getIp(), mne.getPrivKey(), mne.getTxHash(), mne.getOutputIndex(), errorMessage);
534 
535  Object statusObj;
536  statusObj.push_back(Pair("alias", mne.getAlias()));
537  statusObj.push_back(Pair("result", result ? "success" : "failed"));
538 
539  if (result) {
540  successful++;
541  statusObj.push_back(Pair("error", ""));
542  } else {
543  failed++;
544  statusObj.push_back(Pair("error", errorMessage));
545  }
546 
547  resultsObj.push_back(statusObj);
548  }
549  if (fLock)
550  pwalletMain->Lock();
551 
552  Object returnObj;
553  returnObj.push_back(Pair("overall", strprintf("Successfully started %d masternodes, failed to start %d, total %d", successful, failed, successful + failed)));
554  returnObj.push_back(Pair("detail", resultsObj));
555 
556  return returnObj;
557  }
558
