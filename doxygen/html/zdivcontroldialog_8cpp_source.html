43 //Update the tree widget
44 void ZDivControlDialog::updateList()
45 {
46  // need to prevent the slot from being called each time something is changed
47  ui->treeWidget->blockSignals(true);
48  ui->treeWidget->clear();
49 
50  // add a top level item for each denomination
51  QFlags<Qt::ItemFlag> flgTristate = Qt::ItemIsEnabled | Qt::ItemIsUserCheckable | Qt::ItemIsTristate;
52  map<libzerocoin::CoinDenomination, int> mapDenomPosition;
53  for (auto denom : libzerocoin::zerocoinDenomList) {
54  QTreeWidgetItem* itemDenom(new QTreeWidgetItem);
55  ui->treeWidget->addTopLevelItem(itemDenom);
56 
57  //keep track of where this is positioned in tree widget
58  mapDenomPosition[denom] = ui->treeWidget->indexOfTopLevelItem(itemDenom);
59 
60  itemDenom->setFlags(flgTristate);
61  itemDenom->setText(COLUMN_DENOMINATION, QString::number(denom));
62  }
63 
64  // select all unused coins - including not mature. Update status of coins too.
65  std::list<CZerocoinMint> list;
66  model->listZerocoinMints(list, true, false, true);
67  this->listMints = list;
68 
69  //populate rows with mint info
70  for(const CZerocoinMint mint : listMints) {
71  // assign this mint to the correct denomination in the tree view
72  libzerocoin::CoinDenomination denom = mint.GetDenomination();
73  QTreeWidgetItem *itemMint = new QTreeWidgetItem(ui->treeWidget->topLevelItem(mapDenomPosition.at(denom)));
74 
75  // if the mint is already selected, then it needs to have the checkbox checked
76  std::string strPubCoin = mint.GetValue().GetHex();
77  itemMint->setCheckState(COLUMN_CHECKBOX, Qt::Unchecked);
78  if(count(listSelectedMints.begin(), listSelectedMints.end(), strPubCoin))
79  itemMint->setCheckState(COLUMN_CHECKBOX, Qt::Checked);
80 
81  itemMint->setText(COLUMN_DENOMINATION, QString::number(mint.GetDenomination()));
82  itemMint->setText(COLUMN_PUBCOIN, QString::fromStdString(strPubCoin));
83 
84  int nConfirmations = (mint.GetHeight() ? chainActive.Height() - mint.GetHeight() : 0);
85  if (nConfirmations < 0) {
86  // Sanity check
87  nConfirmations = 0;
88  }
89 
90  itemMint->setText(COLUMN_CONFIRMATIONS, QString::number(nConfirmations));
91 
92  // check to make sure there are at least 3 other mints added to the accumulators after this
93  int nMintsAdded = 0;
94  if(mint.GetHeight() != 0 && mint.GetHeight() < chainActive.Height() - 2) {
95  CBlockIndex *pindex = chainActive[mint.GetHeight() + 1];
96  while(pindex->nHeight < chainActive.Height() - 30) { // 30 just to make sure that its at least 2 checkpoints from the top block
97  nMintsAdded += count(pindex->vMintDenominationsInBlock.begin(), pindex->vMintDenominationsInBlock.end(), mint.GetDenomination());
98  if(nMintsAdded >= Params().Zerocoin_RequiredAccumulation())
99  break;
100  pindex = chainActive[pindex->nHeight + 1];
101  }
102  }
103 
104  // disable selecting this mint if it is not spendable - also display a reason why
105  bool fSpendable = nMintsAdded >= Params().Zerocoin_RequiredAccumulation() && nConfirmations >= Params().Zerocoin_MintRequiredConfirmations();
106  if(!fSpendable) {
107  itemMint->setDisabled(true);
108  itemMint->setCheckState(COLUMN_CHECKBOX, Qt::Unchecked);
109 
110  //if this mint is in the selection list, then remove it
111  auto it = std::find(listSelectedMints.begin(), listSelectedMints.end(), mint.GetValue().GetHex());
112  if (it != listSelectedMints.end()) {
113  listSelectedMints.erase(it);
114  }
115 
116  string strReason = "";
117  if(nConfirmations < Params().Zerocoin_MintRequiredConfirmations())
118  strReason = strprintf("Needs %d more confirmations", Params().Zerocoin_MintRequiredConfirmations() - nConfirmations);
119  else
120  strReason = strprintf("Needs %d more mints added to network", Params().Zerocoin_RequiredAccumulation() - nMintsAdded);
121 
122  itemMint->setText(COLUMN_ISSPENDABLE, QString::fromStdString(strReason));
123  } else {
124  itemMint->setText(COLUMN_ISSPENDABLE, QString("Yes"));
125  }
126  }
127 
128  ui->treeWidget->blockSignals(false);
129  updateLabels();
130 }

132 // Update the list when a checkbox is clicked
133 void ZDivControlDialog::updateSelection(QTreeWidgetItem* item, int column)
134 {
135  // only want updates from non top level items that are available to spend
136  if (item->parent() && column == COLUMN_CHECKBOX && !item->isDisabled()){
137 
138  // see if this mint is already selected in the selection list
139  std::string strPubcoin = item->text(COLUMN_PUBCOIN).toStdString();
140  auto iter = std::find(listSelectedMints.begin(), listSelectedMints.end(), strPubcoin);
141  bool fSelected = iter != listSelectedMints.end();
142 
143  // set the checkbox to the proper state and add or remove the mint from the selection list
144  if (item->checkState(COLUMN_CHECKBOX) == Qt::Checked) {
145  if (fSelected) return;
146  listSelectedMints.emplace_back(strPubcoin);
147  } else {
148  if (!fSelected) return;
149  listSelectedMints.erase(iter);
150  }
151  updateLabels();
152  }
153 }

185 // select or deselect all of the mints
186 void ZDivControlDialog::ButtonAllClicked()
187 {
188  ui->treeWidget->blockSignals(true);
189  Qt::CheckState state = Qt::Checked;
190  for(int i = 0; i < ui->treeWidget->topLevelItemCount(); i++) {
191  if(ui->treeWidget->topLevelItem(i)->checkState(COLUMN_CHECKBOX) != Qt::Unchecked) {
192  state = Qt::Unchecked;
193  break;
194  }
195  }
196 
197  //much quicker to start from scratch than to have QT go through all the objects and update
198  ui->treeWidget->clear();
199 
200  if(state == Qt::Checked) {
201  for(const CZerocoinMint mint : listMints)
202  listSelectedMints.emplace_back(mint.GetValue().GetHex());
203  } else {
204  listSelectedMints.clear();
205  }
206 
207  updateList();
208 }
