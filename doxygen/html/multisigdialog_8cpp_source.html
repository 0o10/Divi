329 bool MultisigDialog::createMultisigTransaction(vector<CTxIn> vUserIn, vector<CTxOut> vUserOut, string& feeStringRet, string& errorRet)
330 {
331  try{
332  //attempt to access the given inputs
333  CCoinsViewCache view = getInputsCoinsViewCache(vUserIn);
334 
335  //retrieve total input val and change dest
336  CAmount totalIn = 0;
337  vector<CAmount> vInputVals;
338  CScript changePubKey;
339  bool fFirst = true;
340 
341  for(CTxIn in : vUserIn){
342  const CCoins* coins = view.AccessCoins(in.prevout.hash);
343  if(!coins->IsAvailable(in.prevout.n) || coins == NULL){
344  continue;
345  }
346  CTxOut prevout = coins->vout[in.prevout.n];
347  CScript privKey = prevout.scriptPubKey;
348 
349  vInputVals.push_back(prevout.nValue);
350  totalIn += prevout.nValue;
351 
352  if(!fFirst){
353  if(privKey != changePubKey){
354  throw runtime_error("Address mismatch! Inputs must originate from the same multisignature address.");
355  }
356  }else{
357  fFirst = false;
358  changePubKey = privKey;
359  }
360  }
361 
362  CAmount totalOut = 0;
363 
364  //retrieve total output val
365  for(CTxOut out : vUserOut){
366  totalOut += out.nValue;
367  }
368 
369  if(totalIn < totalOut){
370  throw runtime_error("Not enough PIV provided as input to complete transaction (including fee).");
371  }
372 
373  //calculate change amount
374  CAmount changeAmount = totalIn - totalOut;
375  CTxOut change(changeAmount, changePubKey);
376 
377  //generate random position for change
378  unsigned int changeIndex = rand() % (vUserOut.size() + 1);
379 
380  //insert change into random position
381  if(changeIndex < vUserOut.size()){
382  vUserOut.insert(vUserOut.begin() + changeIndex, change);
383  }else{
384  vUserOut.emplace_back(change);
385  }
386 
387  //populate tx
388  CMutableTransaction tx;
389  tx.vin = vUserIn;
390  tx.vout = vUserOut;
391 
392  const CCoins* coins = view.AccessCoins(tx.vin[0].prevout.hash);
393 
394  if(coins == NULL || !coins->IsAvailable(tx.vin[0].prevout.n)){
395  throw runtime_error("Coins unavailable (unconfirmed/spent)");
396  }
397 
398  CScript prevPubKey = coins->vout[tx.vin[0].prevout.n].scriptPubKey;
399 
400  //get payment destination
401  CTxDestination address;
402  if(!ExtractDestination(prevPubKey, address)){
403  throw runtime_error("Could not find address for destination.");
404  }
405 
406  CScriptID hash = boost::get<CScriptID>(address);
407  CScript redeemScript;
408 
409  if (!pwalletMain->GetCScript(hash, redeemScript)){
410  throw runtime_error("could not redeem");
411  }
412  txnouttype type;
413  vector<CTxDestination> addresses;
414  int nReq;
415  if(!ExtractDestinations(redeemScript, type, addresses, nReq)){
416  throw runtime_error("Could not extract destinations from redeem script.");
417  }
418 
419  for(CTxIn& in : tx.vin){
420  in.scriptSig.clear();
421  //scale estimate to account for multisig scriptSig
422  for(unsigned int i = 0; i < 50*(nReq+addresses.size()); i++){
423  in.scriptSig << INT64_MAX;
424  }
425  }
426 
427  //calculate fee
428  unsigned int nBytes = tx.GetSerializeSize(SER_NETWORK, PROTOCOL_VERSION);
429  CAmount fee = ::minRelayTxFee.GetFee(nBytes);
430 
431  if(tx.vout.at(changeIndex).nValue > fee){
432  tx.vout.at(changeIndex).nValue -= fee;
433  feeStringRet = strprintf("%d",((double)fee)/COIN).c_str();
434  }else{
435  throw runtime_error("Not enough PIV provided to cover fee");
436  }
437 
438  //clear junk from script sigs
439  for(CTxIn& in : tx.vin){
440  in.scriptSig.clear();
441  }
442  multisigTx = tx;
443  }catch(const runtime_error& e){
444  errorRet = e.what();
445  return false;
446  }
447  return true;
448 }

539 bool MultisigDialog::signMultisigTx(CMutableTransaction& tx, string& errorOut, QVBoxLayout* keyList)
540 {
541  //will be set false if all inputs are not fully signed(valid)
542  bool fComplete = true;
543 
544  //if keyslist is not default value AND has items in list then true
545  bool fGivenKeys = (keyList != nullptr) && (keyList->count() > 0);
546 
547  try{
548 
549  //copy of vin for reference before vin is mutated
550  vector<CTxIn> oldVin(tx.vin);
551  CBasicKeyStore privKeystore;
552 
553  //if keys were given, attempt to collect redeem and scriptpubkey
554  if(fGivenKeys){
555  for(int i = 0; i < keyList->count(); i++){
556  QWidget* keyFrame = qobject_cast<QWidget*>(keyList->itemAt(i)->widget());
557  QLineEdit* key = keyFrame->findChild<QLineEdit*>("key");
558  CBitcoinSecret vchSecret;
559  if (!vchSecret.SetString(key->text().toStdString()))
560  throw runtime_error("Invalid private key");
561  CKey cKey = vchSecret.GetKey();
562  if (!cKey.IsValid())
563  throw runtime_error("Private key outside allowed range");
564  privKeystore.AddKey(cKey);
565  }
566 
567  for(CTxIn& txin : tx.vin){
568  //get inputs
569  CTransaction txVin;
570  uint256 hashBlock;
571  if (!GetTransaction(txin.prevout.hash, txVin, hashBlock, true))
572  throw runtime_error("txin could not be found");
573 
574  if (hashBlock == 0)
575  throw runtime_error("txin is unconfirmed");
576 
577  //get pubkey from input
578  CScript prevPubKey = txVin.vout[txin.prevout.n].scriptPubKey;
579 
580  //get payment destination
581  CTxDestination address;
582  if(!ExtractDestination(prevPubKey, address)){
583  throw runtime_error("Could not find address for destination.");
584  }
585 
586  //get redeem script related to destination
587  CScriptID hash = boost::get<CScriptID>(address);
588  CScript redeemScript;
589 
590  if (!pwalletMain->GetCScript(hash, redeemScript)){
591  errorOut = "could not redeem";
592  }
593  privKeystore.AddCScript(redeemScript);
594  }
595  }else{
596  if (model->getEncryptionStatus() == model->Locked) {
597  if (!model->requestUnlock(true).isValid()) {
598  // Unlock wallet was cancelled
599  throw runtime_error("Error: Your wallet is locked. Please enter the wallet passphrase first.");
600  }
601  }
602  }
603 
604  //choose between local wallet and provided
605  const CKeyStore& keystore = fGivenKeys ? privKeystore : *pwalletMain;
606 
607  //attempt to sign each input from local wallet
608  int nIn = 0;
609  for(CTxIn& txin : tx.vin){
610  //get inputs
611  CTransaction txVin;
612  uint256 hashBlock;
613  if (!GetTransaction(txin.prevout.hash, txVin, hashBlock, true))
614  throw runtime_error("txin could not be found");
615 
616  if (hashBlock == 0)
617  throw runtime_error("txin is unconfirmed");
618 
619  txin.scriptSig.clear();
620  CScript prevPubKey = txVin.vout[txin.prevout.n].scriptPubKey;
621 
622  //sign what we can
623  SignSignature(keystore, prevPubKey, tx, nIn);
624 
625  //merge in any previous signatures
626  txin.scriptSig = CombineSignatures(prevPubKey, tx, nIn, txin.scriptSig, oldVin[nIn].scriptSig);
627 
628  if (!VerifyScript(txin.scriptSig, prevPubKey, STANDARD_SCRIPT_VERIFY_FLAGS, MutableTransactionSignatureChecker(&tx, nIn))){
629  fComplete = false;
630  }
631  nIn++;
632  }
633 
634  ui->signButtonStatus->setText(buildMultisigTxStatusString(fComplete, tx));
635 
636  }catch(const runtime_error& e){
637  errorOut = string(e.what());
638  fComplete = false;
639  }
640  return fComplete;
641 }

673 void MultisigDialog::commitMultisigTx()
674 {
675  CMutableTransaction tx(multisigTx);
676  try{
677 #ifdef ENABLE_WALLET
678  CWalletTx wtx(pwalletMain, tx);
679  CReserveKey keyChange(pwalletMain);
680  if (!pwalletMain->CommitTransaction(wtx, keyChange))
681  throw runtime_error(string("Transaction rejected - Failed to commit"));
682 #else
683  uint256 hashTx = tx.GetHash();
684  CCoinsViewCache& view = *pcoinsTip;

712 bool MultisigDialog::createRedeemScript(int m, vector<string> vKeys, CScript& redeemRet, string& errorRet)
713 {
714  try{
715  int n = vKeys.size();
716  //gather pub keys
717  if (n < 1)
718  throw runtime_error("a Multisignature address must require at least one key to redeem");
719  if (n < m)
720  throw runtime_error(
721  strprintf("not enough keys supplied "
722  "(got %d keys, but need at least %d to redeem)",
723  m, n));
724  if (n > 15)
725  throw runtime_error("Number of addresses involved in the Multisignature address creation > 15\nReduce the number");
726 
727  vector<CPubKey> pubkeys;
728  pubkeys.resize(n);
729 
730  int i = 0;
731  for(vector<string>::iterator it = vKeys.begin(); it != vKeys.end(); ++it) {
732  string keyString = *it;
733  #ifdef ENABLE_WALLET
734  // Case 1: DIVX address and we have full public key:
735  CBitcoinAddress address(keyString);
736  if (pwalletMain && address.IsValid()) {
737  CKeyID keyID;
738  if (!address.GetKeyID(keyID)) {
739  throw runtime_error(
740  strprintf("%s does not refer to a key", keyString));
741  }
742  CPubKey vchPubKey;
743  if (!pwalletMain->GetPubKey(keyID, vchPubKey))
744  throw runtime_error(
745  strprintf("no full public key for address %s", keyString));
746  if (!vchPubKey.IsFullyValid()){
747  string sKey = keyString.empty()?"(empty)":keyString;
748  throw runtime_error(" Invalid public key: " + sKey );
749  }
750  pubkeys[i++] = vchPubKey;
751  }
752 
753  //case 2: hex pub key
754  else
755  #endif
756  if (IsHex(keyString)) {
757  CPubKey vchPubKey(ParseHex(keyString));
758  if (!vchPubKey.IsFullyValid()){
759  throw runtime_error(" Invalid public key: " + keyString);
760  }
761  pubkeys[i++] = vchPubKey;
762  } else {
763  throw runtime_error(" Invalid public key: " + keyString);
764  }
765  }
766  //populate redeem script
767  //OP_N for required signatures
768  redeemRet << redeemRet.EncodeOP_N(m);
769  //public keys
770  for(CPubKey& key : pubkeys){
771  vector<unsigned char> vKey= ToByteVector(key);
772  redeemRet << vKey;
773  }
774  //OP_N for total pubkeys
775  redeemRet << redeemRet.EncodeOP_N(pubkeys.size());
776  redeemRet << OP_CHECKMULTISIG;
777  return true;
778  }catch(const runtime_error& e){
779  errorRet = string(e.what());
780  return false;
781  }
782 }
