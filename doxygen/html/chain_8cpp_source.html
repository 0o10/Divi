27 {

30  vHave.reserve(32);
31 
32  if (!pindex)
33  pindex = Tip();
34  while (pindex) {
35  vHave.push_back(pindex->GetBlockHash());
36  // Stop when we have added the genesis block.
37  if (pindex->nHeight == 0)
38  break;
39  // Exponentially larger steps back, plus the genesis block.
40  int nHeight = std::max(pindex->nHeight - nStep, 0);
41  if (Contains(pindex)) {
42  // Use O(1) CChain index if possible.
43  pindex = (*this)[nHeight];

45  // Otherwise, use O(log n) skiplist.
46  pindex = pindex->GetAncestor(nHeight);
47  }
48  if (vHave.size() > 10)
49  nStep *= 2;
50  }

53 }
