215 void SendCoinsDialog::on_sendButton_clicked()
216 {
217  if (!model || !model->getOptionsModel())
218  return;
219 
220  QList<SendCoinsRecipient> recipients;
221  bool valid = true;
222 
223  for (int i = 0; i < ui->entries->count(); ++i) {
224  SendCoinsEntry* entry = qobject_cast<SendCoinsEntry*>(ui->entries->itemAt(i)->widget());
225 
226  //UTXO splitter - address should be our own
227  CBitcoinAddress address = entry->getValue().address.toStdString();
228  if (!model->isMine(address) && ui->splitBlockCheckBox->checkState() == Qt::Checked) {
229  CoinControlDialog::coinControl->fSplitBlock = false;
230  ui->splitBlockCheckBox->setCheckState(Qt::Unchecked);
231  QMessageBox::warning(this, tr("Send Coins"),
232  tr("The split block tool does not work when sending to outside addresses. Try again."),
233  QMessageBox::Ok, QMessageBox::Ok);
234  return;
235  }
236 
237  if (entry) {
238  if (entry->validate()) {
239  recipients.append(entry->getValue());
240  } else {
241  valid = false;
242  }
243  }
244  }
245 
246  if (!valid || recipients.isEmpty()) {
247  return;
248  }
249 
250  //set split block in model
251  CoinControlDialog::coinControl->fSplitBlock = ui->splitBlockCheckBox->checkState() == Qt::Checked;
252 
253  if (ui->entries->count() > 1 && ui->splitBlockCheckBox->checkState() == Qt::Checked) {
254  CoinControlDialog::coinControl->fSplitBlock = false;
255  ui->splitBlockCheckBox->setCheckState(Qt::Unchecked);
256  QMessageBox::warning(this, tr("Send Coins"),
257  tr("The split block tool does not work with multiple addresses. Try again."),
258  QMessageBox::Ok, QMessageBox::Ok);
259  return;
260  }
261 
262  if (CoinControlDialog::coinControl->fSplitBlock)
263  CoinControlDialog::coinControl->nSplitBlock = int(ui->splitBlockLineEdit->text().toInt());
264 
265  QString strFunds = tr("using") + " <b>" + tr("anonymous funds") + "</b>";
266  QString strFee = "";
267  recipients[0].inputType = ALL_COINS;
268  strFunds = tr("using") + " <b>" + tr("any available funds (not recommended)") + "</b>";
269 
270  if (ui->checkSwiftTX->isChecked()) {
271  recipients[0].useSwiftTX = true;
272  strFunds += " ";
273  strFunds += tr("and SwiftX");
274  } else {
275  recipients[0].useSwiftTX = false;
276  }
277 
278 
279  // Format confirmation message
280  QStringList formatted;
281  foreach (const SendCoinsRecipient& rcp, recipients) {
282  // generate bold amount string
283  QString amount = "<b>" + BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), rcp.amount);
284  amount.append("</b> ").append(strFunds);
285 
286  // generate monospace address string
287  QString address = "<span style='font-family: monospace;'>" + rcp.address;
288  address.append("</span>");
289 
290  QString recipientElement;
291 
292  if (!rcp.paymentRequest.IsInitialized()) // normal payment
293  {
294  if (rcp.label.length() > 0) // label with address
295  {
296  recipientElement = tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.label));
297  recipientElement.append(QString(" (%1)").arg(address));
298  } else // just address
299  {
300  recipientElement = tr("%1 to %2").arg(amount, address);
301  }
302  } else if (!rcp.authenticatedMerchant.isEmpty()) // secure payment request
303  {
304  recipientElement = tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.authenticatedMerchant));
305  } else // insecure payment request
306  {
307  recipientElement = tr("%1 to %2").arg(amount, address);
308  }
309 
310  if (fSplitBlock) {
311  recipientElement.append(tr(" split into %1 outputs using the UTXO splitter.").arg(CoinControlDialog::coinControl->nSplitBlock));
312  }
313 
314  formatted.append(recipientElement);
315  }
316 
317  fNewRecipientAllowed = false;
318 
319  // request unlock only if was locked or unlocked for mixing:
320  // this way we let users unlock by walletpassphrase or by menu
321  // and make many transactions while unlocking through this dialog
322  // will call relock
323  WalletModel::EncryptionStatus encStatus = model->getEncryptionStatus();
324  if (encStatus == model->Locked || encStatus == model->UnlockedForAnonymizationOnly) {
325  WalletModel::UnlockContext ctx(model->requestUnlock(true));
326  if (!ctx.isValid()) {
327  // Unlock wallet was cancelled
328  fNewRecipientAllowed = true;
329  return;
330  }
331  send(recipients, strFee, formatted);
332  return;
333  }
334  // already unlocked or not encrypted at all
335  send(recipients, strFee, formatted);
336 }

338 void SendCoinsDialog::send(QList<SendCoinsRecipient> recipients, QString strFee, QStringList formatted)
339 {
340  // prepare transaction for getting txFee earlier
341  WalletModelTransaction currentTransaction(recipients);
342  WalletModel::SendCoinsReturn prepareStatus;
343  if (model->getOptionsModel()->getCoinControlFeatures()) // coin control enabled
344  prepareStatus = model->prepareTransaction(currentTransaction, CoinControlDialog::coinControl);
345  else
346  prepareStatus = model->prepareTransaction(currentTransaction);
347 
348  // process prepareStatus and on error generate message shown to user
349  processSendCoinsReturn(prepareStatus,
350  BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), currentTransaction.getTransactionFee()), true);
351 
352  if (prepareStatus.status != WalletModel::OK) {
353  fNewRecipientAllowed = true;
354  return;
355  }
356 
357  CAmount txFee = currentTransaction.getTransactionFee();
358  QString questionString = tr("Are you sure you want to send?");
359  questionString.append("<br /><br />%1");
360 
361  if (txFee > 0) {
362  // append fee string if a fee is required
363  questionString.append("<hr /><span style='color:#aa0000;'>");
364  questionString.append(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), txFee));
365  questionString.append("</span> ");
366  questionString.append(tr("are added as transaction fee"));
367  questionString.append(" ");
368  questionString.append(strFee);
369 
370  // append transaction size
371  questionString.append(" (" + QString::number((double)currentTransaction.getTransactionSize() / 1000) + " kB)");
372  }
373 
374  // add total amount in all subdivision units
375  questionString.append("<hr />");
376  CAmount totalAmount = currentTransaction.getTotalTransactionAmount() + txFee;
377  QStringList alternativeUnits;
378  foreach (BitcoinUnits::Unit u, BitcoinUnits::availableUnits()) {
379  if (u != model->getOptionsModel()->getDisplayUnit())
380  alternativeUnits.append(BitcoinUnits::formatHtmlWithUnit(u, totalAmount));
381  }
382 
383  // Show total amount + all alternative units
384  questionString.append(tr("Total Amount = <b>%1</b><br />= %2")
385  .arg(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), totalAmount))
386  .arg(alternativeUnits.join("<br />= ")));
387 
388  // Limit number of displayed entries
389  int messageEntries = formatted.size();
390  int displayedEntries = 0;
391  for (int i = 0; i < formatted.size(); i++) {
392  if (i >= MAX_SEND_POPUP_ENTRIES) {
393  formatted.removeLast();
394  i--;
395  } else {
396  displayedEntries = i + 1;
397  }
398  }
399  questionString.append("<hr />");
400  questionString.append(tr("<b>(%1 of %2 entries displayed)</b>").arg(displayedEntries).arg(messageEntries));
401 
402  // Display message box
403  QMessageBox::StandardButton retval = QMessageBox::question(this, tr("Confirm send coins"),
404  questionString.arg(formatted.join("<br />")),
405  QMessageBox::Yes | QMessageBox::Cancel,
406  QMessageBox::Cancel);
407 
408  if (retval != QMessageBox::Yes) {
409  fNewRecipientAllowed = true;
410  return;
411  }
412 
413  // now send the prepared transaction
414  WalletModel::SendCoinsReturn sendStatus = model->sendCoins(currentTransaction);
415  // process sendStatus and on error generate message shown to user
416  processSendCoinsReturn(sendStatus);
417 
418  if (sendStatus.status == WalletModel::OK) {
419  accept();
420  CoinControlDialog::coinControl->UnSelectAll();
421  coinControlUpdateLabels();
422  }
423  fNewRecipientAllowed = true;
424 }

589 void SendCoinsDialog::processSendCoinsReturn(const WalletModel::SendCoinsReturn& sendCoinsReturn, const QString& msgArg, bool fPrepare)
590 {
591  bool fAskForUnlock = false;
592 
593  QPair<QString, CClientUIInterface::MessageBoxFlags> msgParams;
594  // Default to a warning message, override if error message is needed
595  msgParams.second = CClientUIInterface::MSG_WARNING;
596 
597  // This comment is specific to SendCoinsDialog usage of WalletModel::SendCoinsReturn.
598  // WalletModel::TransactionCommitFailed is used only in WalletModel::sendCoins()
599  // all others are used only in WalletModel::prepareTransaction()
600  switch (sendCoinsReturn.status) {
601  case WalletModel::InvalidAddress:
602  msgParams.first = tr("The recipient address is not valid, please recheck.");
603  break;
604  case WalletModel::InvalidAmount:
605  msgParams.first = tr("The amount to pay must be larger than 0.");
606  break;
607  case WalletModel::AmountExceedsBalance:
608  msgParams.first = tr("The amount exceeds your balance.");
609  break;
610  case WalletModel::AmountWithFeeExceedsBalance:
611  msgParams.first = tr("The total exceeds your balance when the %1 transaction fee is included.").arg(msgArg);
612  break;
613  case WalletModel::DuplicateAddress:
614  msgParams.first = tr("Duplicate address found, can only send to each address once per send operation.");
615  break;
616  case WalletModel::TransactionCreationFailed:
617  msgParams.first = tr("Transaction creation failed!");
618  msgParams.second = CClientUIInterface::MSG_ERROR;
619  break;
620  case WalletModel::TransactionCommitFailed:
621  msgParams.first = tr("The transaction was rejected! This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
622  msgParams.second = CClientUIInterface::MSG_ERROR;
623  break;
624  case WalletModel::AnonymizeOnlyUnlocked:
625  // Unlock is only need when the coins are send
626  if(!fPrepare)
627  fAskForUnlock = true;
628  else
629  msgParams.first = tr("Error: The wallet was unlocked only to anonymize coins.");
630  break;
631 
632  case WalletModel::InsaneFee:
633  msgParams.first = tr("A fee %1 times higher than %2 per kB is considered an insanely high fee.").arg(10000).arg(BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), ::minRelayTxFee.GetFeePerK()));
634  break;
635  // included to prevent a compiler warning.
636  case WalletModel::OK:
637  default:
638  return;
639  }
640 
641  // Unlock wallet if it wasn't fully unlocked already
642  if(fAskForUnlock) {
643  model->requestUnlock(false);
644  if(model->getEncryptionStatus () != WalletModel::Unlocked) {
645  msgParams.first = tr("Error: The wallet was unlocked only to anonymize coins. Unlock canceled.");
646  }
647  else {
648  // Wallet unlocked
649  return;
650  }
651  }
652 
653  emit message(tr("Send Coins"), msgParams.first, msgParams.second);
654 }
