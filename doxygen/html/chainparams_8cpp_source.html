32 bool CheckProofOfWorkGen(uint256 hash, unsigned int nBits)
33 {
34  bool fNegative;
35  bool fOverflow;
36  uint256 bnTarget;
37 
38  //if (Params().SkipProofOfWorkCheck())
39  // return true;
40 
41  bnTarget.SetCompact(nBits, &fNegative, &fOverflow);
42 
43  // Check range
44  if (fNegative || bnTarget == 0 || fOverflow || bnTarget > (~uint256(0) >> 20))
45  return false;
46 
47  // Check proof of work matches claimed amount
48  if (hash > bnTarget)
49  return false;
50 
51  return true;
52 }

55 static void convertSeed6(std::vector<CAddress>& vSeedsOut, const SeedSpec6* data, unsigned int count)
56 {
57  // It'll only connect to one or two seed nodes because once it connects,
58  // it'll get a pile of addresses with newer timestamps.
59  // Seed nodes are given a random 'last seen time' of between one and two
60  // weeks ago.
61  const int64_t nOneWeek = 7 * 24 * 60 * 60;
62  for (unsigned int i = 0; i < count; i++) {
63  struct in6_addr ip;
64  memcpy(&ip, data[i].addr, sizeof(ip));
65  CAddress addr(CService(ip, data[i].port));
66  addr.nTime = GetTime() - GetRand(nOneWeek) - nOneWeek;
67  vSeedsOut.push_back(addr);
68  }
69 }
