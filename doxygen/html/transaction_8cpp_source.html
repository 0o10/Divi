136 CAmount CTransaction::GetValueOut() const
137 {
138  CAmount nValueOut = 0;
139  for (std::vector<CTxOut>::const_iterator it(vout.begin()); it != vout.end(); ++it)
140  {
141  // DIVI: previously MoneyRange() was called here. This has been replaced with negative check and boundary wrap check.
142  if (it->nValue < 0)
143  throw std::runtime_error("CTransaction::GetValueOut() : value out of range : less than 0");
144 
145  if ((nValueOut + it->nValue) < nValueOut)

223 unsigned int CTransaction::CalculateModifiedSize(unsigned int nTxSize) const
224 {
225  // In order to avoid disincentivizing cleaning up the UTXO set we don't count
226  // the constant overhead for each txin and up to 110 bytes of scriptSig (which
227  // is enough to cover a compressed pubkey p2sh redemption) for priority.
228  // Providing any more cleanup incentive than making additional inputs free would
229  // risk encouraging people to create junk outputs to redeem later.
230  if (nTxSize == 0)
231  nTxSize = ::GetSerializeSize(*this, SER_NETWORK, PROTOCOL_VERSION);
232  for (std::vector<CTxIn>::const_iterator it(vin.begin()); it != vin.end(); ++it)
233  {
234  unsigned int offset = 41U + std::min(110U, (unsigned int)it->scriptSig.size());
235  if (nTxSize > offset)
236  nTxSize -= offset;
237  }
238  return nTxSize;
239 }
