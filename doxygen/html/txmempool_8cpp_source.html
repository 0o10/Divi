168  void seenTxConfirm(const CFeeRate& feeRate, const CFeeRate& minRelayFee, double dPriority, int nBlocksAgo)
169  {
170  // Last entry records "everything else".
171  int nBlocksTruncated = min(nBlocksAgo, (int)history.size() - 1);
172  assert(nBlocksTruncated >= 0);
173 
174  // We need to guess why the transaction was included in a block-- either
175  // because it is high-priority or because it has sufficient fees.
176  bool sufficientFee = (feeRate > minRelayFee);
177  bool sufficientPriority = AllowFree(dPriority);
178  const char* assignedTo = "unassigned";
179  if (sufficientFee && !sufficientPriority && CBlockAverage::AreSane(feeRate, minRelayFee)) {
180  history[nBlocksTruncated].RecordFee(feeRate);
181  assignedTo = "fee";
182  } else if (sufficientPriority && !sufficientFee && CBlockAverage::AreSane(dPriority)) {
183  history[nBlocksTruncated].RecordPriority(dPriority);
184  assignedTo = "priority";
185  } else {
186  // Neither or both fee and priority sufficient to get confirmed:
187  // don't know why they got confirmed.
188  }
189  LogPrint("estimatefee", "Seen TX confirm: %s : %s fee/%g priority, took %d blocks\n",
190  assignedTo, feeRate.ToString(), dPriority, nBlocksAgo);
191  }
192 
193 public:
194  CMinerPolicyEstimator(int nEntries) : nBestSeenHeight(0)
195  {
196  history.resize(nEntries);
197  }

199  void seenBlock(const std::vector<CTxMemPoolEntry>& entries, int nBlockHeight, const CFeeRate minRelayFee)
200  {
201  if (nBlockHeight <= nBestSeenHeight) {
202  // Ignore side chains and re-orgs; assuming they are random
203  // they don't affect the estimate.
204  // And if an attacker can re-org the chain at will, then
205  // you've got much bigger problems than "attacker can influence
206  // transaction fees."
207  return;
208  }
209  nBestSeenHeight = nBlockHeight;
210 
211  // Fill up the history buckets based on how long transactions took
212  // to confirm.
213  std::vector<std::vector<const CTxMemPoolEntry*> > entriesByConfirmations;
214  entriesByConfirmations.resize(history.size());
215  BOOST_FOREACH (const CTxMemPoolEntry& entry, entries) {
216  // How many blocks did it take for miners to include this transaction?
217  int delta = nBlockHeight - entry.GetHeight();
218  if (delta <= 0) {
219  // Re-org made us lose height, this should only happen if we happen
220  // to re-org on a difficulty transition point: very rare!
221  continue;
222  }
223  if ((delta - 1) >= (int)history.size())
224  delta = history.size(); // Last bucket is catch-all
225  entriesByConfirmations.at(delta - 1).push_back(&entry);
226  }
227  for (size_t i = 0; i < entriesByConfirmations.size(); i++) {
228  std::vector<const CTxMemPoolEntry*>& e = entriesByConfirmations.at(i);
229  // Insert at most 10 random entries per bucket, otherwise a single block
230  // can dominate an estimate:
231  if (e.size() > 10) {
232  std::random_shuffle(e.begin(), e.end());
233  e.resize(10);
234  }
235  BOOST_FOREACH (const CTxMemPoolEntry* entry, e) {
236  // Fees are stored and reported as BTC-per-kb:
237  CFeeRate feeRate(entry->GetFee(), entry->GetTxSize());
238  double dPriority = entry->GetPriority(entry->GetHeight()); // Want priority when it went IN
239  seenTxConfirm(feeRate, minRelayFee, dPriority, i);
240  }
241  }
242 
243  //After new samples are added, we have to clear the sorted lists,
244  //so they'll be resorted the next time someone asks for an estimate
245  sortedFeeSamples.clear();
246  sortedPrioritySamples.clear();
247 
248  for (size_t i = 0; i < history.size(); i++) {
249  if (history[i].FeeSamples() + history[i].PrioritySamples() > 0)
250  LogPrint("estimatefee", "estimates: for confirming within %d blocks based on %d/%d samples, fee=%s, prio=%g\n",
251  i,
252  history[i].FeeSamples(), history[i].PrioritySamples(),
253  estimateFee(i + 1).ToString(), estimatePriority(i + 1));
254  }
255  }
256 
260  CFeeRate estimateFee(int nBlocksToConfirm)
261  {
262  nBlocksToConfirm--;
263 
264  if (nBlocksToConfirm < 0 || nBlocksToConfirm >= (int)history.size())
265  return CFeeRate(0);
266 
267  if (sortedFeeSamples.size() == 0) {
268  for (size_t i = 0; i < history.size(); i++)
269  history.at(i).GetFeeSamples(sortedFeeSamples);
270  std::sort(sortedFeeSamples.begin(), sortedFeeSamples.end(),
271  std::greater<CFeeRate>());
272  }
273  if (sortedFeeSamples.size() < 11) {
274  // Eleven is Gavin's Favorite Number
275  // ... but we also take a maximum of 10 samples per block so eleven means
276  // we're getting samples from at least two different blocks
277  return CFeeRate(0);
278  }
279 
280  int nBucketSize = history.at(nBlocksToConfirm).FeeSamples();
281 
282  // Estimates should not increase as number of confirmations goes up,
283  // but the estimates are noisy because confirmations happen discretely
284  // in blocks. To smooth out the estimates, use all samples in the history
285  // and use the nth highest where n is (number of samples in previous bucket +
286  // half the samples in nBlocksToConfirm bucket):
287  size_t nPrevSize = 0;
288  for (int i = 0; i < nBlocksToConfirm; i++)
289  nPrevSize += history.at(i).FeeSamples();
290  size_t index = min(nPrevSize + nBucketSize / 2, sortedFeeSamples.size() - 1);
291  return sortedFeeSamples[index];
292  }
293  double estimatePriority(int nBlocksToConfirm)
294  {
295  nBlocksToConfirm--;
296 
297  if (nBlocksToConfirm < 0 || nBlocksToConfirm >= (int)history.size())
298  return -1;
299 
300  if (sortedPrioritySamples.size() == 0) {
301  for (size_t i = 0; i < history.size(); i++)
302  history.at(i).GetPrioritySamples(sortedPrioritySamples);
303  std::sort(sortedPrioritySamples.begin(), sortedPrioritySamples.end(),
304  std::greater<double>());
305  }
306  if (sortedPrioritySamples.size() < 11)
307  return -1.0;
308 
309  int nBucketSize = history.at(nBlocksToConfirm).PrioritySamples();
310 
311  // Estimates should not increase as number of confirmations needed goes up,
312  // but the estimates are noisy because confirmations happen discretely
313  // in blocks. To smooth out the estimates, use all samples in the history
314  // and use the nth highest where n is (number of samples in previous buckets +
315  // half the samples in nBlocksToConfirm bucket).
316  size_t nPrevSize = 0;
317  for (int i = 0; i < nBlocksToConfirm; i++)
318  nPrevSize += history.at(i).PrioritySamples();
319  size_t index = min(nPrevSize + nBucketSize / 2, sortedPrioritySamples.size() - 1);
320  return sortedPrioritySamples[index];
321  }

425 void CTxMemPool::remove(const CTransaction& origTx, std::list<CTransaction>& removed, bool fRecursive)
426 {
427  // Remove transaction from memory pool
428  {
429  LOCK(cs);
430  std::deque<uint256> txToRemove;
431  txToRemove.push_back(origTx.GetHash());
432  if (fRecursive && !mapTx.count(origTx.GetHash())) {
433  // If recursively removing but origTx isn't in the mempool
434  // be sure to remove any children that are in the pool. This can
435  // happen during chain re-orgs if origTx isn't re-accepted into
436  // the mempool for any reason.
437  for (unsigned int i = 0; i < origTx.vout.size(); i++) {
438  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(COutPoint(origTx.GetHash(), i));
439  if (it == mapNextTx.end())
440  continue;
441  txToRemove.push_back(it->second.ptx->GetHash());
442  }
443  }
444  while (!txToRemove.empty()) {
445  uint256 hash = txToRemove.front();
446  txToRemove.pop_front();
447  if (!mapTx.count(hash))
448  continue;
449  const CTransaction& tx = mapTx[hash].GetTx();
450  if (fRecursive) {
451  for (unsigned int i = 0; i < tx.vout.size(); i++) {
452  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(COutPoint(hash, i));
453  if (it == mapNextTx.end())
454  continue;
455  txToRemove.push_back(it->second.ptx->GetHash());
456  }
457  }
458  BOOST_FOREACH (const CTxIn& txin, tx.vin)
459  mapNextTx.erase(txin.prevout);
460 
461  removed.push_back(tx);
462  totalTxSize -= mapTx[hash].GetTxSize();
463  mapTx.erase(hash);
464  nTransactionsUpdated++;
465  }
466  }
467 }

541 void CTxMemPool::check(const CCoinsViewCache* pcoins) const
542 {
543  if (!fSanityCheck)
544  return;
545 
546  LogPrint("mempool", "Checking mempool with %u transactions and %u inputs\n", (unsigned int)mapTx.size(), (unsigned int)mapNextTx.size());
547 
548  uint64_t checkTotal = 0;
549 
550  CCoinsViewCache mempoolDuplicate(const_cast<CCoinsViewCache*>(pcoins));
551 
552  LOCK(cs);
553  list<const CTxMemPoolEntry*> waitingOnDependants;
554  for (std::map<uint256, CTxMemPoolEntry>::const_iterator it = mapTx.begin(); it != mapTx.end(); it++) {
555  unsigned int i = 0;
556  checkTotal += it->second.GetTxSize();
557  const CTransaction& tx = it->second.GetTx();
558  bool fDependsWait = false;
559  BOOST_FOREACH (const CTxIn& txin, tx.vin) {
560  // Check that every mempool transaction's inputs refer to available coins, or other mempool tx's.
561  std::map<uint256, CTxMemPoolEntry>::const_iterator it2 = mapTx.find(txin.prevout.hash);
562  if (it2 != mapTx.end()) {
563  const CTransaction& tx2 = it2->second.GetTx();
564  assert(tx2.vout.size() > txin.prevout.n && !tx2.vout[txin.prevout.n].IsNull());
565  fDependsWait = true;
566  } else {
567  const CCoins* coins = pcoins->AccessCoins(txin.prevout.hash);
568  assert(coins && coins->IsAvailable(txin.prevout.n));
569  }
570  // Check whether its inputs are marked in mapNextTx.
571  std::map<COutPoint, CInPoint>::const_iterator it3 = mapNextTx.find(txin.prevout);
572  assert(it3 != mapNextTx.end());
573  assert(it3->second.ptx == &tx);
574  assert(it3->second.n == i);
575  i++;
576  }
577  if (fDependsWait)
578  waitingOnDependants.push_back(&it->second);
579  else {
580  CValidationState state;
581  CTxUndo undo;
582  assert(CheckInputs(tx, state, mempoolDuplicate, false, 0, false, NULL));
583  UpdateCoins(tx, state, mempoolDuplicate, undo, 1000000);
584  }
585  }
586  unsigned int stepsSinceLastRemove = 0;
587  while (!waitingOnDependants.empty()) {
588  const CTxMemPoolEntry* entry = waitingOnDependants.front();
589  waitingOnDependants.pop_front();
590  CValidationState state;
591  if (!mempoolDuplicate.HaveInputs(entry->GetTx())) {
592  waitingOnDependants.push_back(entry);
593  stepsSinceLastRemove++;
594  assert(stepsSinceLastRemove < waitingOnDependants.size());
595  } else {
596  assert(CheckInputs(entry->GetTx(), state, mempoolDuplicate, false, 0, false, NULL));
597  CTxUndo undo;
598  UpdateCoins(entry->GetTx(), state, mempoolDuplicate, undo, 1000000);
599  stepsSinceLastRemove = 0;
600  }
601  }
602  for (std::map<COutPoint, CInPoint>::const_iterator it = mapNextTx.begin(); it != mapNextTx.end(); it++) {
603  uint256 hash = it->second.ptx->GetHash();
604  map<uint256, CTxMemPoolEntry>::const_iterator it2 = mapTx.find(hash);
605  const CTransaction& tx = it2->second.GetTx();
606  assert(it2 != mapTx.end());
607  assert(&tx == it->second.ptx);
608  assert(tx.vin.size() > it->second.n);
609  assert(it->first == it->second.ptx->vin[it->second.n].prevout);
610  }
611 
612  assert(totalTxSize == checkTotal);
613 }

707 bool CCoinsViewMemPool::GetCoins(const uint256& txid, CCoins& coins) const
708 {

715  return true;
716  }
717  return (base->GetCoins(txid, coins) && !coins.IsPruned());
718 }
