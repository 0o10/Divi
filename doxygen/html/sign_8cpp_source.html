112 bool SignSignature(const CKeyStore &keystore, const CScript& fromPubKey, CMutableTransaction& txTo, unsigned int nIn, int nHashType)
113 {
114  assert(nIn < txTo.vin.size());
115  CTxIn& txin = txTo.vin[nIn];
116 
117  // Leave out the signature from the hash, since a signature can't sign itself.
118  // The checksig op will also drop the signatures from its hash.
119  uint256 hash = SignatureHash(fromPubKey, txTo, nIn, nHashType);
120 
121  txnouttype whichType;
122  if (!Solver(keystore, fromPubKey, hash, nHashType, txin.scriptSig, whichType))
123  return false;
124 
125  if (whichType == TX_SCRIPTHASH)
126  {
127  // Solver returns the subscript that need to be evaluated;
128  // the final scriptSig is the signatures from that
129  // and then the serialized subscript:
130  CScript subscript = txin.scriptSig;
131 
132  // Recompute txn hash using subscript in place of scriptPubKey:
133  uint256 hash2 = SignatureHash(subscript, txTo, nIn, nHashType);
134 
135  txnouttype subType;
136  bool fSolved =
137  Solver(keystore, subscript, hash2, nHashType, txin.scriptSig, subType) && subType != TX_SCRIPTHASH;
138  // Append serialized subscript whether or not it is completely signed:
139  txin.scriptSig << static_cast<valtype>(subscript);
140  if (!fSolved) return false;
141  }
142 
143  // Test solution
144  return VerifyScript(txin.scriptSig, fromPubKey, STANDARD_SCRIPT_VERIFY_FLAGS, MutableTransactionSignatureChecker(&txTo, nIn));
145 }

165 static CScript CombineMultisig(const CScript& scriptPubKey, const CTransaction& txTo, unsigned int nIn,
166  const vector<valtype>& vSolutions,
167  const vector<valtype>& sigs1, const vector<valtype>& sigs2)
168 {
169  // Combine all the signatures we've got:
170  set<valtype> allsigs;
171  BOOST_FOREACH(const valtype& v, sigs1)
172  {
173  if (!v.empty())
174  allsigs.insert(v);
175  }
176  BOOST_FOREACH(const valtype& v, sigs2)
177  {
178  if (!v.empty())
179  allsigs.insert(v);
180  }
181 
182  // Build a map of pubkey -> signature by matching sigs to pubkeys:
183  assert(vSolutions.size() > 1);
184  unsigned int nSigsRequired = vSolutions.front()[0];
185  unsigned int nPubKeys = vSolutions.size()-2;
186  map<valtype, valtype> sigs;
187  BOOST_FOREACH(const valtype& sig, allsigs)
188  {
189  for (unsigned int i = 0; i < nPubKeys; i++)
190  {
191  const valtype& pubkey = vSolutions[i+1];
192  if (sigs.count(pubkey))
193  continue; // Already got a sig for this pubkey
194 
195  if (TransactionSignatureChecker(&txTo, nIn).CheckSig(sig, pubkey, scriptPubKey))
196  {
197  sigs[pubkey] = sig;
198  break;
199  }
200  }
201  }
202  // Now build a merged CScript:
203  unsigned int nSigsHave = 0;
204  CScript result; result << OP_0; // pop-one-too-many workaround
205  for (unsigned int i = 0; i < nPubKeys && nSigsHave < nSigsRequired; i++)
206  {
207  if (sigs.count(vSolutions[i+1]))
208  {
209  result << sigs[vSolutions[i+1]];
210  ++nSigsHave;
211  }
212  }
213  // Fill any missing with OP_0:
214  for (unsigned int i = nSigsHave; i < nSigsRequired; i++)
215  result << OP_0;
216 
217  return result;
218 }

220 static CScript CombineSignatures(const CScript& scriptPubKey, const CTransaction& txTo, unsigned int nIn,
221  const txnouttype txType, const vector<valtype>& vSolutions,
222  vector<valtype>& sigs1, vector<valtype>& sigs2)
223 {
224  switch (txType)
225  {
226  case TX_NONSTANDARD:
227  case TX_NULL_DATA:
228  case TX_ZEROCOINMINT:
229  // Don't know anything about this, assume bigger one is correct:
230  if (sigs1.size() >= sigs2.size())
231  return PushAll(sigs1);
232  return PushAll(sigs2);
233  case TX_PUBKEY:
234  case TX_PUBKEYHASH:
235  // Signatures are bigger than placeholders or empty scripts:
236  if (sigs1.empty() || sigs1[0].empty())
237  return PushAll(sigs2);
238  return PushAll(sigs1);
239  case TX_SCRIPTHASH:
240  if (sigs1.empty() || sigs1.back().empty())
241  return PushAll(sigs2);
242  else if (sigs2.empty() || sigs2.back().empty())
243  return PushAll(sigs1);
244  else
245  {
246  // Recur to combine:
247  valtype spk = sigs1.back();
248  CScript pubKey2(spk.begin(), spk.end());
249 
250  txnouttype txType2;
251  vector<vector<unsigned char> > vSolutions2;
252  Solver(pubKey2, txType2, vSolutions2);
253  sigs1.pop_back();
254  sigs2.pop_back();
255  CScript result = CombineSignatures(pubKey2, txTo, nIn, txType2, vSolutions2, sigs1, sigs2);
256  result << spk;
257  return result;
258  }
259  case TX_MULTISIG:
260  return CombineMultisig(scriptPubKey, txTo, nIn, vSolutions, sigs1, sigs2);
261  }
262 
263  return CScript();
264 }
