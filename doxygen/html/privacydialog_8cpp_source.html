148 
149  if(GetAdjustedTime() > GetSporkValue(SPORK_16_ZEROCOIN_MAINTENANCE_MODE)) {

155 
156  // Reset message text

183

302 void PrivacyDialog::sendzPIV()
303 {
304  QSettings settings;
305 
306  // Handle 'Pay To' address options
307  CBitcoinAddress address(ui->payTo->text().toStdString());
308  if(ui->payTo->text().isEmpty()){
309  QMessageBox::information(this, tr("Spend Zerocoin"), tr("No 'Pay To' address provided, creating local payment"), QMessageBox::Ok, QMessageBox::Ok);
310  }
311  else{
312  if (!address.IsValid()) {
313  QMessageBox::warning(this, tr("Spend Zerocoin"), tr("Invalid Divx Address"), QMessageBox::Ok, QMessageBox::Ok);
314  ui->payTo->setFocus();
315  return;
316  }
317  }
318 
319  // Double is allowed now
320  double dAmount = ui->zPIVpayAmount->text().toDouble();
321  CAmount nAmount = roundint64(dAmount* COIN);
322 
323  // Check amount validity
324  if (!MoneyRange(nAmount) || nAmount <= 0.0) {
325  QMessageBox::warning(this, tr("Spend Zerocoin"), tr("Invalid Send Amount"), QMessageBox::Ok, QMessageBox::Ok);
326  ui->zPIVpayAmount->setFocus();
327  return;
328  }
329 
330  // Convert change to zPIV
331  bool fMintChange = ui->checkBoxMintChange->isChecked();
332 
333  // Persist minimize change setting
334  fMinimizeChange = ui->checkBoxMinimizeChange->isChecked();
335  settings.setValue("fMinimizeChange", fMinimizeChange);
336 
337  // Warn for additional fees if amount is not an integer and change as zPIV is requested
338  bool fWholeNumber = floor(dAmount) == dAmount;
339  double dzFee = 0.0;
340 
341  if(!fWholeNumber)
342  dzFee = 1.0 - (dAmount - floor(dAmount));
343 
344  if(!fWholeNumber && fMintChange){
345  QString strFeeWarning = "You've entered an amount with fractional digits and want the change to be converted to Zerocoin.<br /><br /><b>";
346  strFeeWarning += QString::number(dzFee, 'f', 8) + " PIV </b>will be added to the standard transaction fees!<br />";
347  QMessageBox::StandardButton retval = QMessageBox::question(this, tr("Confirm additional Fees"),
348  strFeeWarning,
349  QMessageBox::Yes | QMessageBox::Cancel,
350  QMessageBox::Cancel);
351 
352  if (retval != QMessageBox::Yes) {
353  // Sending canceled
354  ui->zPIVpayAmount->setFocus();
355  return;
356  }
357  }
358 
359  // Persist Security Level for next start
360  nSecurityLevel = ui->securityLevel->value();
361  settings.setValue("nSecurityLevel", nSecurityLevel);
362 
363  // Spend confirmation message box
364 
365  // Add address info if available
366  QString strAddressLabel = "";
367  if(!ui->payTo->text().isEmpty() && !ui->addAsLabel->text().isEmpty()){
368  strAddressLabel = "<br />(" + ui->addAsLabel->text() + ") ";
369  }
370 
371  // General info
372  QString strQuestionString = tr("Are you sure you want to send?<br /><br />");
373  QString strAmount = "<b>" + QString::number(dAmount, 'f', 8) + " zPIV</b>";
374  QString strAddress = tr(" to address ") + QString::fromStdString(address.ToString()) + strAddressLabel + " <br />";
375 
376  if(ui->payTo->text().isEmpty()){
377  // No address provided => send to local address
378  strAddress = tr(" to a newly generated (unused and therefore anonymous) local address <br />");
379  }
380 
381  QString strSecurityLevel = tr("with Security Level ") + ui->securityLevel->text() + " ?";
382  strQuestionString += strAmount + strAddress + strSecurityLevel;
383 
384  // Display message box
385  QMessageBox::StandardButton retval = QMessageBox::question(this, tr("Confirm send coins"),
386  strQuestionString,
387  QMessageBox::Yes | QMessageBox::Cancel,
388  QMessageBox::Cancel);
389 
390  if (retval != QMessageBox::Yes) {
391  // Sending canceled
392  return;
393  }
394 
395  int64_t nTime = GetTimeMillis();
396  ui->TEMintStatus->setPlainText(tr("Spending Zerocoin.\nComputationally expensive, might need several minutes depending on the selected Security Level and your hardware. \nPlease be patient..."));
397  ui->TEMintStatus->repaint();
398 
399  // use mints from zDiv selector if applicable
400  vector<CZerocoinMint> vMintsSelected;
401  if (!ZDivControlDialog::listSelectedMints.empty()) {
402  vMintsSelected = ZDivControlDialog::GetSelectedMints();
403  }
404 
405  // Spend zPIV
406  CWalletTx wtxNew;
407  CZerocoinSpendReceipt receipt;
408  bool fSuccess = false;
409  if(ui->payTo->text().isEmpty()){
410  // Spend to newly generated local address
411  fSuccess = pwalletMain->SpendZerocoin(nAmount, nSecurityLevel, wtxNew, receipt, vMintsSelected, fMintChange, fMinimizeChange);
412  }
413  else {
414  // Spend to supplied destination address
415  fSuccess = pwalletMain->SpendZerocoin(nAmount, nSecurityLevel, wtxNew, receipt, vMintsSelected, fMintChange, fMinimizeChange, &address);
416  }
417 
418  // Display errors during spend
419  if (!fSuccess) {
420  int nNeededSpends = receipt.GetNeededSpends(); // Number of spends we would need for this transaction
421  const int nMaxSpends = Params().Zerocoin_MaxSpendsPerTransaction(); // Maximum possible spends for one zPIV transaction
422  if (nNeededSpends > nMaxSpends) {
423  QString strStatusMessage = tr("Too much inputs (") + QString::number(nNeededSpends, 10) + tr(") needed. \nMaximum allowed: ") + QString::number(nMaxSpends, 10);
424  strStatusMessage += tr("\nEither mint higher denominations (so fewer inputs are needed) or reduce the amount to spend.");
425  QMessageBox::warning(this, tr("Spend Zerocoin"), strStatusMessage.toStdString().c_str(), QMessageBox::Ok, QMessageBox::Ok);
426  ui->TEMintStatus->setPlainText(tr("Spend Zerocoin failed with status = ") +QString::number(receipt.GetStatus(), 10) + "\n" + "Message: " + QString::fromStdString(strStatusMessage.toStdString()));
427  }
428  else {
429  QMessageBox::warning(this, tr("Spend Zerocoin"), receipt.GetStatusMessage().c_str(), QMessageBox::Ok, QMessageBox::Ok);
430  ui->TEMintStatus->setPlainText(tr("Spend Zerocoin failed with status = ") +QString::number(receipt.GetStatus(), 10) + "\n" + "Message: " + QString::fromStdString(receipt.GetStatusMessage()));
431  }
432  ui->zPIVpayAmount->setFocus();
433  ui->TEMintStatus->repaint();
434  return;
435  }
436 
437  // Clear zdiv selector in case it was used
438  ZDivControlDialog::listSelectedMints.clear();
439 
440  // Some statistics for entertainment
441  QString strStats = "";
442  CAmount nValueIn = 0;
443  int nCount = 0;
444  for (CZerocoinSpend spend : receipt.GetSpends()) {
445  strStats += tr("zDiv Spend #: ") + QString::number(nCount) + ", ";
446  strStats += tr("denomination: ") + QString::number(spend.GetDenomination()) + ", ";
447  strStats += tr("serial: ") + spend.GetSerial().ToString().c_str() + "\n";
448  strStats += tr("Spend is 1 of : ") + QString::number(spend.GetMintCount()) + " mints in the accumulator\n";
449  nValueIn += libzerocoin::ZerocoinDenominationToAmount(spend.GetDenomination());
450  }
451 
452  CAmount nValueOut = 0;
453  for (const CTxOut& txout: wtxNew.vout) {
454  strStats += tr("value out: ") + FormatMoney(txout.nValue).c_str() + " Div, ";
455  nValueOut += txout.nValue;
456 
457  strStats += tr("address: ");
458  CTxDestination dest;
459  if(txout.scriptPubKey.IsZerocoinMint())
460  strStats += tr("zDiv Mint");
461  else if(ExtractDestination(txout.scriptPubKey, dest))
462  strStats += tr(CBitcoinAddress(dest).ToString().c_str());
463  strStats += "\n";
464  }
465  double fDuration = (double)(GetTimeMillis() - nTime)/1000.0;
466  strStats += tr("Duration: ") + QString::number(fDuration) + tr(" sec.\n");
467  strStats += tr("Sending successful, return code: ") + QString::number(receipt.GetStatus()) + "\n";
468 
469  QString strReturn;
470  strReturn += tr("txid: ") + wtxNew.GetHash().ToString().c_str() + "\n";
471  strReturn += tr("fee: ") + QString::fromStdString(FormatMoney(nValueIn-nValueOut)) + "\n";
472  strReturn += strStats;
473 
474  // Clear amount to avoid double spending when accidentally clicking twice
475  ui->zPIVpayAmount->setText ("0");
476 
477  ui->TEMintStatus->setPlainText(strReturn);
478  ui->TEMintStatus->repaint();
479 }

540 void PrivacyDialog::setBalance(const CAmount& balance, const CAmount& unconfirmedBalance, const CAmount& immatureBalance,
541  const CAmount& zerocoinBalance, const CAmount& unconfirmedZerocoinBalance, const CAmount& immatureZerocoinBalance,
542  const CAmount& watchOnlyBalance, const CAmount& watchUnconfBalance, const CAmount& watchImmatureBalance)
543 {
544 
545  currentBalance = balance;
546  currentUnconfirmedBalance = unconfirmedBalance;
547  currentImmatureBalance = immatureBalance;
548  currentZerocoinBalance = zerocoinBalance;
549  currentUnconfirmedZerocoinBalance = unconfirmedZerocoinBalance;
550  currentImmatureZerocoinBalance = immatureZerocoinBalance;
551  currentWatchOnlyBalance = watchOnlyBalance;
552  currentWatchUnconfBalance = watchUnconfBalance;
553  currentWatchImmatureBalance = watchImmatureBalance;
554 
555  CWalletDB walletdb(pwalletMain->strWalletFile);
556  list<CZerocoinMint> listMints = walletdb.ListMintedCoins(true, false, true);
557 
558  std::map<libzerocoin::CoinDenomination, CAmount> mapDenomBalances;
559  std::map<libzerocoin::CoinDenomination, int> mapUnconfirmed;
560  std::map<libzerocoin::CoinDenomination, int> mapImmature;
561  for (const auto& denom : libzerocoin::zerocoinDenomList){
562  mapDenomBalances.insert(make_pair(denom, 0));
563  mapUnconfirmed.insert(make_pair(denom, 0));
564  mapImmature.insert(make_pair(denom, 0));
565  }
566 
567  for (auto& mint : listMints){
568  // All denominations

588  }

614

616  QString::number(nCoins) + " = <b>" +

618
