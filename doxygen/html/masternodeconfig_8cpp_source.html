26  boost::filesystem::ifstream streamConfig(pathMasternodeConfigFile);
27 
28  if (!streamConfig.good()) {
29  FILE* configFile = fopen(pathMasternodeConfigFile.string().c_str(), "a");
30  if (configFile != NULL) {
31  std::string strHeader = "# Masternode config file\n"
32  "# Format: alias IP:port masternodeprivkey collateral_output_txid collateral_output_index\n"
33  "# Example: mn1 127.0.0.2:51474 93HaYBVUCYjEMeeH1Y4sBGLALQZE1Yc1K64xiqgX37tGBDQL8Xg 2bcd3c84c84f87eaa86e4e56834c92927a07f9e18718810b92e0d0324456a67c 0\n";
34  fwrite(strHeader.c_str(), std::strlen(strHeader.c_str()), 1, configFile);
35  fclose(configFile);
36  }
37  return true; // Nothing to read, so just return
38  }
39 
40  for (std::string line; std::getline(streamConfig, line); linenumber++) {
41  if (line.empty()) continue;
42 
43  std::istringstream iss(line);
44  std::string comment, alias, ip, privKey, txHash, outputIndex;
45 
46  if (iss >> comment) {
47  if (comment.at(0) == '#') continue;
48  iss.str(line);
49  iss.clear();
50  }
51 
52  if (!(iss >> alias >> ip >> privKey >> txHash >> outputIndex)) {
53  iss.str(line);
54  iss.clear();
55  if (!(iss >> alias >> ip >> privKey >> txHash >> outputIndex)) {
56  strErr = _("Could not parse masternode.conf") + "\n" +
57  strprintf(_("Line: %d"), linenumber) + "\n\"" + line + "\"";
58  streamConfig.close();
59  return false;
60  }
61  }
62 
63  if (Params().NetworkID() == CBaseChainParams::MAIN) {
64  if (CService(ip).GetPort() != 51472) {
65  strErr = _("Invalid port detected in masternode.conf") + "\n" +
66  strprintf(_("Line: %d"), linenumber) + "\n\"" + line + "\"" + "\n" +
67  _("(must be 51472 for mainnet)");
68  streamConfig.close();

70  }
71  } else if (CService(ip).GetPort() == 51472) {
72  strErr = _("Invalid port detected in masternode.conf") + "\n" +
73  strprintf(_("Line: %d"), linenumber) + "\n\"" + line + "\"" + "\n" +
74  _("(51472 could be used only on mainnet)");
75  streamConfig.close();
76  return false;
77  }
78 
79 
80  add(alias, ip, privKey, txHash, outputIndex);
