33 //Load a checkpoint containing 8 32bit checksums of accumulator values.
34 bool AccumulatorMap::Load(uint256 nCheckpoint)
35 {
36  for (auto& denom : zerocoinDenomList) {
37  uint32_t nChecksum = ParseChecksum(nCheckpoint, denom);

42  return false;

44 
45  mapAccumulators.at(denom)->setValue(bnValue);

72 //Calculate a 32bit checksum of each accumulator value. Concatenate checksums into uint256
73 uint256 AccumulatorMap::GetCheckpoint()
74 {
75  uint256 nCheckpoint;
76 
77  //Prevent possible overflows from future changes to the list and forgetting to update this code
78  assert(zerocoinDenomList.size() == 8);
79  for (auto& denom : zerocoinDenomList) {
80  CBigNum bnValue = mapAccumulators.at(denom)->getValue();
81  uint32_t nCheckSum = GetChecksum(bnValue);
82  nCheckpoint = nCheckpoint << 32 | nCheckSum;
83  }
84 
85  return nCheckpoint;
86 }
