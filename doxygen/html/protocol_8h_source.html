67 enum {
68  NODE_NETWORK = (1 << 0),
69 
70  // NODE_BLOOM means the node is capable and willing to handle bloom-filtered connections.
71  // Bitcoin Core nodes used to support this by default, without advertising this bit,
72  // but no longer do as of protocol version 70011 (= NO_BLOOM_VERSION)
73  NODE_BLOOM = (1 << 2),
74 
75  // NODE_BLOOM_WITHOUT_MN means the node has the same features as NODE_BLOOM with the only difference
76  // that the node doens't want to receive master nodes messages. (the 1<<3 was not picked as constant because on bitcoin 0.14 is witness and we want that update here )
77 
78  NODE_BLOOM_WITHOUT_MN = (1 << 4),
79 
80  // Bits 24-31 are reserved for temporary experiments. Just pick a bit that
81  // isn't getting used, or one not being used much, and notify the
82  // bitcoin-development mailing list. Remember that service bits are just
83  // unauthenticated advertisements, so your code must be robust against
84  // collisions and other cases where nodes may be advertising a service they
85  // do not actually support. Other service bits should be allocated via the
86  // BIP process.
87 };
