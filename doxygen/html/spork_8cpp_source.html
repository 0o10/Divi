29 // DIVI: on startup load spork values from previous session if they exist in the sporkDB
30 void LoadSporksFromDB()
31 {
32  for (int i = SPORK_START; i <= SPORK_END; ++i) {
33  // Since not all spork IDs are in use, we have to exclude undefined IDs
34  std::string strSpork = sporkManager.GetSporkNameByID(i);
35  if (strSpork == "Unknown") continue;
36 
37  // attempt to read spork from sporkDB
38  CSporkMessage spork;
39  if (!pSporkDB->ReadSpork(i, spork)) {
40  LogPrintf("%s : no previous value for %s found in database\n", __func__, strSpork);
41  continue;
42  }
43 
44  // add spork to memory
45  mapSporks[spork.GetHash()] = spork;
46  mapSporksActive[spork.nSporkID] = spork;
47  std::time_t result = spork.nValue;
48  // If SPORK Value is greater than 1,000,000 assume it's actually a Date and then convert to a more readable format
49  if (spork.nValue > 1000000) {
50  LogPrintf("%s : loaded spork %s with value %d : %s", __func__,
51  sporkManager.GetSporkNameByID(spork.nSporkID), spork.nValue,
52  std::ctime(&result));
53  } else {
54  LogPrintf("%s : loaded spork %s with value %d\n", __func__,
55  sporkManager.GetSporkNameByID(spork.nSporkID), spork.nValue);
56  }
57  }
58 }

60 void ProcessSpork(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
61 {
62  if (fLiteMode) return; //disable all obfuscation/masternode related functionality
63 
64  if (strCommand == "spork") {
65  //LogPrintf("ProcessSpork::spork\n");
66  CDataStream vMsg(vRecv);
67  CSporkMessage spork;
68  vRecv >> spork;
69 
70  if (chainActive.Tip() == NULL) return;
71 
72  // Ignore spork messages about unknown/deleted sporks
73  std::string strSpork = sporkManager.GetSporkNameByID(spork.nSporkID);
74  if (strSpork == "Unknown") return;
75 
76  uint256 hash = spork.GetHash();
77  if (mapSporksActive.count(spork.nSporkID)) {
78  if (mapSporksActive[spork.nSporkID].nTimeSigned >= spork.nTimeSigned) {
79  if (fDebug) LogPrintf("spork - seen %s block %d \n", hash.ToString(), chainActive.Tip()->nHeight);
80  return;
81  } else {
82  if (fDebug) LogPrintf("spork - got updated spork %s block %d \n", hash.ToString(), chainActive.Tip()->nHeight);
83  }
84  }
85 
86  LogPrintf("spork - new %s ID %d Time %d bestHeight %d\n", hash.ToString(), spork.nSporkID, spork.nValue, chainActive.Tip()->nHeight);
87 
88  if (!sporkManager.CheckSignature(spork)) {
89  LogPrintf("spork - invalid signature\n");
90  Misbehaving(pfrom->GetId(), 100);
91  return;
92  }
93 
94  mapSporks[hash] = spork;
95  mapSporksActive[spork.nSporkID] = spork;
96  sporkManager.Relay(spork);
97 
98  // DIVI: add to spork database.
99  pSporkDB->WriteSpork(spork.nSporkID, spork);
100  }
101  if (strCommand == "getsporks") {
102  std::map<int, CSporkMessage>::iterator it = mapSporksActive.begin();
103 
104  while (it != mapSporksActive.end()) {
105  pfrom->PushMessage("spork", it->second);
106  it++;
107  }
108  }
109 }

112 // grab the value of the spork on the network, or the default
113 int64_t GetSporkValue(int nSporkID)
114 {
115  int64_t r = -1;
116 
117  if (mapSporksActive.count(nSporkID)) {
118  r = mapSporksActive[nSporkID].nValue;
119  } else {
120  if (nSporkID == SPORK_2_SWIFTTX) r = SPORK_2_SWIFTTX_DEFAULT;
121  if (nSporkID == SPORK_3_SWIFTTX_BLOCK_FILTERING) r = SPORK_3_SWIFTTX_BLOCK_FILTERING_DEFAULT;
122  if (nSporkID == SPORK_5_MAX_VALUE) r = SPORK_5_MAX_VALUE_DEFAULT;
123  if (nSporkID == SPORK_7_MASTERNODE_SCANNING) r = SPORK_7_MASTERNODE_SCANNING_DEFAULT;
124  if (nSporkID == SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT) r = SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT_DEFAULT;
125  if (nSporkID == SPORK_9_MASTERNODE_BUDGET_ENFORCEMENT) r = SPORK_9_MASTERNODE_BUDGET_ENFORCEMENT_DEFAULT;
126  if (nSporkID == SPORK_10_MASTERNODE_PAY_UPDATED_NODES) r = SPORK_10_MASTERNODE_PAY_UPDATED_NODES_DEFAULT;
127  if (nSporkID == SPORK_13_ENABLE_SUPERBLOCKS) r = SPORK_13_ENABLE_SUPERBLOCKS_DEFAULT;
128  if (nSporkID == SPORK_14_NEW_PROTOCOL_ENFORCEMENT) r = SPORK_14_NEW_PROTOCOL_ENFORCEMENT_DEFAULT;
129  if (nSporkID == SPORK_15_NEW_PROTOCOL_ENFORCEMENT_2) r = SPORK_15_NEW_PROTOCOL_ENFORCEMENT_2_DEFAULT;
130  if (nSporkID == SPORK_16_ZEROCOIN_MAINTENANCE_MODE) r = SPORK_16_ZEROCOIN_MAINTENANCE_MODE_DEFAULT;
131 
132  if (r == -1) LogPrintf("GetSpork::Unknown Spork %d\n", nSporkID);
133  }
134 
135  return r;

147 void ReprocessBlocks(int nBlocks)
148 {
149  std::map<uint256, int64_t>::iterator it = mapRejectedBlocks.begin();
150  while (it != mapRejectedBlocks.end()) {
151  //use a window twice as large as is usual for the nBlocks we want to reset
152  if ((*it).second > GetTime() - (nBlocks * 60 * 5)) {
153  BlockMap::iterator mi = mapBlockIndex.find((*it).first);
154  if (mi != mapBlockIndex.end() && (*mi).second) {
155  LOCK(cs_main);
156 
157  CBlockIndex* pindex = (*mi).second;
158  LogPrintf("ReprocessBlocks - %s\n", (*it).first.ToString());
159 
160  CValidationState state;
161  ReconsiderBlock(state, pindex);
162  }
163  }
164  ++it;
165  }
166 
167  CValidationState state;
168  {
169  LOCK(cs_main);
170  DisconnectBlocksAndReprocess(nBlocks);
171  }
172 
173  if (state.IsValid()) {
174  ActivateBestChain(state);
175  }
176 }
