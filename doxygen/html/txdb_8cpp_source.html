56 bool CCoinsViewDB::BatchWrite(CCoinsMap& mapCoins, const uint256& hashBlock)
57 {
58  CLevelDBBatch batch;
59  size_t count = 0;
60  size_t changed = 0;
61  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
62  if (it->second.flags & CCoinsCacheEntry::DIRTY) {
63  BatchWriteCoins(batch, it->first, it->second.coins);
64  changed++;
65  }
66  count++;
67  CCoinsMap::iterator itOld = it++;
68  mapCoins.erase(itOld);
69  }
70  if (hashBlock != uint256(0))
71  BatchWriteHashBestChain(batch, hashBlock);
72 
73  LogPrint("coindb", "Committing %u changed transactions (out of %u) to coin database...\n", (unsigned int)changed, (unsigned int)count);
74  return db.WriteBatch(batch);
75 }

120 bool CCoinsViewDB::GetStats(CCoinsStats& stats) const
121 {
122  /* It seems that there are no "const iterators" for LevelDB. Since we
123  only need read operations on it, use a const-cast to get around
124  that restriction. */
125  boost::scoped_ptr<leveldb::Iterator> pcursor(const_cast<CLevelDBWrapper*>(&db)->NewIterator());
126  pcursor->SeekToFirst();
127 
128  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
129  stats.hashBlock = GetBestBlock();
130  ss << stats.hashBlock;
131  CAmount nTotalAmount = 0;
132  while (pcursor->Valid()) {
133  boost::this_thread::interruption_point();
134  try {
135  leveldb::Slice slKey = pcursor->key();
136  CDataStream ssKey(slKey.data(), slKey.data() + slKey.size(), SER_DISK, CLIENT_VERSION);
137  char chType;
138  ssKey >> chType;
139  if (chType == 'c') {
140  leveldb::Slice slValue = pcursor->value();
141  CDataStream ssValue(slValue.data(), slValue.data() + slValue.size(), SER_DISK, CLIENT_VERSION);
142  CCoins coins;
143  ssValue >> coins;
144  uint256 txhash;
145  ssKey >> txhash;
146  ss << txhash;
147  ss << VARINT(coins.nVersion);
148  ss << (coins.fCoinBase ? 'c' : 'n');
149  ss << VARINT(coins.nHeight);
150  stats.nTransactions++;
151  for (unsigned int i = 0; i < coins.vout.size(); i++) {
152  const CTxOut& out = coins.vout[i];
153  if (!out.IsNull()) {
154  stats.nTransactionOutputs++;
155  ss << VARINT(i + 1);
156  ss << out;
157  nTotalAmount += out.nValue;
158  }
159  }
160  stats.nSerializedSize += 32 + slValue.size();
161  ss << VARINT(0);
162  }
163  pcursor->Next();
164  } catch (std::exception& e) {
165  return error("%s : Deserialize or I/O error - %s", __func__, e.what());
166  }
167  }
168  stats.nHeight = mapBlockIndex.find(GetBestBlock())->second->nHeight;
169  stats.hashSerialized = ss.GetHash();
170  stats.nTotalAmount = nTotalAmount;
171  return true;
172 }

211 bool CBlockTreeDB::LoadBlockIndexGuts()
212 {
213  boost::scoped_ptr<leveldb::Iterator> pcursor(NewIterator());
214 
215  CDataStream ssKeySet(SER_DISK, CLIENT_VERSION);
216  ssKeySet << make_pair('b', uint256(0));
217  pcursor->Seek(ssKeySet.str());
218 
219  // Load mapBlockIndex
220  uint256 nPreviousCheckpoint;
221  while (pcursor->Valid()) {
222  boost::this_thread::interruption_point();
223  try {
224  leveldb::Slice slKey = pcursor->key();
225  CDataStream ssKey(slKey.data(), slKey.data() + slKey.size(), SER_DISK, CLIENT_VERSION);
226  char chType;
227  ssKey >> chType;
228  if (chType == 'b') {
229  leveldb::Slice slValue = pcursor->value();
230  CDataStream ssValue(slValue.data(), slValue.data() + slValue.size(), SER_DISK, CLIENT_VERSION);
231  CDiskBlockIndex diskindex;
232  ssValue >> diskindex;
233 
234  // Construct block index object
235  CBlockIndex* pindexNew = InsertBlockIndex(diskindex.GetBlockHash());
236  pindexNew->pprev = InsertBlockIndex(diskindex.hashPrev);
237  pindexNew->pnext = InsertBlockIndex(diskindex.hashNext);
238  pindexNew->nHeight = diskindex.nHeight;
239  pindexNew->nFile = diskindex.nFile;
240  pindexNew->nDataPos = diskindex.nDataPos;
241  pindexNew->nUndoPos = diskindex.nUndoPos;
242  pindexNew->nVersion = diskindex.nVersion;
243  pindexNew->hashMerkleRoot = diskindex.hashMerkleRoot;
244  pindexNew->nTime = diskindex.nTime;
245  pindexNew->nBits = diskindex.nBits;
246  pindexNew->nNonce = diskindex.nNonce;
247  pindexNew->nStatus = diskindex.nStatus;
248  pindexNew->nTx = diskindex.nTx;
249 
250  //zerocoin
251  pindexNew->nAccumulatorCheckpoint = diskindex.nAccumulatorCheckpoint;
252  pindexNew->mapZerocoinSupply = diskindex.mapZerocoinSupply;
253  pindexNew->vMintDenominationsInBlock = diskindex.vMintDenominationsInBlock;
254 
255  //Proof Of Stake
256  pindexNew->nMint = diskindex.nMint;
257  pindexNew->nMoneySupply = diskindex.nMoneySupply;
258  pindexNew->nFlags = diskindex.nFlags;
259  pindexNew->nStakeModifier = diskindex.nStakeModifier;
260  pindexNew->prevoutStake = diskindex.prevoutStake;
261  pindexNew->nStakeTime = diskindex.nStakeTime;
262  pindexNew->hashProofOfStake = diskindex.hashProofOfStake;
263 
264  if (pindexNew->nHeight <= Params().LAST_POW_BLOCK()) {
265  if (!CheckProofOfWork(pindexNew->GetBlockHash(), pindexNew->nBits))
266  return error("LoadBlockIndex() : CheckProofOfWork failed: %s", pindexNew->ToString());
267  }
268  // ppcoin: build setStakeSeen
269  if (pindexNew->IsProofOfStake())
270  setStakeSeen.insert(make_pair(pindexNew->prevoutStake, pindexNew->nStakeTime));
271 
272  //populate accumulator checksum map in memory
273  if(pindexNew->nAccumulatorCheckpoint != 0 && pindexNew->nAccumulatorCheckpoint != nPreviousCheckpoint) {
274  //Don't load any invalid checkpoints
275  if (!InvalidCheckpointRange(pindexNew->nHeight))
276  LoadAccumulatorValuesFromDB(pindexNew->nAccumulatorCheckpoint);
277 
278  nPreviousCheckpoint = pindexNew->nAccumulatorCheckpoint;
279  }
280 
281  pcursor->Next();
282  } else {
283  break; // if shutdown requested or finished loading block index
284  }
285  } catch (std::exception& e) {
286  return error("%s : Deserialize or I/O error - %s", __func__, e.what());
287  }
288  }
289 
290  return true;
291 }
