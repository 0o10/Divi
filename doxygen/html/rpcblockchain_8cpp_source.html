26 double GetDifficulty(const CBlockIndex* blockindex)
27 {
28  // Floating point number that is a multiple of the minimum difficulty,
29  // minimum difficulty = 1.0.
30  if (blockindex == NULL) {
31  if (chainActive.Tip() == NULL)
32  return 1.0;
33  else
34  blockindex = chainActive.Tip();
35  }
36 
37  int nShift = (blockindex->nBits >> 24) & 0xff;
38 
39  double dDiff =
40  (double)0x0000ffff / (double)(blockindex->nBits & 0x00ffffff);
41 
42  while (nShift < 29) {
43  dDiff *= 256.0;
44  nShift++;
45  }
46  while (nShift > 29) {
47  dDiff /= 256.0;
48  nShift--;
49  }
50 
51  return dDiff;
52 }

55 Object blockToJSON(const CBlock& block, const CBlockIndex* blockindex, bool txDetails = false)
56 {
57  Object result;
58  result.push_back(Pair("hash", block.GetHash().GetHex()));
59  int confirmations = -1;
60  // Only report confirmations if the block is on the main chain
61  if (chainActive.Contains(blockindex))
62  confirmations = chainActive.Height() - blockindex->nHeight + 1;
63  result.push_back(Pair("confirmations", confirmations));
64  result.push_back(Pair("size", (int)::GetSerializeSize(block, SER_NETWORK, PROTOCOL_VERSION)));
65  result.push_back(Pair("height", blockindex->nHeight));
66  result.push_back(Pair("version", block.nVersion));
67  result.push_back(Pair("merkleroot", block.hashMerkleRoot.GetHex()));
68  result.push_back(Pair("acc_checkpoint", block.nAccumulatorCheckpoint.GetHex()));
69  Array txs;
70  BOOST_FOREACH (const CTransaction& tx, block.vtx) {
71  if (txDetails) {
72  Object objTx;
73  TxToJSON(tx, uint256(0), objTx);
74  txs.push_back(objTx);
75  } else
76  txs.push_back(tx.GetHash().GetHex());
77  }
78  result.push_back(Pair("tx", txs));
79  result.push_back(Pair("time", block.GetBlockTime()));
80  result.push_back(Pair("nonce", (uint64_t)block.nNonce));
81  result.push_back(Pair("bits", strprintf("%08x", block.nBits)));
82  result.push_back(Pair("difficulty", GetDifficulty(blockindex)));
83  result.push_back(Pair("chainwork", blockindex->nChainWork.GetHex()));
84 
85  if (blockindex->pprev)
86  result.push_back(Pair("previousblockhash", blockindex->pprev->GetBlockHash().GetHex()));
87  CBlockIndex* pnext = chainActive.Next(blockindex);
88  if (pnext)
89  result.push_back(Pair("nextblockhash", pnext->GetBlockHash().GetHex()));
90 
91  result.push_back(Pair("moneysupply",ValueFromAmount(blockindex->nMoneySupply)));
92 
93  Object zdivObj;
94  for (auto denom : libzerocoin::zerocoinDenomList) {
95  zdivObj.push_back(Pair(to_string(denom), ValueFromAmount(blockindex->mapZerocoinSupply.at(denom) * (denom*COIN))));
96  }
97  zdivObj.emplace_back(Pair("total", ValueFromAmount(blockindex->GetZerocoinSupply())));
98  result.emplace_back(Pair("zDIVsupply", zdivObj));
99 
100  return result;
101 }

551 Value getchaintips(const Array& params, bool fHelp)
552 {
553  if (fHelp || params.size() != 0)
554  throw runtime_error(
555  "getchaintips\n"
556  "Return information about all known tips in the block tree,"
557  " including the main chain as well as orphaned branches.\n"
558  "\nResult:\n"
559  "[\n"
560  " {\n"
561  " \"height\": xxxx, (numeric) height of the chain tip\n"
562  " \"hash\": \"xxxx\", (string) block hash of the tip\n"
563  " \"branchlen\": 0 (numeric) zero for main chain\n"
564  " \"status\": \"active\" (string) \"active\" for the main chain\n"
565  " },\n"
566  " {\n"
567  " \"height\": xxxx,\n"
568  " \"hash\": \"xxxx\",\n"
569  " \"branchlen\": 1 (numeric) length of branch connecting the tip to the main chain\n"
570  " \"status\": \"xxxx\" (string) status of the chain (active, valid-fork, valid-headers, headers-only, invalid)\n"
571  " }\n"
572  "]\n"
573  "Possible values for status:\n"
574  "1. \"invalid\" This branch contains at least one invalid block\n"
575  "2. \"headers-only\" Not all blocks for this branch are available, but the headers are valid\n"
576  "3. \"valid-headers\" All blocks are available for this branch, but they were never fully validated\n"
577  "4. \"valid-fork\" This branch is not part of the active chain, but is fully validated\n"
578  "5. \"active\" This is the tip of the active main chain, which is certainly valid\n"
579  "\nExamples:\n" +
580  HelpExampleCli("getchaintips", "") + HelpExampleRpc("getchaintips", ""));
581 
582  /* Build up a list of chain tips. We start with the list of all
583  known blocks, and successively remove blocks that appear as pprev
584  of another block. */
585  std::set<const CBlockIndex*, CompareBlocksByHeight> setTips;
586  BOOST_FOREACH (const PAIRTYPE(const uint256, CBlockIndex*) & item, mapBlockIndex)
587  setTips.insert(item.second);
588  BOOST_FOREACH (const PAIRTYPE(const uint256, CBlockIndex*) & item, mapBlockIndex) {
589  const CBlockIndex* pprev = item.second->pprev;
590  if (pprev)
591  setTips.erase(pprev);
592  }
593 
594  // Always report the currently active tip.
595  setTips.insert(chainActive.Tip());
596 
597  /* Construct the output array. */
598  Array res;
599  BOOST_FOREACH (const CBlockIndex* block, setTips) {
600  Object obj;
601  obj.push_back(Pair("height", block->nHeight));
602  obj.push_back(Pair("hash", block->phashBlock->GetHex()));
603 
604  const int branchLen = block->nHeight - chainActive.FindFork(block)->nHeight;
605  obj.push_back(Pair("branchlen", branchLen));
606 
607  string status;
608  if (chainActive.Contains(block)) {
609  // This block is part of the currently active chain.
610  status = "active";
611  } else if (block->nStatus & BLOCK_FAILED_MASK) {
612  // This block or one of its ancestors is invalid.
613  status = "invalid";
614  } else if (block->nChainTx == 0) {
615  // This block cannot be connected because full block data for it or one of its parents is missing.
616  status = "headers-only";
617  } else if (block->IsValid(BLOCK_VALID_SCRIPTS)) {
618  // This block is fully validated, but no longer part of the active chain. It was probably the active block once, but was reorganized.
619  status = "valid-fork";
620  } else if (block->IsValid(BLOCK_VALID_TREE)) {
621  // The headers for this block are valid, but it has not been validated. It was probably never part of the most-work chain.
622  status = "valid-headers";
623  } else {
624  // No clue.
625  status = "unknown";
626  }
627  obj.push_back(Pair("status", status));
628 
629  res.push_back(obj);
630  }
631 
632  return res;
633 }
