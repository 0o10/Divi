36 bool CMasternodePaymentDB::Write(const CMasternodePayments& objToSave)
37 {
38  int64_t nStart = GetTimeMillis();
39 
40  // serialize, checksum data up to that point, then append checksum
41  CDataStream ssObj(SER_DISK, CLIENT_VERSION);
42  ssObj << strMagicMessage; // masternode cache file specific magic message
43  ssObj << FLATDATA(Params().MessageStart()); // network specific magic number
44  ssObj << objToSave;
45  uint256 hash = Hash(ssObj.begin(), ssObj.end());
46  ssObj << hash;
47 
48  // open output file, and associate with CAutoFile
49  FILE* file = fopen(pathDB.string().c_str(), "wb");
50  CAutoFile fileout(file, SER_DISK, CLIENT_VERSION);
51  if (fileout.IsNull())
52  return error("%s : Failed to open file %s", __func__, pathDB.string());
53 
54  // Write and commit header, data
55  try {
56  fileout << ssObj;
57  } catch (std::exception& e) {
58  return error("%s : Serialize or I/O error - %s", __func__, e.what());
59  }
60  fileout.fclose();
61 
62  LogPrint("masternode","Written info to mnpayments.dat %dms\n", GetTimeMillis() - nStart);
63 
64  return true;
65 }

67 CMasternodePaymentDB::ReadResult CMasternodePaymentDB::Read(CMasternodePayments& objToLoad, bool fDryRun)
68 {
69  int64_t nStart = GetTimeMillis();
70  // open input file, and associate with CAutoFile
71  FILE* file = fopen(pathDB.string().c_str(), "rb");
72  CAutoFile filein(file, SER_DISK, CLIENT_VERSION);
73  if (filein.IsNull()) {
74  error("%s : Failed to open file %s", __func__, pathDB.string());
75  return FileError;
76  }
77 
78  // use file size to size memory buffer
79  int fileSize = boost::filesystem::file_size(pathDB);
80  int dataSize = fileSize - sizeof(uint256);
81  // Don't try to resize to a negative number if file is small
82  if (dataSize < 0)
83  dataSize = 0;
84  vector<unsigned char> vchData;
85  vchData.resize(dataSize);
86  uint256 hashIn;
87 
88  // read data and checksum from file
89  try {
90  filein.read((char*)&vchData[0], dataSize);
91  filein >> hashIn;
92  } catch (std::exception& e) {
93  error("%s : Deserialize or I/O error - %s", __func__, e.what());
94  return HashReadError;
95  }
96  filein.fclose();
97 
98  CDataStream ssObj(vchData, SER_DISK, CLIENT_VERSION);
99 
100  // verify stored checksum matches input data
101  uint256 hashTmp = Hash(ssObj.begin(), ssObj.end());
102  if (hashIn != hashTmp) {
103  error("%s : Checksum mismatch, data corrupted", __func__);
104  return IncorrectHash;
105  }
106 
107  unsigned char pchMsgTmp[4];
108  std::string strMagicMessageTmp;
109  try {
110  // de-serialize file header (masternode cache file specific magic message) and ..
111  ssObj >> strMagicMessageTmp;
112 
113  // ... verify the message matches predefined one
114  if (strMagicMessage != strMagicMessageTmp) {
115  error("%s : Invalid masternode payement cache magic message", __func__);
116  return IncorrectMagicMessage;
117  }
118 
119 
120  // de-serialize file header (network specific magic number) and ..
121  ssObj >> FLATDATA(pchMsgTmp);
122 
123  // ... verify the network matches ours
124  if (memcmp(pchMsgTmp, Params().MessageStart(), sizeof(pchMsgTmp))) {
125  error("%s : Invalid network magic number", __func__);
126  return IncorrectMagicNumber;
127  }
128 
129  // de-serialize data into CMasternodePayments object
130  ssObj >> objToLoad;
131  } catch (std::exception& e) {
132  objToLoad.Clear();
133  error("%s : Deserialize or I/O error - %s", __func__, e.what());
134  return IncorrectFormat;
135  }
136 
137  LogPrint("masternode","Loaded info from mnpayments.dat %dms\n", GetTimeMillis() - nStart);
138  LogPrint("masternode"," %s\n", objToLoad.ToString());
139  if (!fDryRun) {
140  LogPrint("masternode","Masternode payments manager - cleaning....\n");
141  objToLoad.CleanPaymentList();
142  LogPrint("masternode","Masternode payments manager - result:\n");
143  LogPrint("masternode"," %s\n", objToLoad.ToString());
144  }
145 
146  return Ok;
147 }

176 bool IsBlockValueValid(const CBlock& block, CAmount nExpectedValue, CAmount nMinted)
177 {
178  CBlockIndex* pindexPrev = chainActive.Tip();
179  if (pindexPrev == NULL) return true;
180 
181  int nHeight = 0;
182  if (pindexPrev->GetBlockHash() == block.hashPrevBlock) {
183  nHeight = pindexPrev->nHeight + 1;
184  } else { //out of order
185  BlockMap::iterator mi = mapBlockIndex.find(block.hashPrevBlock);
186  if (mi != mapBlockIndex.end() && (*mi).second)
187  nHeight = (*mi).second->nHeight + 1;
188  }
189 
190  if (nHeight == 0) {
191  LogPrint("masternode","IsBlockValueValid() : WARNING: Couldn't find previous block\n");
192  }
193 
194  //LogPrintf("XX69----------> IsBlockValueValid(): nMinted: %d, nExpectedValue: %d\n", FormatMoney(nMinted), FormatMoney(nExpectedValue));
195 
196  if (!masternodeSync.IsSynced()) { //there is no budget data to use to check anything
197  //super blocks will always be on these blocks, max 100 per budgeting
198  if (nHeight % GetBudgetPaymentCycleBlocks() < 100) {
199  return true;
200  } else {
201  if (nMinted > nExpectedValue) {
202  return false;
203  }
204  }
205  } else { // we're synced and have data so check the budget schedule
206 
207  //are these blocks even enabled
208  if (!IsSporkActive(SPORK_13_ENABLE_SUPERBLOCKS)) {
209  return nMinted <= nExpectedValue;
210  }
211 
212  if (budget.IsBudgetPaymentBlock(nHeight)) {
213  //the value of the block is evaluated in CheckBlock
214  return true;
215  } else {
216  if (nMinted > nExpectedValue) {
217  return false;
218  }
219  }
220  }
221 
222  return true;
223 }

283 void CMasternodePayments::FillBlockPayee(CMutableTransaction& txNew, int64_t nFees, bool fProofOfStake)
284 {
285  CBlockIndex* pindexPrev = chainActive.Tip();
286  if (!pindexPrev) return;
287 
288  bool hasPayment = true;
289  CScript payee;
290 
291  //spork
292  if (!masternodePayments.GetBlockPayee(pindexPrev->nHeight + 1, payee)) {
293  //no masternode detected
294  CMasternode* winningNode = mnodeman.GetCurrentMasterNode(1);
295  if (winningNode) {
296  payee = GetScriptForDestination(winningNode->pubKeyCollateralAddress.GetID());
297  } else {
298  LogPrint("masternode","CreateNewBlock: Failed to detect masternode to pay\n");
299  hasPayment = false;
300  }
301  }
302 
303  CAmount blockValue = GetBlockValue(pindexPrev->nHeight);
304  CAmount masternodePayment = GetMasternodePayment(pindexPrev->nHeight, blockValue);
305 
306  if (hasPayment) {
307  if (fProofOfStake) {
313  unsigned int i = txNew.vout.size();
314  txNew.vout.resize(i + 1);
315  txNew.vout[i].scriptPubKey = payee;
316  txNew.vout[i].nValue = masternodePayment;
317 
318  //subtract mn payment from the stake reward
319  txNew.vout[i - 1].nValue -= masternodePayment;
320  } else {
321  txNew.vout.resize(2);
322  txNew.vout[1].scriptPubKey = payee;
323  txNew.vout[1].nValue = masternodePayment;
324  txNew.vout[0].nValue = blockValue - masternodePayment;
325  }
326 
327  CTxDestination address1;
328  ExtractDestination(payee, address1);
329  CBitcoinAddress address2(address1);
330 
331  LogPrint("masternode","Masternode payment of %s to %s\n", FormatMoney(masternodePayment).c_str(), address2.ToString().c_str());
332  }

343 void CMasternodePayments::ProcessMessageMasternodePayments(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
344 {
345  if (!masternodeSync.IsBlockchainSynced()) return;
346 
347  if (fLiteMode) return; //disable all Obfuscation/Masternode related functionality
348 
349 
350  if (strCommand == "mnget") { //Masternode Payments Request Sync
351  if (fLiteMode) return; //disable all Obfuscation/Masternode related functionality
352 
353  int nCountNeeded;
354  vRecv >> nCountNeeded;
355 
356  if (Params().NetworkID() == CBaseChainParams::MAIN) {
357  if (pfrom->HasFulfilledRequest("mnget")) {
358  LogPrint("masternode","mnget - peer already asked me for the list\n");
359  Misbehaving(pfrom->GetId(), 20);

361  }
362  }
363

365  masternodePayments.Sync(pfrom, nCountNeeded);
366  LogPrint("mnpayments", "mnget - Sent Masternode winners to peer %i\n", pfrom->GetId());
367  } else if (strCommand == "mnw") { //Masternode Payments Declare Winner
368  //this is required in litemodef
369  CMasternodePaymentWinner winner;

377  if (!locked || chainActive.Tip() == NULL) return;

390  return;
391  }
392 
393  std::string strError = "";
394  if (!winner.IsValid(pfrom, strError)) {
395  // if(strError != "") LogPrint("masternode","mnw - invalid message - %s\n", strError);
396  return;
397  }
398 
399  if (!masternodePayments.CanVote(winner.vinMasternode.prevout, winner.nBlockHeight)) {
400  // LogPrint("masternode","mnw - masternode already voted - %s\n", winner.vinMasternode.prevout.ToStringShort());
401  return;
402  }
403 
404  if (!winner.SignatureValid()) {
405  // LogPrint("masternode","mnw - invalid signature\n");
406  if (masternodeSync.IsSynced()) Misbehaving(pfrom->GetId(), 20);
407  // it could just be a non-synced masternode
408  mnodeman.AskForMN(pfrom, winner.vinMasternode);
409  return;
410  }
411 
412  CTxDestination address1;
413  ExtractDestination(winner.payee, address1);
414  CBitcoinAddress address2(address1);
415 
416  // LogPrint("mnpayments", "mnw - winning vote - Addr %s Height %d bestHeight %d - %s\n", address2.ToString().c_str(), winner.nBlockHeight, nHeight, winner.vinMasternode.prevout.ToStringShort());
417 
418  if (masternodePayments.AddWinningMasternode(winner)) {
419  winner.Relay();
420  masternodeSync.AddedMasternodeWinner(winner.GetHash());
421  }

514 bool CMasternodeBlockPayees::IsTransactionValid(const CTransaction& txNew)
515 {
516  LOCK(cs_vecPayments);
517 
518  int nMaxSignatures = 0;
519  int nMasternode_Drift_Count = 0;
520 
521  std::string strPayeesPossible = "";
522 
523  CAmount nReward = GetBlockValue(nBlockHeight);
524 
525  if (IsSporkActive(SPORK_8_MASTERNODE_PAYMENT_ENFORCEMENT)) {
526  // Get a stable number of masternodes by ignoring newly activated (< 8000 sec old) masternodes
527  nMasternode_Drift_Count = mnodeman.stable_size() + Params().MasternodeCountDrift();
528  }
529  else {
530  //account for the fact that all peers do not see the same masternode count. A allowance of being off our masternode count is given
531  //we only need to look at an increased masternode count because as count increases, the reward decreases. This code only checks
532  //for mnPayment >= required, so it only makes sense to check the max node count allowed.
533  nMasternode_Drift_Count = mnodeman.size() + Params().MasternodeCountDrift();
534  }
535 
536  CAmount requiredMasternodePayment = GetMasternodePayment(nBlockHeight, nReward, nMasternode_Drift_Count);
537 
538  //require at least 6 signatures
539  BOOST_FOREACH (CMasternodePayee& payee, vecPayments)
540  if (payee.nVotes >= nMaxSignatures && payee.nVotes >= MNPAYMENTS_SIGNATURES_REQUIRED)
541  nMaxSignatures = payee.nVotes;
542 
543  // if we don't have at least 6 signatures on a payee, approve whichever is the longest chain
544  if (nMaxSignatures < MNPAYMENTS_SIGNATURES_REQUIRED) return true;
545 
546  BOOST_FOREACH (CMasternodePayee& payee, vecPayments) {
547  bool found = false;
548  BOOST_FOREACH (CTxOut out, txNew.vout) {
549  if (payee.scriptPubKey == out.scriptPubKey) {
550  if(out.nValue >= requiredMasternodePayment)
551  found = true;
552  else
553  LogPrint("masternode","Masternode payment is out of drift range. Paid=%s Min=%s\n", FormatMoney(out.nValue).c_str(), FormatMoney(requiredMasternodePayment).c_str());
554  }
555  }
556 
557  if (payee.nVotes >= MNPAYMENTS_SIGNATURES_REQUIRED) {
558  if (found) return true;
559 
560  CTxDestination address1;
561  ExtractDestination(payee.scriptPubKey, address1);
562  CBitcoinAddress address2(address1);
563 
564  if (strPayeesPossible == "") {
565  strPayeesPossible += address2.ToString();
566  } else {
567  strPayeesPossible += "," + address2.ToString();
568  }
569  }
570  }
571 
572  LogPrint("masternode","CMasternodePayments::IsTransactionValid - Missing required payment of %s to %s\n", FormatMoney(requiredMasternodePayment).c_str(), strPayeesPossible.c_str());
573  return false;
574 }
