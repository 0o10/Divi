22 string FormatScript(const CScript& script)
23 {
24  string ret;
25  CScript::const_iterator it = script.begin();
26  opcodetype op;
27  while (it != script.end()) {
28  CScript::const_iterator it2 = it;
29  vector<unsigned char> vch;
30  if (script.GetOp2(it, op, &vch)) {
31  if (op == OP_0) {
32  ret += "0 ";
33  continue;
34  } else if ((op >= OP_1 && op <= OP_16) || op == OP_1NEGATE) {
35  ret += strprintf("%i ", op - OP_1NEGATE - 1);
36  continue;
37  } else if (op >= OP_NOP && op <= OP_CHECKMULTISIGVERIFY) {
38  string str(GetOpName(op));
39  if (str.substr(0, 3) == string("OP_")) {
40  ret += str.substr(3, string::npos) + " ";
41  continue;
42  }
43  }
44  if (vch.size() > 0) {
45  ret += strprintf("0x%x 0x%x ", HexStr(it2, it - vch.size()), HexStr(it - vch.size(), it));
46  } else {
47  ret += strprintf("0x%x", HexStr(it2, it));
48  }
49  continue;
50  }
51  ret += strprintf("0x%x ", HexStr(it2, script.end()));
52  break;
53  }
54  return ret.substr(0, ret.size() - 1);
55 }

90 void TxToUniv(const CTransaction& tx, const uint256& hashBlock, UniValue& entry)
91 {
92  entry.pushKV("txid", tx.GetHash().GetHex());
93  entry.pushKV("version", tx.nVersion);
94  entry.pushKV("locktime", (int64_t)tx.nLockTime);
95 
96  UniValue vin(UniValue::VARR);
97  BOOST_FOREACH (const CTxIn& txin, tx.vin) {
98  UniValue in(UniValue::VOBJ);
99  if (tx.IsCoinBase())
100  in.pushKV("coinbase", HexStr(txin.scriptSig.begin(), txin.scriptSig.end()));
101  else {
102  in.pushKV("txid", txin.prevout.hash.GetHex());
103  in.pushKV("vout", (int64_t)txin.prevout.n);
104  UniValue o(UniValue::VOBJ);
105  o.pushKV("asm", txin.scriptSig.ToString());
106  o.pushKV("hex", HexStr(txin.scriptSig.begin(), txin.scriptSig.end()));

108  }
109  in.pushKV("sequence", (int64_t)txin.nSequence);
110  vin.push_back(in);
111  }
112  entry.pushKV("vin", vin);
113 
114  UniValue vout(UniValue::VARR);
115  for (unsigned int i = 0; i < tx.vout.size(); i++) {
116  const CTxOut& txout = tx.vout[i];
117 
118  UniValue out(UniValue::VOBJ);
119 
120  UniValue outValue(UniValue::VNUM, FormatMoney(txout.nValue));
121  out.pushKV("value", outValue);
122  out.pushKV("n", (int64_t)i);
123 
124  UniValue o(UniValue::VOBJ);

126  out.pushKV("scriptPubKey", o);

130

133

135 }
