19 //Get the last hash that matches the modulus given. Processed in reverse order
20 bool GetBlockHash(uint256& hash, int nBlockHeight)
21 {
22  if (chainActive.Tip() == NULL) return false;
23 
24  if (nBlockHeight == 0)
25  nBlockHeight = chainActive.Tip()->nHeight;
26 
27  if (mapCacheBlockHashes.count(nBlockHeight)) {
28  hash = mapCacheBlockHashes[nBlockHeight];
29  return true;
30  }
31 
32  const CBlockIndex* BlockLastSolved = chainActive.Tip();
33  const CBlockIndex* BlockReading = chainActive.Tip();
34 
35  if (BlockLastSolved == NULL || BlockLastSolved->nHeight == 0 || chainActive.Tip()->nHeight + 1 < nBlockHeight) return false;
36 
37  int nBlocksAgo = 0;
38  if (nBlockHeight > 0) nBlocksAgo = (chainActive.Tip()->nHeight + 1) - nBlockHeight;
39  assert(nBlocksAgo >= 0);
40 
41  int n = 0;
42  for (unsigned int i = 1; BlockReading && BlockReading->nHeight > 0; i++) {
43  if (n >= nBlocksAgo) {
44  hash = BlockReading->GetBlockHash();
45  mapCacheBlockHashes[nBlockHeight] = hash;
46  return true;
47  }
48  n++;
49 
50  if (BlockReading->pprev == NULL) {
51  assert(BlockReading);
52  break;
53  }
54  BlockReading = BlockReading->pprev;
55  }
56 
57  return false;
58 }

158 //
159 // Deterministically calculate a given "score" for a Masternode depending on how close it's hash is to
160 // the proof of work for that block. The further away they are the better, the furthest will win the election
161 // and get paid this block
162 //
163 uint256 CMasternode::CalculateScore(int mod, int64_t nBlockHeight)
164 {
165  if (chainActive.Tip() == NULL) return 0;
166 
167  uint256 hash = 0;
168  uint256 aux = vin.prevout.hash + vin.prevout.n;
169 
170  if (!GetBlockHash(hash, nBlockHeight)) {
171  LogPrint("masternode","CalculateScore ERROR - nHeight %d - Returned 0\n", nBlockHeight);
172  return 0;
173  }
174 
175  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
176  ss << hash;
177  uint256 hash2 = ss.GetHash();
178 
179  CHashWriter ss2(SER_GETHASH, PROTOCOL_VERSION);
180  ss2 << hash;
181  ss2 << aux;
182  uint256 hash3 = ss2.GetHash();
183 
184  uint256 r = (hash3 > hash2 ? hash3 - hash2 : hash2 - hash3);
185 
186  return r;
187 }

189 void CMasternode::Check(bool forceCheck)
190 {
191  if (ShutdownRequested()) return;
192 
193  if (!forceCheck && (GetTime() - lastTimeChecked < MASTERNODE_CHECK_SECONDS)) return;
194  lastTimeChecked = GetTime();
195 
196 
197  //once spent, stop doing the checks
198  if (activeState == MASTERNODE_VIN_SPENT) return;
199 
200 
201  if (!IsPingedWithin(MASTERNODE_REMOVAL_SECONDS)) {
202  activeState = MASTERNODE_REMOVE;
203  return;
204  }
205 
206  if (!IsPingedWithin(MASTERNODE_EXPIRATION_SECONDS)) {
207  activeState = MASTERNODE_EXPIRED;
208  return;
209  }
210 
211  if (!unitTest) {
212  CValidationState state;
213  CMutableTransaction tx = CMutableTransaction();
214  CTxOut vout = CTxOut(9999.99 * COIN, obfuScationPool.collateralPubKey);
215  tx.vin.push_back(vin);
216  tx.vout.push_back(vout);
217 
218  {
219  TRY_LOCK(cs_main, lockMain);
220  if (!lockMain) return;
221 
222  if (!AcceptableInputs(mempool, state, CTransaction(tx), false, NULL)) {
223  activeState = MASTERNODE_VIN_SPENT;
224  return;
225  }
226  }
227  }
228 
229  activeState = MASTERNODE_ENABLED; // OK
230 }

232 int64_t CMasternode::SecondsSincePayment()
233 {
234  CScript pubkeyScript;
235  pubkeyScript = GetScriptForDestination(pubKeyCollateralAddress.GetID());
236 
237  int64_t sec = (GetAdjustedTime() - GetLastPaid());
238  int64_t month = 60 * 60 * 24 * 30;
239  if (sec < month) return sec; //if it's less than 30 days, give seconds
240 
241  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
242  ss << vin;
243  ss << sigTime;
244  uint256 hash = ss.GetHash();
245 
246  // return some deterministic value for unknown/unpaid but force it to be more than 30 days old
247  return month + hash.GetCompact(false);

250 int64_t CMasternode::GetLastPaid()
251 {
252  CBlockIndex* pindexPrev = chainActive.Tip();
253  if (pindexPrev == NULL) return false;
254 
255  CScript mnpayee;
256  mnpayee = GetScriptForDestination(pubKeyCollateralAddress.GetID());
257 
258  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
259  ss << vin;
260  ss << sigTime;
261  uint256 hash = ss.GetHash();
262 
263  // use a deterministic offset to break a tie -- 2.5 minutes
264  int64_t nOffset = hash.GetCompact(false) % 150;
265 
266  if (chainActive.Tip() == NULL) return false;
267 
268  const CBlockIndex* BlockReading = chainActive.Tip();
269 
270  int nMnCount = mnodeman.CountEnabled() * 1.25;
271  int n = 0;
272  for (unsigned int i = 1; BlockReading && BlockReading->nHeight > 0; i++) {
273  if (n >= nMnCount) {
274  return 0;
275  }
276  n++;
277 
278  if (masternodePayments.mapMasternodeBlocks.count(BlockReading->nHeight)) {
279  /*
280  Search for this payee, with at least 2 votes. This will aid in consensus allowing the network
281  to converge on the same payees quickly, then keep the same schedule.
282  */
283  if (masternodePayments.mapMasternodeBlocks[BlockReading->nHeight].HasPayeeWithVotes(mnpayee, 2)) {
284  return BlockReading->nTime + nOffset;
285  }
286  }
287 
288  if (BlockReading->pprev == NULL) {
289  assert(BlockReading);
290  break;
291  }
292  BlockReading = BlockReading->pprev;
293  }
294 
295  return 0;
296 }

469 bool CMasternodeBroadcast::CheckAndUpdate(int& nDos)
470 {
471  // make sure signature isn't in the future (past is OK)
472  if (sigTime > GetAdjustedTime() + 60 * 60) {
473  LogPrint("masternode","mnb - Signature rejected, too far into the future %s\n", vin.prevout.hash.ToString());
474  nDos = 1;
475  return false;
476  }
477 
478  std::string vchPubKey(pubKeyCollateralAddress.begin(), pubKeyCollateralAddress.end());
479  std::string vchPubKey2(pubKeyMasternode.begin(), pubKeyMasternode.end());
480  std::string strMessage = addr.ToString() + boost::lexical_cast<std::string>(sigTime) + vchPubKey + vchPubKey2 + boost::lexical_cast<std::string>(protocolVersion);
481 
482  if (protocolVersion < masternodePayments.GetMinMasternodePaymentsProto()) {
483  LogPrint("masternode","mnb - ignoring outdated Masternode %s protocol version %d\n", vin.prevout.hash.ToString(), protocolVersion);
484  return false;
485  }
486 
487  CScript pubkeyScript;
488  pubkeyScript = GetScriptForDestination(pubKeyCollateralAddress.GetID());
489 
490  if (pubkeyScript.size() != 25) {
491  LogPrint("masternode","mnb - pubkey the wrong size\n");
492  nDos = 100;
493  return false;
494  }
495 
496  CScript pubkeyScript2;
497  pubkeyScript2 = GetScriptForDestination(pubKeyMasternode.GetID());
498 
499  if (pubkeyScript2.size() != 25) {
500  LogPrint("masternode","mnb - pubkey2 the wrong size\n");
501  nDos = 100;
502  return false;
503  }
504 
505  if (!vin.scriptSig.empty()) {
506  LogPrint("masternode","mnb - Ignore Not Empty ScriptSig %s\n", vin.prevout.hash.ToString());
507  return false;
508  }
509 
510  std::string errorMessage = "";
511  if (!obfuScationSigner.VerifyMessage(pubKeyCollateralAddress, sig, strMessage, errorMessage)) {
512  LogPrint("masternode","mnb - Got bad Masternode address signature\n");
513  nDos = 100;
514  return false;
515  }
516 
517  if (Params().NetworkID() == CBaseChainParams::MAIN) {
518  if (addr.GetPort() != 51472) return false;
519  } else if (addr.GetPort() == 51472)
520  return false;
521 
522  //search existing Masternode list, this is where we update existing Masternodes with new mnb broadcasts
523  CMasternode* pmn = mnodeman.Find(vin);
524 
525  // no such masternode, nothing to update
526  if (pmn == NULL)
527  return true;
528  else {
529  // this broadcast older than we have, it's bad.
530  if (pmn->sigTime > sigTime) {
531  LogPrint("masternode","mnb - Bad sigTime %d for Masternode %s (existing broadcast is at %d)\n",
532  sigTime, vin.prevout.hash.ToString(), pmn->sigTime);
533  return false;
534  }
535  // masternode is not enabled yet/already, nothing to update
536  if (!pmn->IsEnabled()) return true;
537  }
538 
539  // mn.pubkey = pubkey, IsVinAssociatedWithPubkey is validated once below,
540  // after that they just need to match
541  if (pmn->pubKeyCollateralAddress == pubKeyCollateralAddress && !pmn->IsBroadcastedWithin(MASTERNODE_MIN_MNB_SECONDS)) {
542  //take the newest entry
543  LogPrint("masternode","mnb - Got updated entry for %s\n", vin.prevout.hash.ToString());
544  if (pmn->UpdateFromNewBroadcast((*this))) {
545  pmn->Check();
546  if (pmn->IsEnabled()) Relay();
547  }
548  masternodeSync.AddedMasternodeList(GetHash());
549  }
550 
551  return true;
552 }

554 bool CMasternodeBroadcast::CheckInputsAndAdd(int& nDoS)
555 {
556  // we are a masternode with the same vin (i.e. already activated) and this mnb is ours (matches our Masternode privkey)
557  // so nothing to do here for us
558  if (fMasterNode && vin.prevout == activeMasternode.vin.prevout && pubKeyMasternode == activeMasternode.pubKeyMasternode)
559  return true;
560 
561  // search existing Masternode list
562  CMasternode* pmn = mnodeman.Find(vin);
563 
564  if (pmn != NULL) {
565  // nothing to do here if we already know about this masternode and it's enabled
566  if (pmn->IsEnabled()) return true;
567  // if it's not enabled, remove old MN first and continue
568  else
569  mnodeman.Remove(pmn->vin);
570  }
571 
572  CValidationState state;
573  CMutableTransaction tx = CMutableTransaction();
574  CTxOut vout = CTxOut(9999.99 * COIN, obfuScationPool.collateralPubKey);
575  tx.vin.push_back(vin);
576  tx.vout.push_back(vout);
577 
578  {
579  TRY_LOCK(cs_main, lockMain);
580  if (!lockMain) {
581  // not mnb fault, let it to be checked again later
582  mnodeman.mapSeenMasternodeBroadcast.erase(GetHash());
583  masternodeSync.mapSeenSyncMNB.erase(GetHash());
584  return false;
585  }
586 
587  if (!AcceptableInputs(mempool, state, CTransaction(tx), false, NULL)) {
588  //set nDos
589  state.IsInvalid(nDoS);
590  return false;
591  }
592  }
593 
594  LogPrint("masternode", "mnb - Accepted Masternode entry\n");
595 
596  if (GetInputAge(vin) < MASTERNODE_MIN_CONFIRMATIONS) {
597  LogPrint("masternode","mnb - Input must have at least %d confirmations\n", MASTERNODE_MIN_CONFIRMATIONS);
598  // maybe we miss few blocks, let this mnb to be checked again later
599  mnodeman.mapSeenMasternodeBroadcast.erase(GetHash());
600  masternodeSync.mapSeenSyncMNB.erase(GetHash());
601  return false;
602  }
603 
604  // verify that sig time is legit in past
605  // should be at least not earlier than block when 1000 PIV tx got MASTERNODE_MIN_CONFIRMATIONS
606  uint256 hashBlock = 0;
607  CTransaction tx2;
608  GetTransaction(vin.prevout.hash, tx2, hashBlock, true);
609  BlockMap::iterator mi = mapBlockIndex.find(hashBlock);

705 bool CMasternodePing::CheckAndUpdate(int& nDos, bool fRequireEnabled)
706 {
707  if (sigTime > GetAdjustedTime() + 60 * 60) {
708  LogPrint("masternode","CMasternodePing::CheckAndUpdate - Signature rejected, too far into the future %s\n", vin.prevout.hash.ToString());
709  nDos = 1;
710  return false;
711  }
712 
713  if (sigTime <= GetAdjustedTime() - 60 * 60) {
714  LogPrint("masternode","CMasternodePing::CheckAndUpdate - Signature rejected, too far into the past %s - %d %d \n", vin.prevout.hash.ToString(), sigTime, GetAdjustedTime());
715  nDos = 1;
716  return false;
717  }
718 
719  LogPrint("masternode","CMasternodePing::CheckAndUpdate - New Ping - %s - %lli\n", blockHash.ToString(), sigTime);
720 
721  // see if we have this Masternode
722  CMasternode* pmn = mnodeman.Find(vin);
723  if (pmn != NULL && pmn->protocolVersion >= masternodePayments.GetMinMasternodePaymentsProto()) {
724  if (fRequireEnabled && !pmn->IsEnabled()) return false;
725 
726  // LogPrint("masternode","mnping - Found corresponding mn for vin: %s\n", vin.ToString());
727  // update only if there is no known ping for this masternode or
728  // last ping was more then MASTERNODE_MIN_MNP_SECONDS-60 ago comparing to this one
729  if (!pmn->IsPingedWithin(MASTERNODE_MIN_MNP_SECONDS - 60, sigTime)) {
730  std::string strMessage = vin.ToString() + blockHash.ToString() + boost::lexical_cast<std::string>(sigTime);
731 
732  std::string errorMessage = "";
733  if (!obfuScationSigner.VerifyMessage(pmn->pubKeyMasternode, vchSig, strMessage, errorMessage)) {
734  LogPrint("masternode","CMasternodePing::CheckAndUpdate - Got bad Masternode address signature %s\n", vin.prevout.hash.ToString());
735  nDos = 33;
736  return false;
737  }
738 
739  BlockMap::iterator mi = mapBlockIndex.find(blockHash);
740  if (mi != mapBlockIndex.end() && (*mi).second) {
741  if ((*mi).second->nHeight < chainActive.Height() - 24) {
742  LogPrint("masternode","CMasternodePing::CheckAndUpdate - Masternode %s block hash %s is too old\n", vin.prevout.hash.ToString(), blockHash.ToString());
743  // Do nothing here (no Masternode update, no mnping relay)
744  // Let this node to be visible but fail to accept mnping
745 
746  return false;
747  }
748  } else {
749  if (fDebug) LogPrint("masternode","CMasternodePing::CheckAndUpdate - Masternode %s block hash %s is unknown\n", vin.prevout.hash.ToString(), blockHash.ToString());
750  // maybe we stuck so we shouldn't ban this node, just fail to accept it
751  // TODO: or should we also request this block?
752 
753  return false;
754  }
755 
756  pmn->lastPing = *this;
757 
758  //mnodeman.mapSeenMasternodeBroadcast.lastPing is probably outdated, so we'll update it
759  CMasternodeBroadcast mnb(*pmn);
760  uint256 hash = mnb.GetHash();
761  if (mnodeman.mapSeenMasternodeBroadcast.count(hash)) {
762  mnodeman.mapSeenMasternodeBroadcast[hash].lastPing = *this;
763  }
764 
765  pmn->Check(true);
766  if (!pmn->IsEnabled()) return false;
767 
768  LogPrint("masternode", "CMasternodePing::CheckAndUpdate - Masternode ping accepted, vin: %s\n", vin.prevout.hash.ToString());
769 
770  Relay();
771  return true;
772  }
773  LogPrint("masternode", "CMasternodePing::CheckAndUpdate - Masternode ping arrived too early, vin: %s\n", vin.prevout.hash.ToString());
774  //nDos = 1; //disable, this is happening frequently and causing banned peers
775  return false;
776  }
777  LogPrint("masternode", "CMasternodePing::CheckAndUpdate - Couldn't find compatible Masternode entry, vin: %s\n", vin.prevout.hash.ToString());
778 
779  return false;
780 }
