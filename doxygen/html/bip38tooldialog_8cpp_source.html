112 void Bip38ToolDialog::on_encryptKeyButton_ENC_clicked()
113 {
114  if (!model)
115  return;
116 
117  QString qstrPassphrase = ui->passphraseIn_ENC->text();
118  QString strInvalid;
119  if (!isValidPassphrase(qstrPassphrase, strInvalid)) {
120  ui->statusLabel_ENC->setStyleSheet("QLabel { color: red; }");
121  ui->statusLabel_ENC->setText(tr("The entered passphrase is invalid. ") + strInvalid + QString(" is not valid") + QString(" ") + tr("Allowed: 0-9,a-z,A-Z,") + specialChar);
122  return;
123  }
124 
125  CBitcoinAddress addr(ui->addressIn_ENC->text().toStdString());
126  if (!addr.IsValid()) {
127  ui->statusLabel_ENC->setStyleSheet("QLabel { color: red; }");
128  ui->statusLabel_ENC->setText(tr("The entered address is invalid.") + QString(" ") + tr("Please check the address and try again."));
129  return;
130  }
131 
132  CKeyID keyID;
133  if (!addr.GetKeyID(keyID)) {
134  ui->addressIn_ENC->setValid(false);
135  ui->statusLabel_ENC->setStyleSheet("QLabel { color: red; }");
136  ui->statusLabel_ENC->setText(tr("The entered address does not refer to a key.") + QString(" ") + tr("Please check the address and try again."));
137  return;
138  }
139 
140  WalletModel::UnlockContext ctx(model->requestUnlock(true));
141  if (!ctx.isValid()) {
142  ui->statusLabel_ENC->setStyleSheet("QLabel { color: red; }");
143  ui->statusLabel_ENC->setText(tr("Wallet unlock was cancelled."));
144  return;
145  }
146 
147  CKey key;
148  if (!pwalletMain->GetKey(keyID, key)) {
149  ui->statusLabel_ENC->setStyleSheet("QLabel { color: red; }");
150  ui->statusLabel_ENC->setText(tr("Private key for the entered address is not available."));
151  return;
152  }
153 
154  std::string encryptedKey = BIP38_Encrypt(addr.ToString(), qstrPassphrase.toStdString(), key.GetPrivKey_256(), key.IsCompressed());
155  ui->encryptedKeyOut_ENC->setText(QString::fromStdString(encryptedKey));
156 }

195 void Bip38ToolDialog::on_importAddressButton_DEC_clicked()
196 {
197  WalletModel::UnlockContext ctx(model->requestUnlock(true));
198  if (!ctx.isValid()) {
199  ui->statusLabel_DEC->setStyleSheet("QLabel { color: red; }");
200  ui->statusLabel_DEC->setText(tr("Wallet unlock was cancelled."));
201  return;
202  }
203 
204  CBitcoinAddress address(ui->addressOut_DEC->text().toStdString());
205  CPubKey pubkey = key.GetPubKey();
206 
207  if (!address.IsValid() || !key.IsValid() || CBitcoinAddress(pubkey.GetID()).ToString() != address.ToString()) {
208  ui->statusLabel_DEC->setStyleSheet("QLabel { color: red; }");
209  ui->statusLabel_DEC->setText(tr("Data Not Valid.") + QString(" ") + tr("Please try again."));
210  return;
211  }
212 
213  CKeyID vchAddress = pubkey.GetID();
214  {
215  ui->statusLabel_DEC->setStyleSheet("QLabel { color: red; }");
216  ui->statusLabel_DEC->setText(tr("Please wait while key is imported"));
217 
218  pwalletMain->MarkDirty();
219  pwalletMain->SetAddressBook(vchAddress, "", "receive");
220 
221  // Don't throw error in case a key is already there
222  if (pwalletMain->HaveKey(vchAddress)) {
223  ui->statusLabel_DEC->setStyleSheet("QLabel { color: red; }");
224  ui->statusLabel_DEC->setText(tr("Key Already Held By Wallet"));
225  return;
226  }
227 
228  pwalletMain->mapKeyMetadata[vchAddress].nCreateTime = 1;
229 
230  if (!pwalletMain->AddKeyPubKey(key, pubkey)) {
231  ui->statusLabel_DEC->setStyleSheet("QLabel { color: red; }");
232  ui->statusLabel_DEC->setText(tr("Error Adding Key To Wallet"));
233  return;
234  }
235 
236  // whenever a key is imported, we need to scan the whole chain
237  pwalletMain->nTimeFirstKey = 1; // 0 would be considered 'no value'
238  pwalletMain->ScanForWalletTransactions(chainActive.Genesis(), true);
239  }
240 
241  ui->statusLabel_DEC->setStyleSheet("QLabel { color: green; }");
242  ui->statusLabel_DEC->setText(tr("Successfully Added Private Key To Wallet"));
243 }
