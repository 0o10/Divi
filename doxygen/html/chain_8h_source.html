423 class CDiskBlockIndex : public CBlockIndex

429  CDiskBlockIndex()
430  {

435  explicit CDiskBlockIndex(CBlockIndex* pindex) : CBlockIndex(*pindex)
436  {
437  hashPrev = (pprev ? pprev->GetBlockHash() : uint256());
438  }

440  ADD_SERIALIZE_METHODS;
441 
442  template <typename Stream, typename Operation>
443  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion)
444  {
445  if (!(nType & SER_GETHASH))
446  READWRITE(VARINT(nVersion));
447 
448  READWRITE(VARINT(nHeight));
449  READWRITE(VARINT(nStatus));
450  READWRITE(VARINT(nTx));
451  if (nStatus & (BLOCK_HAVE_DATA | BLOCK_HAVE_UNDO))
452  READWRITE(VARINT(nFile));
453  if (nStatus & BLOCK_HAVE_DATA)
454  READWRITE(VARINT(nDataPos));
455  if (nStatus & BLOCK_HAVE_UNDO)
456  READWRITE(VARINT(nUndoPos));
457 
458 
459  READWRITE(nMint);
460  READWRITE(nMoneySupply);
461  READWRITE(nFlags);
462  READWRITE(nStakeModifier);
463  if (IsProofOfStake()) {
464  READWRITE(prevoutStake);
465  READWRITE(nStakeTime);
466  } else {
467  const_cast<CDiskBlockIndex*>(this)->prevoutStake.SetNull();
468  const_cast<CDiskBlockIndex*>(this)->nStakeTime = 0;
469  const_cast<CDiskBlockIndex*>(this)->hashProofOfStake = uint256();
470  }
471 
472  // block header
473  READWRITE(this->nVersion);
474  READWRITE(hashPrev);
475  READWRITE(hashNext);
476  READWRITE(hashMerkleRoot);
477  READWRITE(nTime);
478  READWRITE(nBits);
479  READWRITE(nNonce);
480  if(this->nVersion > 3) {
481  READWRITE(nAccumulatorCheckpoint);
482  READWRITE(mapZerocoinSupply);
483  READWRITE(vMintDenominationsInBlock);
484  }
485 
486  }
487 
488  uint256 GetBlockHash() const
489  {
490  CBlockHeader block;
491  block.nVersion = nVersion;
492  block.hashPrevBlock = hashPrev;
493  block.hashMerkleRoot = hashMerkleRoot;
494  block.nTime = nTime;
495  block.nBits = nBits;
496  block.nNonce = nNonce;
497  block.nAccumulatorCheckpoint = nAccumulatorCheckpoint;
498  return block.GetHash();
499  }

503  {
