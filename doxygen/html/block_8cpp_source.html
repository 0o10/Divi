24 uint256 CBlock::BuildMerkleTree(bool* fMutated) const
25 {
26  /* WARNING! If you're reading this because you're learning about crypto
27  and/or designing a new system that will use merkle trees, keep in mind
28  that the following merkle tree algorithm has a serious flaw related to
29  duplicate txids, resulting in a vulnerability (CVE-2012-2459).
30 
31  The reason is that if the number of hashes in the list at a given time
32  is odd, the last one is duplicated before computing the next level (which
33  is unusual in Merkle trees). This results in certain sequences of
34  transactions leading to the same merkle root. For example, these two
35  trees:
36 
37  A A
38  / \ / \
39  B C B C
40  / \ | / \ / \
41  D E F D E F F
42  / \ / \ / \ / \ / \ / \ / \
43  1 2 3 4 5 6 1 2 3 4 5 6 5 6
44 
45  for transaction lists [1,2,3,4,5,6] and [1,2,3,4,5,6,5,6] (where 5 and
46  6 are repeated) result in the same root hash A (because the hash of both
47  of (F) and (F,F) is C).
48 
49  The vulnerability results from being able to send a block with such a
50  transaction list, with the same merkle root, and the same block hash as
51  the original without duplication, resulting in failed validation. If the
52  receiving node proceeds to mark that block as permanently invalid
53  however, it will fail to accept further unmodified (and thus potentially
54  valid) versions of the same block. We defend against this by detecting
55  the case where we would hash two identical hashes at the end of the list
56  together, and treating that identically to the block having an invalid
57  merkle root. Assuming no double-SHA256 collisions, this will detect all
58  known ways of changing the transactions without affecting the merkle
59  root.
60  */
61  vMerkleTree.clear();
62  vMerkleTree.reserve(vtx.size() * 2 + 16); // Safe upper bound for the number of total nodes.
63  for (std::vector<CTransaction>::const_iterator it(vtx.begin()); it != vtx.end(); ++it)
64  vMerkleTree.push_back(it->GetHash());
65  int j = 0;
66  bool mutated = false;
67  for (int nSize = vtx.size(); nSize > 1; nSize = (nSize + 1) / 2)
68  {
69  for (int i = 0; i < nSize; i += 2)
70  {
71  int i2 = std::min(i+1, nSize-1);
72  if (i2 == i + 1 && i2 + 1 == nSize && vMerkleTree[j+i] == vMerkleTree[j+i2]) {
73  // Two identical hashes at the end of the list at a particular level.
74  mutated = true;
75  }
76  vMerkleTree.push_back(Hash(BEGIN(vMerkleTree[j+i]), END(vMerkleTree[j+i]),
77  BEGIN(vMerkleTree[j+i2]), END(vMerkleTree[j+i2])));
78  }
79  j += nSize;
80  }
81  if (fMutated) {
82  *fMutated = mutated;
83  }
84  return (vMerkleTree.empty() ? uint256() : vMerkleTree.back());
85 }

87 std::vector<uint256> CBlock::GetMerkleBranch(int nIndex) const
88 {
89  if (vMerkleTree.empty())
90  BuildMerkleTree();
91  std::vector<uint256> vMerkleBranch;
92  int j = 0;
93  for (int nSize = vtx.size(); nSize > 1; nSize = (nSize + 1) / 2)
94  {
95  int i = std::min(nIndex^1, nSize-1);
96  vMerkleBranch.push_back(vMerkleTree[j+i]);
97  nIndex >>= 1;
98  j += nSize;
99  }
100  return vMerkleBranch;
101 }

103 uint256 CBlock::CheckMerkleBranch(uint256 hash, const std::vector<uint256>& vMerkleBranch, int nIndex)
104 {
105  if (nIndex == -1)
106  return uint256();
107  for (std::vector<uint256>::const_iterator it(vMerkleBranch.begin()); it != vMerkleBranch.end(); ++it)
108  {
109  if (nIndex & 1)
110  hash = Hash(BEGIN(*it), END(*it), BEGIN(hash), END(hash));
111  else
112  hash = Hash(BEGIN(hash), END(hash), BEGIN(*it), END(*it));
113  nIndex >>= 1;
114  }
115  return hash;
