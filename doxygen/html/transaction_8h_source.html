161  bool IsDust(CFeeRate minRelayTxFee) const
162  {
163  // "Dust" is defined in terms of CTransaction::minRelayTxFee, which has units duffs-per-kilobyte.
164  // If you'd pay more than 1/3 in fees to spend something, then we consider it dust.
165  // A typical txout is 34 bytes big, and will need a CTxIn of at least 148 bytes to spend
166  // i.e. total is 148 + 32 = 182 bytes. Default -minrelaytxfee is 10000 duffs per kB
167  // and that means that fee per txout is 182 * 10000 / 1000 = 1820 duffs.
168  // So dust is a txout less than 1820 *3 = 5460 duffs
169  // with default -minrelaytxfee = minRelayTxFee = 10000 duffs per kB.
170  size_t nSize = GetSerializeSize(SER_DISK,0)+148u;
171  return (nValue < 3*minRelayTxFee.GetFee(nSize));
172  }
173 
174  bool IsZerocoinMint() const
175  {
176  return !scriptPubKey.empty() && scriptPubKey.IsZerocoinMint();
177  }
178 
179  friend bool operator==(const CTxOut& a, const CTxOut& b)
180  {
181  return (a.nValue == b.nValue &&
182  a.scriptPubKey == b.scriptPubKey &&
183  a.nRounds == b.nRounds);
184  }
