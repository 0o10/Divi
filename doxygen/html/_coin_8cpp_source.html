70  // we obtain a prime number
71  for(uint32_t attempt = 0; attempt < MAX_COINMINT_ATTEMPTS; attempt++) {
72 
73  // Generate a random serial number in the range 0...{q-1} where
74  // "q" is the order of the commitment group.
75  CBigNum s = CBigNum::randBignum(this->params->coinCommitmentGroup.groupOrder);
76

78  Commitment coin(&params->coinCommitmentGroup, s);

101 void PrivateCoin::mintCoinFast(const CoinDenomination denomination) {
102 
103  // Generate a random serial number in the range 0...{q-1} where
104  // "q" is the order of the commitment group.
105  CBigNum s = CBigNum::randBignum(this->params->coinCommitmentGroup.groupOrder);
106 
107  // Generate a random number "r" in the range 0...{q-1}
108  CBigNum r = CBigNum::randBignum(this->params->coinCommitmentGroup.groupOrder);
109 
110  // Manually compute a Pedersen commitment to the serial number "s" under randomness "r"
111  // C = g^s * h^r mod p
112  CBigNum commitmentValue = this->params->coinCommitmentGroup.g.pow_mod(s, this->params->coinCommitmentGroup.modulus).mul_mod(this->params->coinCommitmentGroup.h.pow_mod(r, this->params->coinCommitmentGroup.modulus), this->params->coinCommitmentGroup.modulus);
113 
114  // Repeat this process up to MAX_COINMINT_ATTEMPTS times until
115  // we obtain a prime number
116  for (uint32_t attempt = 0; attempt < MAX_COINMINT_ATTEMPTS; attempt++) {
117  // First verify that the commitment is a prime number
118  // in the appropriate range. If not, we'll throw this coin
119  // away and generate a new one.
120  if (commitmentValue.isPrime(ZEROCOIN_MINT_PRIME_PARAM) &&
121  commitmentValue >= params->accumulatorParams.minCoinValue &&
122  commitmentValue <= params->accumulatorParams.maxCoinValue) {
123  // Found a valid coin. Store it.
124  this->serialNumber = s;
125  this->randomness = r;
126  this->publicCoin = PublicCoin(params, commitmentValue, denomination);
127 
128  // Success! We're done.
129  return;
130  }
131 
132  // Generate a new random "r_delta" in 0...{q-1}
133  CBigNum r_delta = CBigNum::randBignum(this->params->coinCommitmentGroup.groupOrder);
134 
135  // The commitment was not prime. Increment "r" and recalculate "C":
136  // r = r + r_delta mod q
137  // C = C * h mod p
138  r = (r + r_delta) % this->params->coinCommitmentGroup.groupOrder;
139  commitmentValue = commitmentValue.mul_mod(this->params->coinCommitmentGroup.h.pow_mod(r_delta, this->params->coinCommitmentGroup.modulus), this->params->coinCommitmentGroup.modulus);
140  }
141 
142  // We only get here if we did not find a coin within
143  // MAX_COINMINT_ATTEMPTS. Throw an exception.
144  throw std::runtime_error("Unable to mint a new Zerocoin (too many attempts)");
145 }
