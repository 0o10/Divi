270  Object ret;

348 CScript _createmultisig_redeemScript(const Array& params)
349 {
350  int nRequired = params[0].get_int();
351  const Array& keys = params[1].get_array();
352 
353  // Gather public keys
354  if (nRequired < 1)
355  throw runtime_error("a multisignature address must require at least one key to redeem");
356  if ((int)keys.size() < nRequired)
357  throw runtime_error(
358  strprintf("not enough keys supplied "
359  "(got %u keys, but need at least %d to redeem)",
360  keys.size(), nRequired));
361  if (keys.size() > 16)
362  throw runtime_error("Number of addresses involved in the multisignature address creation > 16\nReduce the number");
363  std::vector<CPubKey> pubkeys;
364  pubkeys.resize(keys.size());
365  for (unsigned int i = 0; i < keys.size(); i++) {
366  const std::string& ks = keys[i].get_str();
367 #ifdef ENABLE_WALLET
368  // Case 1: DIVI address and we have full public key:
369  CBitcoinAddress address(ks);
370  if (pwalletMain && address.IsValid()) {
371  CKeyID keyID;
372  if (!address.GetKeyID(keyID))
373  throw runtime_error(
374  strprintf("%s does not refer to a key", ks));
375  CPubKey vchPubKey;
376  if (!pwalletMain->GetPubKey(keyID, vchPubKey))
377  throw runtime_error(
378  strprintf("no full public key for address %s", ks));
379  if (!vchPubKey.IsFullyValid())
380  throw runtime_error(" Invalid public key: " + ks);
381  pubkeys[i] = vchPubKey;
382  }
383 
384  // Case 2: hex public key
385  else
386 #endif
387  if (IsHex(ks)) {
388  CPubKey vchPubKey(ParseHex(ks));
389  if (!vchPubKey.IsFullyValid())
390  throw runtime_error(" Invalid public key: " + ks);
391  pubkeys[i] = vchPubKey;
392  } else {
393  throw runtime_error(" Invalid public key: " + ks);
394  }
395  }
396  CScript result = GetScriptForMultisig(nRequired, pubkeys);
397 
398  if (result.size() > MAX_SCRIPT_ELEMENT_SIZE)
399  throw runtime_error(
400  strprintf("redeemScript exceeds size limit: %d > %d", result.size(), MAX_SCRIPT_ELEMENT_SIZE));
401 
402  return result;
403 }

445 Value verifymessage(const Array& params, bool fHelp)
446 {
447  if (fHelp || params.size() != 3)
448  throw runtime_error(
449  "verifymessage \"diviaddress\" \"signature\" \"message\"\n"
450  "\nVerify a signed message\n"
451  "\nArguments:\n"
452  "1. \"diviaddress\" (string, required) The divi address to use for the signature.\n"
453  "2. \"signature\" (string, required) The signature provided by the signer in base 64 encoding (see signmessage).\n"
454  "3. \"message\" (string, required) The message that was signed.\n"
455  "\nResult:\n"
456  "true|false (boolean) If the signature is verified or not.\n"
457  "\nExamples:\n"
458  "\nUnlock the wallet for 30 seconds\n" +
459  HelpExampleCli("walletpassphrase", "\"mypassphrase\" 30") +
460  "\nCreate the signature\n" + HelpExampleCli("signmessage", "\"1D1ZrZNe3JUo7ZycKEYQQiQAWd9y54F4XZ\" \"my message\"") +
461  "\nVerify the signature\n" + HelpExampleCli("verifymessage", "\"1D1ZrZNe3JUo7ZycKEYQQiQAWd9y54F4XZ\" \"signature\" \"my message\"") +
462  "\nAs json rpc\n" + HelpExampleRpc("verifymessage", "\"1D1ZrZNe3JUo7ZycKEYQQiQAWd9y54F4XZ\", \"signature\", \"my message\""));
463 
464  string strAddress = params[0].get_str();
465  string strSign = params[1].get_str();
466  string strMessage = params[2].get_str();
467 
468  CBitcoinAddress addr(strAddress);
469  if (!addr.IsValid())
470  throw JSONRPCError(RPC_TYPE_ERROR, "Invalid address");
471 
472  CKeyID keyID;
473  if (!addr.GetKeyID(keyID))
474  throw JSONRPCError(RPC_TYPE_ERROR, "Address does not refer to key");
475 
476  bool fInvalid = false;
477  vector<unsigned char> vchSig = DecodeBase64(strSign.c_str(), &fInvalid);
478 
479  if (fInvalid)
480  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Malformed base64 encoding");
481 
482  CHashWriter ss(SER_GETHASH, 0);
483  ss << strMessageMagic;
484  ss << strMessage;
485 
486  CPubKey pubkey;
487  if (!pubkey.RecoverCompact(ss.GetHash(), vchSig))
488  return false;
489 
490  return (pubkey.GetID() == keyID);
491 }
